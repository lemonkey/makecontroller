00082 /** \defgroup Servo
00083 * The Servo Motor subsystem controls speed and position control for up to 4 standard servo motors.
00084 * Up to 4 standard servo motors can be controlled with the MAKE Application Board.\n\n
00085 * Standard servos have a range of motion of approximately 180 degrees, although this varies from motor to motor.
00086 * Be sure to plug in the connector with the correct orientation with regard to the GND/5V signals on the board.\n\n
00087 * Set the position of each motor by specifying a value from 0-1023, which will be mapped to the full range of motion
00088 * of the motor.  You can also specify the speed with which the motors will respond to new position commands - a high
00089 * value will result in an immediate response, while a lower value can offer some smoothing when appropriate.
00090 * \ingroup AppBoard
00091 * @{
00092 */

00125 /** 
00126   Set the position of the specified servo motor.
00127   Most servos move within a 180 degree range.  The 0 - 1023 range of the position parameter represents
00128   the full range of motion of the motor.
00129   @param index An integer specifying which servo (0 - 3).
00130   @param position An integer specifying the servo position (0 - 1023).
00131   @return status (0 = OK).
00132 */
00133 int Servo_SetPosition( int index, int position )
00134 {
00135   if ( index < 0 || index >= SERVO_COUNT )
00136     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00137 
00138   if ( Servo.control[ index ].users < 1 )
00139   {
00140     int status = Servo_Start( index );
00141     if ( status != CONTROLLER_OK )
00142       return status;
00143   }
00144 
00145   if ( position < 0 )
00146     position = 0;
00147   if ( position > 1023 )
00148     position = 1023;
00149 
00150   DisableFIQFromThumb();
00151   Servo.control[ index ].positionRequested = position << 6;
00152   EnableFIQFromThumb();
00153 
00154   return CONTROLLER_OK;
00155 }

00235     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00236

00240     if ( Servo.users++ == 0 )

00242       int status = Servo_Init();

00276 int Servo_Stop( int index )
00277 {
00278   if ( index < 0 || index >= SERVO_COUNT )
00279     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00280 
00281   ServoControl* sc = &Servo.control[ index ]; 
00282 
00283   if ( sc->users <= 0 )
00284     return CONTROLLER_ERROR_TOO_MANY_STOPS;

00357 void Servo_IRQCallback( int id )
00358 {
00359   int period;
00360 
00361   switch ( Servo.state )
00362   {
00363     case 0:
00364     {
00365       if ( ++Servo.index >= SERVO_COUNT || Servo.index < 0 )
00366         Servo.index = 0;
00367       ServoControl* s = &Servo.control[ Servo.index ];
00368       
00369       if ( s->position != s->positionRequested )
00370       {
00371         if ( s->speed == -1 )
00372           s->position = s->positionRequested;
00373         else
00374         {
00375           int diff = s->positionRequested - s->position;
00376           if ( diff < 0 )
00377           {
00378             s->position -= s->speed;
00379             if ( s->position < s->positionRequested )
00380               s->position = s->positionRequested;
00381           }
00382           else
00383           {
00384             s->position += s->speed;
00385             if ( s->position > s->positionRequested )
00386               s->position = s->positionRequested;
00387           }
00388         }
00389       }
00390 
00391       period = s->position >> 6;
00392       if ( period >= 0 && period <= 1023 )
00393       {
00394         s->pIoBase->PIO_CODR = s->pin;
00395       }
00396       else
00397         period = 1023;
00398       FastTimer_SetTime( &Servo.fastTimerEntry, period + 988 );
00399       Servo.state = 1;
00400       break;
00401     }
00402     case 1:
00403     {
00404       ServoControl* s = &Servo.control[ Servo.index ];
00405       period = s->position >> 6;
00406       s->pIoBase->PIO_SODR = s->pin;
00407       FastTimer_SetTime( &Servo.fastTimerEntry, Servo.gap + ( 1023 - period ) );
00408       Servo.state = 0;
00409       break;
00410     }
00411   }
00412 }

00416 /** \defgroup ServoOSC Servo - OSC
00417   Control Servo motors with the Application Board via OSC.
00418   \ingroup OSC
00419   
00420   \section devices Devices
00421   There are 4 Servo controllers available on the Application Board, numbered 0 - 3.\n
00422   See the servo section in the Application Board user's guide for more information
00423   on hooking up servos to the board.
00424   
00425   \section properties Properties
00426   Each servo controller has three properties - 'position', 'speed', and 'active'.
00427 
00428   \par Position
00429   The 'position' property corresponds to the position of the servo motor within its range of motion.
00430   This value can be both read and written.  The relevant range of values is from 0 - 1023.  
00431   A position of 0 sets the servo to one extreme of its range of motion and 1023 to its opposite extreme.
00432   \par
00433   To set the first servo to one quarter its position, send the message
00434   \verbatim /servo/0/position 256 \endverbatim
00435   Leave the argument value off to read the position of the servo:
00436   \verbatim /servo/0/position \endverbatim
00437   
00438   \par Speed
00439   The 'speed' property corresponds to the speed with which the servo responds to changes 
00440   of position.
00441   This value can be both read and written, and the range of values is 0 - 1023.  A speed of 1023
00442   means the servo responds immediately, and lower values will result in slower, and often smoother, 
00443   responses.  
00444   \par
00445   To set the speed of the first servo to just under full speed, send a message like
00446   \verbatim /servo/0/speed 975 \endverbatim
00447   Adjust the argument value to one that suits your application.\n
00448   Leave the argument value off to read the position of the servo:
00449   \verbatim /servo/0/speed \endverbatim
00450   
00451   \par Active
00452   The 'active' property corresponds to the active state of the servo.
00453   If the servo is set to be active, no other tasks will be able to
00454   write to the same I/O lines.  If you're not seeing appropriate
00455   responses to your messages to a servo, check the whether it's 
00456   locked by sending a message like
00457   \verbatim /servo/3/state \endverbatim
00458 */
