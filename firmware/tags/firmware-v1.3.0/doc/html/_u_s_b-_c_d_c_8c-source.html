00131 void vUSBCDCTask( void *pvParameters )
00132 {
00133 xISRStatus *pxMessage;
00134 unsigned portLONG ulStatus;
00135 unsigned portLONG ulRxBytes;
00136 
00137   ( void ) pvParameters;
00138 
00139   /* Disconnect USB device from hub.  For debugging - causes host to register reset */
00140   portENTER_CRITICAL();
00141      vDetachUSBInterface();
00142   portEXIT_CRITICAL();
00143   
00144   vTaskDelay( portTICK_RATE_MS * 60 );
00145 
00146   /* Init USB interface */
00147   portENTER_CRITICAL();
00148     vInitUSBInterface();
00149   portEXIT_CRITICAL();
00150   
00151   /* Main task loop.  Process incoming endpoint 0 interrupts, handle data transfers. */
00152    
00153   /* MAKINGTHINGS: USB RUNNING */
00154   Usb_Running = 1;
00155 
00156   for( ;; )
00157   {
00158     /* Look for data coming from the ISR. */
00159     if( xQueueReceive( xUSBInterruptQueue, &pxMessage, usbSHORTEST_DELAY ) )
00160     {
00161       if( pxMessage->ulISR & AT91C_UDP_EPINT0 )
00162       {
00163         /* All endpoint 0 interrupts are handled here. */
00164         prvProcessEndPoint0Interrupt( pxMessage );
00165       }
00166 
00167       if( pxMessage->ulISR & AT91C_UDP_ENDBUSRES )
00168       {
00169         /* End of bus reset - reset the endpoints and de-configure. */
00170         prvResetEndPoints();    
00171       }
00172     }
00173     
00174     /* See if we're ready to send and receive data. */
00175     if( eDriverState == eREADY_TO_SEND && ucControlState ) 
00176     {
00177       if( ( !(AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] & AT91C_UDP_TXPKTRDY) ) && uxQueueMessagesWaiting( xTxCDC ) )
00178       {
00179         xBULKBUFFER Queue2USB;
00180 
00181         if( xQueueReceive( xTxCDC, &Queue2USB, 0 ) ) 
00182         {
00183           unsigned char out = 0;
00184 
00185           while (out < Queue2USB.Count)
00186             AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_2 ] = Queue2USB.Data[out++];
00187         }
00188         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] |= AT91C_UDP_TXPKTRDY;
00189       }
00190 
00191       /* Check for incoming data (host-to-device) on endpoint 1. */
00192       while( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] & (AT91C_UDP_RX_DATA_BK0 | AT91C_UDP_RX_DATA_BK1) )
00193       {
00194         //if (!OscBusy)
00195         //{
00196         /* Only process FIFO if there's room to store it in the queue */
00197         if ( USB_CDC_QUEUE_SIZE > uxQueueMessagesWaiting( xRxCDC ))
00198         { 
00199           xBULKBUFFER USB2Queue;
00200 
00201           ulRxBytes = (AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] >> 16) & usbRX_COUNT_MASK;
00202           // collect however many bytes are in endpoint ... may be a whole osc packet
00203           // may just be a fragment.
00204           USB2Queue.Count = 0;
00205           while( ulRxBytes-- )
00206             USB2Queue.Data[USB2Queue.Count++] = AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_1 ];
00207 
00208           // put it in the queue
00209           xQueueSend( xRxCDC, &USB2Queue, 0 );
00210 
00211           /* Release the FIFO */
00212           portENTER_CRITICAL();
00213           {
00214             ulStatus = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ];
00215             usbCSR_CLEAR_BIT( &ulStatus, uiCurrentBank );
00216             AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] = ulStatus;
00217           }
00218           portEXIT_CRITICAL();
00219 
00220           /* Re-enable endpoint 1's interrupts */
00221           AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT1;
00222         
00223           /* Update the current bank in use */
00224           if( uiCurrentBank == AT91C_UDP_RX_DATA_BK0 ) 
00225             uiCurrentBank = AT91C_UDP_RX_DATA_BK1;
00226           else 
00227             uiCurrentBank = AT91C_UDP_RX_DATA_BK0;
00228 
00229         }
00230         else 
00231           break;
00232         //}
00233       }
00234     }
00235   }
00236 }

00364 static void prvProcessEndPoint0Interrupt( xISRStatus *pxMessage )
00365 {
00366 static xUSB_REQUEST xRequest;
00367 unsigned portLONG ulRxBytes;
00368 
00369   /* Get number of bytes received, if any */
00370   ulRxBytes = pxMessage->ulCSR0 >> 16;
00371   ulRxBytes &= usbRX_COUNT_MASK;
00372 
00373   if( pxMessage->ulCSR0 & AT91C_UDP_TXCOMP )
00374   {
00375     /* We received a TX complete interrupt.  What we do depends on
00376     what we sent to get this interrupt. */
00377 
00378     if( eDriverState == eJUST_GOT_CONFIG )
00379     {
00380       /* We sent an acknowledgement of a SET_CONFIG request.  We
00381       are now at the end of the enumeration.
00382       
00383       TODO: Config 0 sets unconfigured state, should enter Address state.
00384       Request for unsupported config should stall. */
00385       AT91C_BASE_UDP->UDP_GLBSTATE = AT91C_UDP_CONFG;
00386       
00387       /* Set up endpoints */
00388       portENTER_CRITICAL();
00389       {
00390         unsigned portLONG ulTemp;
00391 
00392         /* Set endpoint 1 to bulk-out */
00393         ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ];         
00394         usbCSR_SET_BIT( &ulTemp, AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_BULK_OUT );
00395         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] = ulTemp;   
00396         AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT1;
00397         /* Set endpoint 2 to bulk-in */
00398         ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ];         
00399         usbCSR_SET_BIT( &ulTemp, AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_BULK_IN );
00400         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] = ulTemp;   
00401         AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT2;
00402           /* Set endpoint 3 to interrupt-in, enable it, and enable interrupts */
00403         ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_3 ];         
00404         usbCSR_SET_BIT( &ulTemp, AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_INT_IN );
00405         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_3 ] = ulTemp;   
00406         /*AT91F_UDP_EnableIt( AT91C_BASE_UDP, AT91C_UDP_EPINT3 );        */
00407       }
00408       portEXIT_CRITICAL();
00409 
00410       eDriverState = eREADY_TO_SEND;
00411     }   
00412     else if( eDriverState == eJUST_GOT_ADDRESS )
00413     {
00414       /* We sent an acknowledgement of a SET_ADDRESS request.  Move
00415       to the addressed state. */
00416       if( ulReceivedAddress != ( unsigned portLONG ) 0 )
00417       {     
00418         AT91C_BASE_UDP->UDP_GLBSTATE = AT91C_UDP_FADDEN;
00419       }
00420       else
00421       {
00422         AT91C_BASE_UDP->UDP_GLBSTATE = 0;
00423       }     
00424 
00425       AT91C_BASE_UDP->UDP_FADDR = ( AT91C_UDP_FEN | ulReceivedAddress );    
00426       eDriverState = eNOTHING;
00427     }
00428     else
00429     {   
00430       /* The TXCOMP was not for any special type of transmission.  See
00431       if there is any more data to send. */
00432       prvSendNextSegment();
00433     }
00434   }
00435 
00436   if( pxMessage->ulCSR0 & AT91C_UDP_RX_DATA_BK0 )
00437   {
00438     /* Received a control data packet.  May be a 0-length ACK or a data stage. */
00439     unsigned portCHAR ucBytesToGet;
00440    
00441     /* Got data.  Cancel any outgoing data. */
00442     pxControlTx.ulNextCharIndex = pxControlTx.ulTotalDataLength;
00443     
00444      /* Determine how many bytes we need to receive. */
00445     ucBytesToGet = pxControlRx.ulTotalDataLength - pxControlRx.ulNextCharIndex;
00446     if( ucBytesToGet > ulRxBytes ) 
00447     { 
00448       ucBytesToGet = ulRxBytes;
00449     }
00450 
00451     /* If we're not expecting any data, it's an ack - just quit now. */
00452     if( !ucBytesToGet )
00453     {
00454        return;
00455     }
00456 
00457     /* Get the required data and update the index. */
00458     memcpy( pxControlRx.ucBuffer, pxMessage->ucFifoData, ucBytesToGet );
00459     pxControlRx.ulNextCharIndex += ucBytesToGet;  
00460   }
00461 
00462   if( pxMessage->ulCSR0 & AT91C_UDP_RXSETUP )
00463   {
00464     /* Received a SETUP packet.  May be followed by data packets. */
00465 
00466     if( ulRxBytes >= usbEXPECTED_NUMBER_OF_BYTES )
00467     {       
00468       /* Create an xUSB_REQUEST variable from the raw bytes array. */
00469 
00470       xRequest.ucReqType = pxMessage->ucFifoData[ usbREQUEST_TYPE_INDEX ];
00471       xRequest.ucRequest = pxMessage->ucFifoData[ usbREQUEST_INDEX ];
00472 
00473       xRequest.usValue = pxMessage->ucFifoData[ usbVALUE_HIGH_BYTE ];
00474       xRequest.usValue <<= 8;
00475       xRequest.usValue |= pxMessage->ucFifoData[ usbVALUE_LOW_BYTE ];
00476             
00477       xRequest.usIndex = pxMessage->ucFifoData[ usbINDEX_HIGH_BYTE ];
00478       xRequest.usIndex <<= 8;
00479       xRequest.usIndex |= pxMessage->ucFifoData[ usbINDEX_LOW_BYTE ];
00480       
00481       xRequest.usLength = pxMessage->ucFifoData[ usbLENGTH_HIGH_BYTE ];
00482       xRequest.usLength <<= 8;
00483       xRequest.usLength |= pxMessage->ucFifoData[ usbLENGTH_LOW_BYTE ];
00484 
00485       pxControlRx.ulNextCharIndex = 0;
00486       if( ! (xRequest.ucReqType & 0x80) ) /* Host-to-Device transfer, may need to get data first */
00487       {
00488         if( xRequest.usLength > usbMAX_CONTROL_MESSAGE_SIZE )
00489         { 
00490           /* Too big!  No space for control data, stall and abort. */
00491           prvSendStall();
00492           return;
00493         }
00494 
00495         pxControlRx.ulTotalDataLength = xRequest.usLength;
00496       }
00497       else
00498       {
00499         /* We're sending the data, don't wait for any. */
00500         pxControlRx.ulTotalDataLength = 0; 
00501       }
00502     }
00503   }
00504 
00505   /* See if we've got a pending request and all its associated data ready */
00506   if( ( pxMessage->ulCSR0 & ( AT91C_UDP_RX_DATA_BK0 | AT91C_UDP_RXSETUP ) ) 
00507     && ( pxControlRx.ulNextCharIndex >= pxControlRx.ulTotalDataLength ) )
00508   {
00509     unsigned portCHAR ucRequest;
00510 
00511     /* Manipulate the ucRequestType and the ucRequest parameters to 
00512     generate a zero based request selection.  This is just done to 
00513     break up the requests into subsections for clarity.  The 
00514     alternative would be to have more huge switch statement that would
00515     be difficult to optimise. */
00516     ucRequest = ( ( xRequest.ucReqType & 0x60 ) >> 3 );
00517     ucRequest |= ( xRequest.ucReqType & 0x03 );
00518       
00519     switch( ucRequest )
00520     {
00521       case usbSTANDARD_DEVICE_REQUEST:  
00522         /* Standard Device request */
00523         prvHandleStandardDeviceRequest( &xRequest );
00524         break;
00525 
00526       case usbSTANDARD_INTERFACE_REQUEST: 
00527         /* Standard Interface request */
00528         prvHandleStandardInterfaceRequest( &xRequest );
00529         break;
00530 
00531       case usbSTANDARD_END_POINT_REQUEST: 
00532         /* Standard Endpoint request */
00533         prvHandleStandardEndPointRequest( &xRequest );
00534         break;
00535 
00536       case usbCLASS_INTERFACE_REQUEST:  
00537         /* Class Interface request */
00538         prvHandleClassInterfaceRequest( &xRequest );
00539         break;
00540 
00541       default:  /* This is not something we want to respond to. */
00542         prvSendStall(); 
00543     }
00544   }
00545 }

00548 static void prvGetStandardDeviceDescriptor( xUSB_REQUEST *pxRequest )
00549 {
00550   /* The type is in the high byte.  Return whatever has been requested. */
00551   switch( ( pxRequest->usValue & 0xff00 ) >> 8 )
00552   {
00553     case usbDESCRIPTOR_TYPE_DEVICE:
00554       prvSendControlData( ( unsigned portCHAR * ) &pxDeviceDescriptor, pxRequest->usLength, sizeof( pxDeviceDescriptor ), pdTRUE );
00555       break;
00556 
00557     case usbDESCRIPTOR_TYPE_CONFIGURATION:
00558       prvSendControlData( ( unsigned portCHAR * ) &( pxConfigDescriptor ), pxRequest->usLength, sizeof( pxConfigDescriptor ), pdTRUE );
00559       break;
00560 
00561     case usbDESCRIPTOR_TYPE_STRING:
00562 
00563       /* The index to the string descriptor is the lower byte. */
00564       switch( pxRequest->usValue & 0xff )
00565       {     
00566         case usbLANGUAGE_STRING:
00567           prvSendControlData( ( unsigned portCHAR * ) &pxLanguageStringDescriptor, pxRequest->usLength, sizeof(pxLanguageStringDescriptor), pdTRUE );
00568           break;
00569 
00570         case usbMANUFACTURER_STRING:
00571           prvSendControlData( ( unsigned portCHAR * ) &pxManufacturerStringDescriptor, pxRequest->usLength, sizeof( pxManufacturerStringDescriptor ), pdTRUE );
00572           break;
00573 
00574         case usbPRODUCT_STRING:
00575           prvSendControlData( ( unsigned portCHAR * ) &pxProductStringDescriptor, pxRequest->usLength, sizeof( pxProductStringDescriptor ), pdTRUE );
00576           break;
00577 
00578         case usbCONFIGURATION_STRING:
00579           prvSendControlData( ( unsigned portCHAR * ) &pxConfigurationStringDescriptor, pxRequest->usLength, sizeof( pxConfigurationStringDescriptor ), pdTRUE );
00580           break;
00581 
00582         case usbINTERFACE_STRING:
00583           prvSendControlData( ( unsigned portCHAR * ) &pxInterfaceStringDescriptor, pxRequest->usLength, sizeof( pxInterfaceStringDescriptor ), pdTRUE );
00584           break;
00585 
00586         default:
00587           prvSendStall();
00588           break;
00589       }
00590       break;
00591 
00592     default:
00593       prvSendStall();
00594       break;
00595   }
00596 }

00600 {
00601 unsigned portSHORT usStatus = 0;
00602 
00603   switch( pxRequest->ucRequest )
00604   {
00605     case usbGET_STATUS_REQUEST:
00606       /* Just send two byte dummy status. */
00607       prvSendControlData( ( unsigned portCHAR * ) &usStatus, sizeof( usStatus ), sizeof( usStatus ), pdFALSE );
00608       break;
00609 
00610     case usbGET_DESCRIPTOR_REQUEST:
00611       /* Send device descriptor */
00612       prvGetStandardDeviceDescriptor( pxRequest );
00613       break;
00614 
00615     case usbGET_CONFIGURATION_REQUEST:
00616       /* Send selected device configuration */
00617       prvSendControlData( ( unsigned portCHAR * ) &ucUSBConfig, sizeof( ucUSBConfig ), sizeof( ucUSBConfig ), pdFALSE );
00618       break;
00619 
00620     case usbSET_FEATURE_REQUEST:
00621       prvSendZLP();
00622       break;
00623 
00624     case usbSET_ADDRESS_REQUEST:      
00625       /* Get assigned address and send ack, but don't implement new address until we get a TXCOMP */
00626       prvSendZLP();     
00627       eDriverState = eJUST_GOT_ADDRESS;     
00628       ulReceivedAddress = ( unsigned portLONG ) pxRequest->usValue;
00629       break;
00630 
00631     case usbSET_CONFIGURATION_REQUEST:
00632       /* Ack SET_CONFIGURATION request, but don't implement until TXCOMP */
00633       ucUSBConfig = ( unsigned portCHAR ) ( pxRequest->usValue & 0xff );
00634       eDriverState = eJUST_GOT_CONFIG;
00635       prvSendZLP();
00636       break;
00637 
00638     default:
00639       /* Any unsupported request results in a STALL response. */
00640       prvSendStall();
00641       break;
00642   }
00643 }

00851 static void prvSendControlData( unsigned portCHAR *pucData, unsigned portSHORT usRequestedLength, unsigned portLONG ulLengthToSend, portLONG lSendingDescriptor )
00852 {
00853   if( ( ( unsigned portLONG ) usRequestedLength < ulLengthToSend ) )
00854   {
00855     /* Cap the data length to that requested. */
00856     ulLengthToSend = ( unsigned portSHORT ) usRequestedLength;
00857   }
00858   else if( ( ulLengthToSend < ( unsigned portLONG ) usRequestedLength ) && lSendingDescriptor )
00859   {
00860     /* We are sending a descriptor.  If the descriptor is an exact 
00861     multiple of the FIFO length then it will have to be terminated
00862     with a NULL packet.  Set the state to indicate this if
00863     necessary. */
00864     if( ( ulLengthToSend % usbFIFO_LENGTH ) == 0 )
00865     {
00866       eDriverState = eSENDING_EVEN_DESCRIPTOR;
00867     }
00868   }
00869 
00870   /* Here we assume that the previous message has been sent.  THERE IS NO
00871   BUFFER OVERFLOW PROTECTION HERE.
00872 
00873   Copy the data to send into the buffer as we cannot send it all at once
00874   (if it is greater than 8 bytes in length). */
00875   memcpy( pxControlTx.ucBuffer, pucData, ulLengthToSend );
00876 
00877   /* Reinitialise the buffer index so we start sending from the start of 
00878   the data. */
00879   pxControlTx.ulTotalDataLength = ulLengthToSend;
00880   pxControlTx.ulNextCharIndex = ( unsigned portLONG ) 0;
00881 
00882   /* Send the first 8 bytes now.  The rest will get sent in response to 
00883   TXCOMP interrupts. */
00884   prvSendNextSegment();
00885 }
00886 /*-----------------------------------------------------------*/
00887 
00888 static void prvSendNextSegment( void )
00889 {
00890 volatile unsigned portLONG ulNextLength, ulStatus, ulLengthLeftToSend;
00891 
00892   /* Is there any data to send? */
00893   if( pxControlTx.ulTotalDataLength > pxControlTx.ulNextCharIndex )
00894   {
00895     ulLengthLeftToSend = pxControlTx.ulTotalDataLength - pxControlTx.ulNextCharIndex;
00896   
00897     /* We can only send 8 bytes to the fifo at a time. */
00898     if( ulLengthLeftToSend > usbFIFO_LENGTH )
00899     {
00900       ulNextLength = usbFIFO_LENGTH;
00901     }
00902     else
00903     {
00904       ulNextLength = ulLengthLeftToSend;
00905     }
00906 
00907     /* Wait until we can place data in the fifo.  THERE IS NO TIMEOUT
00908     HERE! */
00909     while( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_TXPKTRDY )
00910     {
00911       vTaskDelay( usbSHORTEST_DELAY );
00912     }
00913 
00914     /* Write the data to the FIFO. */
00915     while( ulNextLength > ( unsigned portLONG ) 0 )
00916     {
00917       AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_0 ] = pxControlTx.ucBuffer[ pxControlTx.ulNextCharIndex ];
00918   
00919       ulNextLength--;
00920       pxControlTx.ulNextCharIndex++;
00921     }
00922   
00923     /* Start the transmission. */
00924     portENTER_CRITICAL();
00925     {
00926       ulStatus = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ];
00927       usbCSR_SET_BIT( &ulStatus, ( ( unsigned portLONG ) 0x10 ) );
00928       AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] = ulStatus;
00929     }
00930     portEXIT_CRITICAL();
00931   }
00932   else
00933   {
00934     /* There is no data to send.  If we were sending a descriptor and the 
00935     descriptor was an exact multiple of the max packet size then we need
00936     to send a null to terminate the transmission. */
00937     if( eDriverState == eSENDING_EVEN_DESCRIPTOR )
00938     {
00939       prvSendZLP();
00940       eDriverState = eNOTHING;
00941     }
00942   }
00943 }
