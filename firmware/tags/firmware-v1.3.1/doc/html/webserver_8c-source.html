00085 /** \defgroup webserver Web Server
00086   Very simple Web Server
00087 
00088   This Web Server implementation is based on the ServerSocket and Socket functions defined
00089   in \ref Sockets.  When started (usually by the Network subsystem), a ServerSocket is opened
00090   and set to listen on port 80.
00091 
00092   When the server receives a request, it checks the incoming address against a list of handlers.
00093   If the address matches, the handler is invoked and checking stops.  Users can add their own
00094   handlers to return custom information.  Handlers persist even when the server is deactivated.
00095 
00096   There are two default handlers.  One is mounted at "/test" and will reply with the message "Test"
00097   and a current hit count.  This handler will only be mounted if no other handlers are present when
00098   the server is started up.  The other handler is a default handler which will match any address not 
00099   already matched.  This is the standard information page showing free memory, running tasks, etc.  This
00100   page will reload every second from "/info".
00101 
00102   \ingroup Controller
00103   @{
00104 */

00168 /**
00169   Adds a route handler to the WebServer.
00170 
00171   This function binds an address fragment (e.g. "/", "/adcs", etc.) to a handler
00172   that will be called when the address matches.  Matching ignores characters in the 
00173   incoming address that are beyond the end of the address specified to this function.
00174   The first function to match the address will receive all the traffic and no more 
00175   handlers will be checked.  Thus if a handler is set up to match "/images" it will match
00176   "/images" and "/images/logo.png" and so on.  Also if there is a subseqent handler 
00177   set to match "/images/diagram" it will never be called since the prior handler
00178   will match the entire "/images" space.
00179 
00180   The handler will be called with the request type specified (usually "GET" or  "PUT"), 
00181   the incoming address ( "/device/0", "/images/logo.png", etc.) A buffer (and max length) is passed in that
00182   can be used to receive the rest of the message if necessary. Then there will be 
00183   the socket which will take the response and a helpful large buffer or specified length
00184   which can be used to build strings.
00185 
00186   At the time the handler is called, only the first line of the request has been read.  It is used
00187   to determing the request type and the address.  If you need to process the request further
00188   its contents may be read into a buffer (the request buffer passed in might be good once the request type 
00189   address have been used since they're in there initially).  It is suggested that you use the 
00190   SocketReadLine( ) function to read a line of the request at a time.
00191 
00192   The handler itself must first write the response using one of WebServer_WriteResponseOkHTML for
00193   sending HTML or WebServer_WriteResponseOkPlain for returning plain text.  
00194   These send the appropriate HTTP header (for example "HTTP/1.0 200 OK\r\nContent-type:text/html\r\n\r\n").
00195   All responses may be simply written to the Socket, but several helpers are provided to assist in the 
00196   construction of simple web pages (for example WebServer_WriteHeader, WebServer_WriteBodyStart, etc.)
00197 
00198   Here is an example handler which is installed when the server is started if there are no handlers already present.
00199   \code
00200 int TestHandler( char* requestType, char* address, char* requestBuffer, int requestMaxSize, void* socket, char* buffer, int len )
00201 {
00202   (void)requestType;
00203   (void)address;
00204   WebServer_WriteResponseOkHTML( socket );
00205   WebServer_WriteHeader( true, socket, buffer, len );
00206   WebServer_WriteBodyStart( address, socket, buffer, len );
00207   snprintf( buffer, len, "<H1>TEST</H1>%d hits", WebServer->hits );
00208   SocketWrite( socket, buffer, strlen( buffer ) );
00209   WebServer_WriteBodyEnd( socket );
00210   return true;
00211 }
00212   \endcode
00213   @param address An string specify the addresses to match.
00214   @param handler pointer to a handler function that will be called when the address is matched.
00215   @return CONTROLLER_OK (=0) on success or the appropriate error if not
00216 */
00217 int WebServer_Route( char* address, int (*handler)( char* requestType, char* address, char* requestBuffer, int requestMaxSize, void* socket, char* buffer, int len )  )
00218 {
00219   if ( WebServerHandlers == NULL )
00220   {
00221     WebServerHandlers = MallocWait( sizeof( WebServerHandlers_ ), 100 );    
00222     if ( WebServerHandlers != NULL )
00223       WebServerHandlers->count = 0;
00224   }
00225   if ( WebServerHandlers != NULL )
00226   {
00227     if ( WebServerHandlers->count < HANDLERS_MAX )
00228     {
00229       WebServerHandler* hp = &WebServerHandlers->handlers[ WebServerHandlers->count++ ];
00230       hp->address = address;
00231       hp->handler = handler;
00232       return CONTROLLER_OK;
00233     }
00234    return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00235   }
00236 
00237   return CONTROLLER_ERROR_NOT_OPEN;
00238 }

00489 char* WebServer_GetRequestAddress( char* request, int length, char** requestType  )
00490 {
00491   char *last = request + length;
00492   *requestType = NULL;
00493   char* address = NULL;
00494 
00495   // Skip any initial spaces
00496   while( *request == ' ' )
00497     request++;
00498 
00499   if ( request > last )
00500     return address;
00501 
00502   *requestType = request;
00503 
00504   // Skip the request type
00505   while ( *request != ' ' )
00506     request++;
00507 
00508   if ( request > last )
00509     return address;
00510 
00511   // Skip any subsequent spaces
00512   while( *request == ' ' )
00513     request++;
00514 
00515   if ( request > last )
00516     return address;
00517 
00518   address = request;
00519 
00520   while( !isspace( *request ) )
00521     request++;
00522   
00523   if ( request < last )
00524     *request = 0;
00525 
00526   return address;
