00070 /** \defgroup Io
00071   A mechanism to manage the 64 parallel IO lines on the controller.
00072 
00073   Control of the SAM7X's IO lines is more complex than most microcontrollers.  All 
00074   lines may be inputs or outputs, have pull-ups or not, have a glitch filter enabled 
00075   or not, and so on.  In addition all IO pins serve at least double and sometimes 
00076   triple duty, being general IO lines and also being IO lines for one or more of 
00077   the controller's many on-chip peripherals.  All these functions can be controlled
00078   from the IO subsystem.
00079 
00080   This API can be used in two ways - either one line at a time or many at a time.  
00081   Functions supporting the latter are suffixed with \b Bits.  So the one-at-a-time 
00082   mechanism for setting an IO to true is called Io_SetTrue().  It takes an index and
00083   the ones for setting a number of lines is Io_SetTrueBits( ).
00084 
00085   The pattern of use is as follows:  prior to using an IO line, Call Io_Start() on it, the 
00086   return value indicates if the start operation was successful.  After this, call any 
00087   of the other Io functions.
00088 
00089   \todo Consider changing the api slightly to permit the immediate use of the io
00090         without starting it.
00091   \todo Add glitch filter control
00092   \todo Implement more bits-style functions
00093   \ingroup Controller
00094   @{
00095 */

00097 int Io_Start( int index, bool lock )
00098 {
00099   if ( index < 0 || index > IO_PIN_COUNT )
00100     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00101   IoPin* p = &Io.pins[ index ];
00102   // If it's not locked at all and 
00103   //    either no lock is requested or no a lock is requested but no other are using the io and
00104   //    there are not a huge number of users
00105   if ( !p->lock && ( !lock || ( lock && p->users == 0 ) ) && ( p->users < IO_USERS_MAX - 2 ) )
00106   {
00107     if ( Io.users == 0 )
00108       Io_Init();
00109    
00110     // Lock it up.
00111     Io.users++;
00112     p->users++;
00113     if ( lock )
00114       p->lock = true;
00115 
00116     // Output Pair 1
00117     // Switch to PIO
00118     // AT91C_BASE_PIOA->PIO_PER = AT91C_PIO_PA26;
00119     // AT91C_BASE_PIOB->PIO_PER = AT91C_PIO_PB23;
00120     
00121     return CONTROLLER_OK;
00122   }
00123   else
00124     return CONTROLLER_ERROR_TOO_MANY_USERS;
00125 }

00127 int Io_Stop( int index )
00128 {
00129   if ( index < 0 || index > IO_PIN_COUNT )
00130     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00131   IoPin* p = &Io.pins[ index ];
00132   if ( p->users > 0 )
00133   {
00134     Io.users--;
00135     p->users--;
00136     p->lock = false;
00137 
00138     if ( Io.users == 0 )
00139       Io_Deinit();
00140 
00141     return CONTROLLER_OK;
00142   }
00143   else
00144     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00145 }

00147 int Io_StartBits( longlong bits, bool lock )
00148 {
00149   IoPin* p = &Io.pins[ 0 ];
00150   int i = 0;
00151   int failure = -1;
00152   ullong mask = 1LL;
00153   for ( i = 0; i < 64; i++ )
00154   {
00155     if ( bits & mask )
00156     {
00157       if ( !p->lock && ( !lock || ( lock && p->users == 0 ) ) && p->users < IO_USERS_MAX - 2 )
00158       {
00159         if ( Io.users == 0 )
00160           Io_Init();
00161         if ( lock )
00162           p->lock = true;
00163         p->users++;
00164         Io.users++;
00165       }
00166       else
00167       {
00168         failure = i;
00169         break;
00170       }
00171     }
00172     mask <<= 1;
00173     p++;
00174   }
00175 
00176   if ( failure != -1 )
00177   {
00178     IoPin* p = &Io.pins[ 0 ];
00179     mask = 1LL;
00180     for ( i = 0; i < failure; i++ )
00181     {
00182       if ( bits & mask )
00183       {
00184         if ( lock )
00185           p->lock = false;
00186         p->users--;
00187         Io.users--;
00188         if ( Io.users == 0 )
00189           Io_Deinit();
00190       }
00191       mask <<= 1;
00192       p++;
00193     }
00194     return CONTROLLER_ERROR_CANT_LOCK;
00195   }
00196 
00197   return CONTROLLER_OK;
00198 }

00585 /** \defgroup IoOSC IO - OSC
00586   Control any pin's direction and output from OSC.
00587 
00588   The IO system allows for the manual manipulation of individual signal lines on the Make Controller.
00589   Proceed with care if you're using these at the same time as any of the other subsystems, as most 
00590   other subsystems rely on the IO system internally to maintain control of the appropriate signal lines.
00591   \ingroup OSC
00592   
00593   \section devices Devices
00594   There are 64 IO's on the Make Controller, numbered 0 - 63.
00595   - IOs <b>0 - 31</b> correspond to \b PortA
00596   - IOs <b>32 - 63 </b> correspond to \b PortB.
00597   
00598   \section properties Properties
00599   Each IO has four properties:
00600   - output
00601   - value
00602   - pullup
00603   - active
00604 
00605   \par Output
00606   The \b output property sets whether the IO is an output.
00607   A value of \b 1 makes it an output, and a value of \b 0 makes it an input.
00608   For example, to make pin 4 an output, send a message like
00609   \verbatim /io/4/output 1\endverbatim
00610   Send the message
00611   \verbatim /io/4/output 0\endverbatim
00612   to make it an input.
00613 
00614   \par Pullup
00615   The \b pullup property determines whether a given IOs pullup resistor is enabled.
00616   To enable it on IO 2, send the message
00617   \verbatim /io/2/pullup 1\endverbatim
00618   and to disable it, send the message
00619   \verbatim /io/2/pullup 0\endverbatim
00620 
00621   \par Value
00622   Writing the \b value property sets the on/off value of a given IO.
00623   The IO system only permits digital IO so, we only want to send ones and zeros as values.
00624   For example, to activate pin 4, send a message like
00625   \verbatim /io/4/value 1\endverbatim
00626   Change the argument 1 to a 0 to turn it off.
00627   Reading the \b value property returns the value of a given IO.
00628   For example, to read pin 4, send a message like
00629   \verbatim /io/4/value\endverbatim
00630     
00631   \par Active
00632   The \b active property corresponds to the active state of an IO.
00633   Read whether IO 0 is active by sending the message
00634   \verbatim /io/0/active \endverbatim
00635   You can set the active flag by sending
00636   \verbatim /io/0/active 1 \endverbatim
00637 */
