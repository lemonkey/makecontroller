00094 /** \defgroup Stepper
00095   The Stepper Motor subsystem provides speed and position control for one or two stepper motors.
00096   Up to 2 stepper motors can be controlled with the Make Application Board.
00097   Specify settings for your stepper motor by setting whether it's:
00098   - bipolar or unipolar
00099   - normal of half-stepping
00100 
00101   You can generally use the stepper motor in 2 modes - \b absolute positioning or \b relative positioning.
00102 
00103   For absolute positioning, call Stepper_SetPositionRequested() with the desired position, and the motor will move there.
00104   You can read back the stepper's position at any point along the way to determine where it is at a given moment.  The board
00105   keeps an internal count of how many steps the motor has taken in order to keep track of where it is.
00106 
00107   For relative positioning, use Stepper_Step( ) to simply move a number of steps from the current position.
00108 * \ingroup AppBoard
00109 * @{

00451 int Stepper_Start( int index )
00452 {
00453   static int status; // why does this not seem to work at -O2 when this is not declared as static?
00454 
00455   if ( index < 0 || index >= STEPPER_COUNT )
00456     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00457 
00458   StepperControl* sc = Stepper->control[ index ];
00459   if ( sc->users++ == 0 )
00460   {
00461     status = Pwm_Start( index * 2 );
00462     if ( status != CONTROLLER_OK )
00463     {
00464       sc->users--;
00465       Stepper->users--;
00466       return status;
00467     }
00468 
00469     status = Pwm_Start( index * 2 + 1 );
00470     if ( status != CONTROLLER_OK )
00471     {
00472       Pwm_Stop( index * 2 );
00473       sc->users--;
00474       Stepper->users--;
00475       return status;
00476     }
00477 
00478     // Fire the PWM's up
00479     sc->duty = 1023;
00480 
00481     int pwm = index * 2;
00482     Pwm_Set( pwm, sc->duty );
00483     Pwm_Set( pwm + 1, sc->duty );
00484 
00485     // Get IO's
00486     int i;
00487     for ( i = 0; i < 4; i++ )
00488     {
00489       int io = Stepper_GetIo( index, i );      
00490       sc->io[ i ] = io;
00491         
00492       // Try to lock the stepper output
00493       status = Io_Start( io, true );
00494       if ( status != CONTROLLER_OK )
00495       {
00496         // Damn - unlock any that we did get
00497         int j;
00498         for ( j = 0; j < i; j++ )
00499           Io_Stop( sc->io[ j ] );
00500 
00501         Pwm_Stop( pwm );
00502         Pwm_Stop( pwm + 1 );
00503 
00504         Stepper->users--;
00505         sc->users--;
00506         return status;
00507       }
00508       Io_PioEnable( io );
00509       Io_SetTrue( io );
00510       Io_SetOutput( io );
00511     }
00512 
00513     DisableFIQFromThumb();
00514     sc->position = 0;
00515     sc->positionRequested = 0;
00516     sc->speed = 10;
00517     sc->timerRunning = 0;
00518     sc->halfStep = false;
00519     sc->bipolar = true;
00520     EnableFIQFromThumb();
00521 
00522     FastTimer_InitializeEntry( &sc->fastTimerEntry, Stepper_IRQCallback, index, sc->speed * 1000, true );
00523     Stepper->users++;
00524   }
00525 
00526   return CONTROLLER_OK;
00527 }

00529 int Stepper_Stop( int index )
00530 {
00531   if ( index < 0 || index >= STEPPER_COUNT || Stepper == NULL )
00532     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00533 
00534   if( Stepper->control[ index ] == NULL )
00535     return CONTROLLER_OK;
00536 
00537   StepperControl* s = Stepper->control[ index ]; 
00538 
00539   if ( s->users <= 0 )
00540     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00541 
00542   if ( --s->users == 0 )
00543   {
00544     if ( s->timerRunning )
00545     {
00546       DisableFIQFromThumb();
00547       FastTimer_Cancel( &s->fastTimerEntry );
00548       EnableFIQFromThumb();
00549     }
00550 
00551     int i;
00552     for ( i = 0; i < 4; i++ )
00553     {
00554       int io = s->io[ i ];
00555       Io_SetFalse( io );
00556       Io_Stop( io );
00557     }
00558  
00559     int pwm = index * 2;
00560     Pwm_Stop( pwm );
00561     Pwm_Stop( pwm + 1 );
00562 
00563     Free( s );
00564     s = NULL;
00565 
00566     if ( --Stepper->users == 0 )
00567     {
00568       Free( Stepper );
00569       Stepper = NULL;
00570     }
00571   }
00572 
00573   return CONTROLLER_OK;
00574 }

00640   if ( s->position > s->positionRequested )

00642 
00643   if ( s->bipolar )

00645     if ( s->halfStep )
00646       Stepper_SetBipolarHalfStepOutput( s, s->position );
00647     else
00648       Stepper_SetBipolarOutput( s, s->position );
00649   }
00650   else
00651   {
00652     if ( s->halfStep ) 
00653       Stepper_SetUnipolarHalfStepOutput( s, s->position );
00654     else

00660     FastTimer_Cancel( &s->fastTimerEntry );

00951 /** \defgroup StepperOSC Stepper - OSC
00952   \ingroup OSC
00953   Control Stepper motors with the Application Board via OSC.
00954   Specify settings for your stepper motor by setting whether it's:
00955   - bipolar or unipolar
00956   - normal of half-stepping
00957 
00958   You can generally use the stepper motor in 2 modes - \b absolute positioning or \b relative positioning.
00959 
00960   For absolute positioning, set \b positionrequested with the desired position, and the motor will move there.
00961   You can read back the stepper's \b position property at any point along the way to determine where it is at a given moment.  The board
00962   keeps an internal count of how many steps the motor has taken in order to keep track of where it is.
00963 
00964   For relative positioning, use the \b step property to simply move a number of steps from the current position.
00965   
00966   \section devices Devices
00967   There are 2 Stepper controllers available on the Application Board, numbered 0 & 1.
00968   See the Stepper section in the Application Board user's guide for more information
00969   on hooking steppers up to the board.
00970   
00971   \section properties Properties
00972   Each stepper controller has eight properties:
00973   - position
00974   - positionrequested
00975   - speed
00976   - duty
00977   - bipolar
00978   - halfstep
00979   - step
00980   - active
00981 
00982   \par Step
00983   The \b step property simply tells the motor to take a certain number of steps.
00984   This is a write-only value.
00985   \par
00986   To take 1000 steps with the first stepper, send the message
00987   \verbatim /stepper/0/step 1000\endverbatim
00988   
00989   \par Position
00990   The \b position property corresponds to the current step position of the stepper motor
00991   This value can be both read and written.  Writing this value changes where the motor thinks it is.
00992   The initial value of this parameter is 0.
00993   \par
00994   To set the first stepper to step position 10000, send the message
00995   \verbatim /stepper/0/position 10000\endverbatim
00996   Leave the argument value off to read the position of the stepper:
00997   \verbatim /stepper/0/position \endverbatim
00998 
00999   \par PositionRequested
01000   The \b positionrequested property describes the desired step position of the stepper motor
01001   This value can be both read and written.  Writing this value changes the motor's destination.
01002   \par
01003   To set the first stepper to go to position 10000, send the message
01004   \verbatim /stepper/0/positionrequested 10000\endverbatim
01005   Leave the argument value off to read the last requested position of the stepper:
01006   \verbatim /stepper/0/positionrequested \endverbatim
01007 
01008   \par Speed
01009   The \b speed property corresponds to the speed with which the stepper responds to changes 
01010   of position.  This value is the number of milliseconds between each step.  So, a speed of one
01011   would be a step every millisecond, or 1000 steps a second.  
01012   \par
01013   Note that not all stepper motors can be stepped quite that fast.  If you find your stepper motor acting strangely, 
01014   experiment with slowing down the speed a bit.
01015   \par
01016   To set the speed of the first stepper to step at 100ms per step, send a message like
01017   \verbatim /stepper/0/speed 100 \endverbatim
01018   Adjust the argument value to one that suits your application.\n
01019   Leave the argument value off to read the speed of the stepper:
01020   \verbatim /stepper/0/speed \endverbatim
01021   
01022   \par Duty
01023   The \b duty property corresponds to the how much of the power supply is to be sent to the
01024   stepper.  This is handy for when the stepper is static and not being required to perform too
01025   much work and reducing its power helps reduce heat dissipation.
01026   This value can be both read and written, and the range of values is 0 - 1023.  A duty of 0
01027   means the stepper gets no power, and the value of 1023 means the stepper gets full power.  
01028   \par
01029   To set the duty of the first stepper to 500, send a message like
01030   \verbatim /stepper/0/duty 500 \endverbatim
01031   Adjust the argument value to one that suits your application.\n
01032   Leave the argument value off to read the duty of the stepper:
01033   \verbatim /stepper/0/duty \endverbatim
01034 
01035   \par Bipolar
01036   The \b bipolar property is set to the style of stepper being used.  A value of 1 specifies bipolar
01037   (the default) and 0 specifies a unipolar stepper.
01038   This value can be both read and written.
01039 
01040   \par HalfStep
01041   The \b halfstep property controls whether the stepper is being half stepped or not.  A 0 here implies full stepping
01042   (the default) and 1 implies a half stepping.
01043   This value can be both read and written.
01044 
01045   \par Active
01046   The \b active property corresponds to the active state of the stepper.
01047   If the stepper is set to be active, no other tasks will be able to
01048   write to the same I/O lines.  If you're not seeing appropriate
01049   responses to your messages to a stepper, check the whether it's 
01050   locked by sending a message like
01051   \verbatim /stepper/1/active \endverbatim
01052 */
