00035 /** \defgroup FreeRTOS
00036   The FreeRTOS subsystem provides a real-time operating system (RTOS) for the MAKE Controller.
00037   FreeRTOS is an open source Real-Time Operating System.  It implements a scheduler that 
00038   gives time to "concurrent" tasks according to their given priority.  It allows programmers to 
00039   focus on programming each process on the board separately, letting the RTOS 
00040   determine when each of those tasks should be given processor time.
00041 
00042   Each process on the board is known as a \b task, and each task is generally implemented as a 
00043   continuous loop.  It is important to ensure that each task will block at some point so that
00044   other tasks will have a chance to run - otherwise the Controller will lock up. The easiest way to
00045   do this is with a call to Sleep( ) when the task is not urgent.
00046   
00047   More info at http://www.freertos.org
00048 * \ingroup Controller
00049 * @{
00050 */
00051 
00052 /** 
00053   Put a task to sleep for a given number of milliseconds.
00054   This lets the processor give time to other tasks.  You'll always want to include some amount of Sleep( ) in your tasks
00055   to avoid locking up the Controller.
00056   @param timems An integer specifying how long to sleep in milliseconds.
00057   
00058 \par Example
00059 \code
00060 void MyTask( void *p) // in your task
00061 {
00062   while( 1 )
00063   {
00064     Led_SetState( 0 ); // turn the LED off
00065     Sleep( 900 ); // leave it off for 900 milliseconds
00066     Led_SetState( 1 ); // turn the LED on
00067     Sleep( 10 ); // leave it on for 10 milliseconds
00068   }
00069 }
00070 \endcode
00071 */
00072 void Sleep( int timems )
00073 {
00074   vTaskDelay( timems / portTICK_RATE_MS );
00075 }

00089 /** 
00090   Create a new task.
00091   All tasks have the same signature:
00092   
00093   \code void TaskName( void *parameters ); \endcode
00094   
00095   When creating a new task, be sure that you allocate it enough stack space.  A task must have enough stack space 
00096   to load any of the potential data structures it will be working with, otherwise it will die and the Controller will crash.
00097   
00098   Since FreeRTOS is pre-emptive, higher priority tasks will get processor time before lower priority tasks.  The priority
00099   level is a bit arbitrary, but choose something that makes sense in relation to the other tasks you have running. Always be 
00100   sure to block a task when possible, in order to give processor time to other tasks.
00101   @param taskCode A pointer to the function that will run when the task starts.
00102   @param name A character string containing the name of the task.
00103   @param stackDepth An integer specifying in bytes the size of the task's stack.
00104   @param parameters A void* parameter which will be passed into the task at create time.
00105   @param priority An integer specifying the task's priority (0 lowest - 7 highest priority).
00106   @return A pointer to the handle for the newly created task, or NULL if it didn't succeed.
00107   @see TaskDelete( )
00108   
00109 \par Example
00110 \code
00111 // create the task MyTask, called "Me", with a stack of 1000 bytes, no parameters, at priority 1
00112 void *taskPtr = TaskCreate( MyTask, "Me", 1000, 0, 1 );
00113 
00114 void MyTask( void *p) // in your task
00115 {
00116   // set up any variables that you'll need
00117   int counter = 0;
00118   // then start your continuous loop
00119   while( 1 )
00120   {
00121     counter++;
00122     Sleep( 100 );
00123   }
00124 }
00125 \endcode
00126 */
00127 void* TaskCreate( void (taskCode)( void*), char* name, int stackDepth, void* parameters, int priority )
00128 {
00129   void* taskHandle;
00130   if( xTaskCreate( taskCode, (signed char*)name, ( stackDepth >> 2 ), parameters, priority, &taskHandle ) == 1 )
00131     return taskHandle;
00132   else
00133     return NULL;
00134 }

00193 /** 
00194   Same as Malloc, but keeps trying until it succeeds.
00195   This is a convenience function that will continue to call Malloc( ) at a given interval until
00196   it returns successfully.  Once this function returns, you know you have been allocated the memory
00197   you asked for.
00198   
00199   @param size An integer specifying the amount of memory, in bytes, you want to allocate.
00200   @param interval An integer specifying the number of milliseconds to wait between successive attempts to allocate memory.  
00201   It's best to not set this too low, as it will be a bit taxing on the system.
00202   @return A pointer to the allocated memory, or NULL if the memory was not available.
00203   @see Free(), System_GetFreeMemory( )
00204   
00205 \par Example
00206 \code
00207 char *bufferPtr = MallocWait( 1024, 100 );
00208 // now you have a 1024 character buffer
00209 \endcode
00210 */
00211 void* MallocWait( int size, int interval )
00212 {
00213   void* retval = NULL;
00214   while( retval == NULL )
00215   {
00216     retval = Malloc( size );
00217     if( retval )
00218       return retval;
00219     else
00220       Sleep( interval );
00221   }
00222   return NULL; // should never get here
00223 }

00478 void* TaskGetNext_internal( void* task )
00479 {
00480   int currentID;
00481   int lowID = 1024; // some large number that will be greater than the number of tasks...
00482   int highID = 1024;
00483   void* lowTask = NULL;
00484   void* highTask = NULL;
00485   void* tcb = NULL;
00486   unsigned portBASE_TYPE uxQueue = TaskGetTopPriorityUsed( ) + 1;
00487 
00488   if( task == NULL )
00489     currentID = 1;
00490   else
00491     currentID = TaskGetIDNumber( task );
00492 
00493   // look through all the possible lists of tasks
00494     do
00495     {
00496       uxQueue--;
00497       if( !listLIST_IS_EMPTY( GetReadyTaskByPriority( uxQueue ) ) )
00498       {
00499         tcb = iterateForNextTask( &lowTask, &lowID, &highTask, &highID, currentID, GetReadyTaskByPriority( uxQueue ) );
00500         if( tcb != NULL )
00501           return tcb;
00502       }   
00503     }while( uxQueue > ( unsigned portSHORT ) tskIDLE_PRIORITY );
00504 
00505     // check the list of delayed tasks
00506     if( !listLIST_IS_EMPTY( GetDelayedTaskList( ) ) )
00507     {
00508       tcb = iterateForNextTask( &lowTask, &lowID, &highTask, &highID, currentID, GetDelayedTaskList( ) );
00509       if( tcb != NULL )
00510         return tcb;
00511     }
00512     // check the list of overflow delayed tasks
00513     if( !listLIST_IS_EMPTY( GetOverflowDelayedTaskList( ) ) )
00514     {
00515       tcb = iterateForNextTask( &lowTask, &lowID, &highTask, &highID, currentID, GetOverflowDelayedTaskList( ) );
00516       if( tcb != NULL )
00517         return tcb;
00518     }
00519     // check the list of tasks about to die
00520     if( !listLIST_IS_EMPTY( GetListOfTasksWaitingTermination( ) ) )
00521     {
00522       tcb = iterateForNextTask( &lowTask, &lowID, &highTask, &highID, currentID, GetListOfTasksWaitingTermination( ) );
00523       if( tcb != NULL )
00524         return tcb;
00525     }
00526     // check the list of suspended tasks
00527     if( !listLIST_IS_EMPTY( GetSuspendedTaskList( ) ) )
00528       tcb = iterateForNextTask( &lowTask, &lowID, &highTask, &highID, currentID, GetSuspendedTaskList( ) );
00529 
00530     
00531   if( highTask )
00532     return highTask;
00533   else
00534     return lowTask;
00535 }

00537 // one function to search either by name or ID
00538 // pass in a garbage value for the parameter you're not interested in
00539 void* findTask( char *taskName, int taskID )
00540 {
00541   bool byName;
00542   if( taskName == NULL && taskID > 0 )
00543     byName = false;
00544   if( taskName != NULL && taskID < 0 )
00545     byName = true;
00546 
00547   unsigned portBASE_TYPE uxQueue = TaskGetTopPriorityUsed( ) + 1;
00548   void *tcb = NULL;
00549 
00550   // look through all the possible lists of tasks
00551     do
00552     {
00553       uxQueue--;
00554       if( !listLIST_IS_EMPTY( GetReadyTaskByPriority( uxQueue ) ) )
00555       {
00556         if( byName )
00557           tcb = iterateByName( taskName, GetReadyTaskByPriority( uxQueue ) );
00558         else
00559           tcb = iterateByID( taskID, GetReadyTaskByPriority( uxQueue ) );
00560 
00561         if( tcb != NULL )
00562           return tcb;
00563       }   
00564     }while( uxQueue > ( unsigned portSHORT ) tskIDLE_PRIORITY );
00565 
00566     // check the list of delayed tasks
00567     if( !listLIST_IS_EMPTY( GetDelayedTaskList( ) ) )
00568     {
00569       if( byName)
00570         tcb = iterateByName( taskName, GetDelayedTaskList( ) );
00571       else
00572         tcb = iterateByID( taskID, GetDelayedTaskList( ) );
00573 
00574       if( tcb != NULL )
00575         return tcb;
00576     }
00577     // check the list of overflow delayed tasks
00578     if( !listLIST_IS_EMPTY( GetOverflowDelayedTaskList( ) ) )
00579     {
00580       if( byName )
00581         tcb = iterateByName( taskName, GetOverflowDelayedTaskList( ) );
00582       else
00583         tcb = iterateByID( taskID, GetOverflowDelayedTaskList( ) );
00584       if( tcb != NULL )
00585         return tcb;
00586     }
00587     // check the list of tasks about to die
00588     if( !listLIST_IS_EMPTY( GetListOfTasksWaitingTermination( ) ) )
00589     {
00590       if( byName )
00591         tcb = iterateByName( taskName, GetListOfTasksWaitingTermination( ) );
00592       else
00593         tcb = iterateByID( taskID, GetListOfTasksWaitingTermination( ) );
00594 
00595       if( tcb != NULL )
00596         return tcb;
00597     }
00598     // check the list of suspended tasks
00599     if( !listLIST_IS_EMPTY( GetSuspendedTaskList( ) ) )
00600     {
00601       if( byName )
00602         tcb = iterateByName( taskName, GetSuspendedTaskList( ) );
00603       else
00604         tcb = iterateByID( taskID, GetSuspendedTaskList( ) );
00605     }
00606     return tcb;
00607 }

00610 {
00611   void *pxNextTCB, *pxFirstTCB;
00612   listGET_OWNER_OF_NEXT_ENTRY( pxFirstTCB, pxList );
00613   do
00614   {
00615     listGET_OWNER_OF_NEXT_ENTRY( pxNextTCB, pxList );
00616     if( TaskGetIDNumber( pxNextTCB ) == id )
00617       return pxNextTCB;
00618     
00619   } while( pxNextTCB != pxFirstTCB );
00620   // if we get down here, we didn't find it.
00621   return NULL;

00639 void* iterateForNextTask( void** lowTask, int* lowID, void** highTask, int* highID, int currentID, xList* pxList )
00640 {
00641   void *pxNextTCB, *pxFirstTCB;
00642   listGET_OWNER_OF_NEXT_ENTRY( pxFirstTCB, pxList );
00643   do
00644   {
00645     listGET_OWNER_OF_NEXT_ENTRY( pxNextTCB, pxList );
00646     int id = TaskGetIDNumber( pxNextTCB );
00647     if( id == (currentID + 1) ) // if it's the ID we're looking for plus one, we got it and we're out.
00648       return pxNextTCB;
00649     if( id > currentID && id < *highID )
00650     {
00651       *highID = id;
00652       *highTask = pxNextTCB;
00653     }
00654     if( id < currentID && id < *lowID && id != 1 ) // we don't want id 1 since that's IDLE and there's no real task there.
00655     {
00656       *lowID = id;
00657       *lowTask = pxNextTCB;
