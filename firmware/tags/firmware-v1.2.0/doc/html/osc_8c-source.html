00018 /** @defgroup OSC
00019   Communicate with the Make Controller Kit via OSC.
00020   
00021   \section osc OSC
00022   "Open Sound Control (OSC) is a protocol for communication among computers, sound synthesizers, 
00023   and other multimedia devices that is optimized for modern networking technology."  
00024   With OSC implemented on the Make Controller Kit, it can already talk to 
00025   a wide variety of environments and devices like Java, Max/MSP, Pd, Flash, Processing, SuperCollider,
00026   and many others.
00027   
00028   OSC is based on the notion of \b messages, which are composed of an address, and the data to 
00029   be sent to that address.  The address looks a lot like a URL that you might type into
00030   your internet browser.  Each element in the address is like a directory, with other 
00031   elements inside it, and each element is separated from the next by a slash (/).  Each OSC
00032   message must also start with a slash.  
00033   
00034   \par Example:
00035   For instance, the OSC address
00036   \code /make/controller/kit \endcode
00037   says, "start in the 'make' directory, go down to the 'controller' directory, and 
00038   finally to the 'kit' directory.  
00039   
00040   Any number of \b argument \b values can be sent to that address by including them in the message
00041   after the address.  These values can be integers (ints), floats, or strings.  
00042   
00043   \par Example:
00044   If we wanted to send the value 35.4 to the address above, we would create the message
00045   \code /make/controller/kit 35.4 \endcode
00046   with a space between the address and the data.\n\n
00047   Additional data can be added, each separated by a space
00048   \code /make/controller/kit 35.4 lawn 12 \endcode
00049   
00050   \section osc_mck OSC & the Make Controller Kit
00051   Many devices on the Make Controller Kit can be addressed via OSC.  In sending messages to them,
00052   we need to know the OSC address, and the appropriate argument values to send.
00053   
00054   The Make Controller Kit is orgranized, for OSC, into \b subsystems.  Each subsystem has one or more
00055   \b devices, and each device has one or more \b properties.  To address a particular device, you'll
00056   need to create an OSC message specifying the address in that format: 
00057   \code /subsystem/device/property \endcode
00058   Each of the modules above provide the details for each of the subsystems on the board, along with their
00059   devices and properties.  A simple example is given below.
00060   
00061   \par Example:
00062   To create an OSC message to turn an LED on, first identify the appropriate \b subsystem.
00063   In this case, the subsystem is called \b appled.\n\n
00064   There are 4 LEDs, so we need to specify which one to control.
00065   The LEDs are numbered 0 -3, so choosing the first LED means the \b device value is 0.\n\n
00066   The \b property of the LED that turns it on and off is its 'state'.\n\n
00067   Lastly, we must specify what the state should actually be, by including an \b argument value after the address.
00068   To turn it on, this value should be 1.  0 would turn it off.\n
00069   The complete OSC message looks like
00070   \code /appled/0/state 1 \endcode
00071 */

00481 int Osc_ReceivePacket( int channel, char* packet, int length )
00482 {
00483   // Got a packet.  Unpacket.
00484   int status = -1;
00485   switch ( *packet )
00486   {
00487     case '/':
00488       status = Osc_ReceiveMessage( channel, packet, length );
00489       break;
00490     case '#':
00491       if ( strcmp( packet, "#bundle" ) == 0 )
00492       {
00493         // skip bundle text and timetag
00494         packet += 16;
00495         length -= 16;
00496         while ( length > 0 )
00497         {
00498           // read the length (pretend packet is a pointer to integer)
00499           int messageLength = Osc_EndianSwap( *((int*)packet) );
00500           packet += 4;
00501           length -= 4;
00502           if ( messageLength <= length )
00503             Osc_ReceivePacket( channel, packet, messageLength );
00504           length -= messageLength;
00505           packet += messageLength;
00506         }
00507       }
00508       break;
00509     default:
00510       // Something else?
00511       Osc_CreateMessage( channel, "/error", ",s", "Packet Error" );
00512       break;
00513   }
00514 
00515   return Osc_SendPacket( channel );
00516 }

00518 int Osc_ReceiveMessage( int channel, char* message, int length )
00519 {
00520   // Got a packet.  Unpacket.
00521 
00522   // Confirm it's a message
00523   if ( *message == '/' )
00524   {
00525     if( strlen(message) > (unsigned int)length )
00526       return CONTROLLER_ERROR_BAD_DATA;
00527 
00528     int i;
00529     char* nextChar = message + 1; // first, try to see if it was a "help" query
00530     if( *nextChar == '\0' || *nextChar == ' ' )
00531     {
00532       for ( i = 0; i < Osc->registeredSubsystems; i++ )
00533       {
00534         OscSubsystem* sub = Osc->subsystem[ i ];
00535         Osc_CreateMessage( channel, "/", ",s", sub->name );
00536       }
00537       return CONTROLLER_OK;
00538     }
00539 
00540     char* nextSlash = strchr( message + 1, '/' );
00541     if ( nextSlash != NULL )
00542       *nextSlash = 0;
00543     
00544     int count = 0;
00545     for ( i = 0; i < Osc->registeredSubsystems; i++ )
00546     {
00547       OscSubsystem* sub = Osc->subsystem[ i ];
00548       if ( Osc_PatternMatch( message + 1, sub->name ) )
00549       {
00550         count++;
00551         if ( nextSlash )
00552           (sub->receiveMessage)( channel, nextSlash + 1, length - ( nextSlash - message ) - 1 );
00553         else
00554         {
00555           char* noNextSlash = message + strlen(message);
00556           (sub->receiveMessage)( channel, noNextSlash, 0 );
00557         }
00558       }
00559     }
00560     if ( count == 0 )
00561     {
00562       Osc_LockScratchBuf( Osc->scratch1Semaphore );
00563       snprintf( Osc->scratch1, OSC_SCRATCH_SIZE, "No Subsystem Match - %s", message + 1 );
00564       Osc_CreateMessage( channel, "/error", ",s", Osc->scratch1 );
00565       Osc_UnlockScratchBuf( Osc->scratch1Semaphore );
00566     }
00567   }
00568   else
00569   {
00570     return CONTROLLER_ERROR_BAD_DATA;
00571   }
00572   return CONTROLLER_OK;
00573 }

01366 int Osc_CreateMessage( int channel, char* address, char* format, ... )
01367 {
01368   if ( address == NULL || format == NULL || *format != ',' )
01369     return CONTROLLER_ERROR_BAD_DATA;
01370 
01371   if ( channel < 0 || channel >= OSC_CHANNEL_COUNT )
01372     return CONTROLLER_ERROR_ILLEGAL_INDEX;
01373 
01374   OscChannel* ch = Osc->channel[ channel ];
01375 
01376   if ( !ch->running )
01377     return CONTROLLER_ERROR_SUBSYSTEM_INACTIVE;
01378 
01379   if ( channel == OSC_CHANNEL_UDP && ch->replyAddress == 0 )
01380     return CONTROLLER_ERROR_NO_ADDRESS;
01381 
01382   // Check for sender
01383   if ( ch->sendMessage == NULL )
01384     return CONTROLLER_ERROR_RESOURCE_MISSING;
01385 
01386   if ( Osc_Lock( ch ) != CONTROLLER_OK )
01387     return CONTROLLER_ERROR_LOCK_ERROR;
01388 
01389   if ( ch->bufferPointer == NULL )
01390     Osc_ResetChannel( ch );
01391 
01392   // try to send this message - if there's a problem somewhere, 
01393   // send the existing buffer - freeing up space, then try (once) again.
01394   int count = 0;
01395   char *bp;
01396   do
01397   {  
01398     count++;
01399 
01400     char* buffer = ch->bufferPointer;
01401     int length = ch->bufferRemaining;
01402   
01403     bp = buffer;
01404   
01405     // First message in the buffer?
01406     if ( bp == ch->buffer )
01407     {
01408       bp = Osc_CreateBundle( bp, &length, 0, 0 );
01409       if ( bp == NULL )
01410         return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
01411     }
01412   
01413     // Make room for the new message
01414     int* lp = (int *)bp;
01415     bp += 4;
01416     length -= 4;
01417 
01418     // remember the start of the message
01419     char* mp = bp;    
01420 
01421     if ( length > 0 )
01422     {      
01423       // Set up to iterate through the arguments
01424       va_list args;
01425       va_start( args, format );
01426     
01427       bp = Osc_CreateMessageInternal( bp, &length, address, format, args ); 
01428 
01429       //va_end( args );
01430     }
01431     else
01432       bp = 0;
01433       
01434     if ( bp != 0 )
01435     {
01436       // Set the size
01437       *lp = Osc_EndianSwap( bp - mp ); 
01438   
01439       ch->bufferPointer = bp;
01440       ch->bufferRemaining = length;
01441       ch->messages++;
01442     }
01443     else
01444     {
01445       Osc_SendPacketInternal( ch );
01446     }
01447   } while ( bp == 0 && count == 1 );
01448 
01449   Osc_Unlock( ch );
01450 
01451   return ( bp != 0 ) ? CONTROLLER_OK : CONTROLLER_ERROR_ILLEGAL_PARAMETER_VALUE;
01452 }
