00047 /** \defgroup serial Serial
00048   Serial provides a way to send and receive data via the serial port.
00049 
00050   The subsystem is supplied with small input and output buffers (of 100 characters each) and at present
00051   the implementation is interrupt per character so it's not particularly fast.
00052 
00053   Permits all of the common serial characteristics to be set including:
00054   - baud - the speed of the connection (110 - > 2M) in baud or raw bits per second.  9600 baud is the default setting.
00055   - bits - the size of each character (5 - 8).  8 bits is the default setting.
00056   - stopBits - the number of stop bits transmitted (1 or 2)  1 stop bit is the default setting.
00057   - parity - the parity policy (-1 is odd, 0 is none and 1 is even).  Even is the default setting.
00058   - hardwareHandshake - whether hardware handshaking is used or not.  HardwareHandshaking is off by default.
00059 
00060   \todo Need to complete support for the Hardware Handshaking
00061   \todo Convert to DMA interface for higher performance.
00062 
00063   \ingroup Controller
00064   @{
00065 */

00068 /**

00105 /** 
00106   Write a block of data to the Serial port.  Will block for the time specified (in ms)
00107   if the queue fills up.
00108   @param buffer A pointer to the buffer to write from.
00109   @param count An integer specifying the number of bytes to write.
00110   @param timeout Time in milliseconds to block waiting for the queue to free up. 0 means don't wait.
00111   @return status.
00112 */
00113 int Serial_Write( uchar* buffer, int count, int timeout )
00114 {
00115   if ( Serial.users == 0 )
00116   {
00117     int status = Serial_SetActive( 1 );
00118     if ( status != CONTROLLER_OK )
00119       return status;
00120   }
00121 
00122   // Do the business
00123   while ( count )
00124   {
00125     if( xQueueSend( Serial.transmitQueue, buffer++, timeout / portTICK_RATE_MS  ) == 0 ) 
00126       return CONTROLLER_ERROR_QUEUE_ERROR; 
00127     count--;
00128   }
00129    
00130   /* Turn on the Tx interrupt so the ISR will remove the character from the 
00131   queue and send it. This does not need to be in a critical section as 
00132   if the interrupt has already removed the character the next interrupt 
00133   will simply turn off the Tx interrupt again. */ 
00134   AT91C_BASE_US0->US_IER = AT91C_US_TXRDY; 
00135  
00136   return CONTROLLER_OK;
00137 }

00139 /** 
00140   Read data from the Serial port.  Will block for the time specified (in ms) if 
00141   there are insufficient characters.  Blocking can be avoided if Serial_GetReadable( )
00142   is used to determine how many characters are available to read prior to calling
00143   this function.
00144   @param buffer A pointer to the buffer to read into.
00145   @param size An integer specifying the maximum number of bytes to read.
00146   @param timeout Time in milliseconds to block waiting for the specified number of bytes. 0 means don't wait.
00147   @return number of bytes read (>=0) or error <0 .
00148 */
00149 int Serial_Read( uchar* buffer, int size, int timeout )
00150 {
00151   if ( Serial.users == 0 )
00152   {
00153     int status = Serial_SetActive( 1 );
00154     if ( status != CONTROLLER_OK )
00155       return status;
00156   }
00157 
00158   // Do the business
00159   int count = 0;
00160   while ( count < size )
00161   {
00162     /* Place the character in the queue of characters to be transmitted. */ 
00163     if( xQueueReceive( Serial.receiveQueue, buffer++, timeout ) == 0 )
00164       break;
00165     count++;
00166   }
00167 
00168   return count;
00169 }

00395 int Serial_Init()
00396 {
00397   // If there are no default values, get some
00398   if ( !Serial.detailsInitialized )
00399     Serial_SetDefault();
00400 
00401   // Enable the peripheral clock
00402   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_US0;
00403 
00404   int status;
00405    
00406   status = Io_StartBits( IO_PA00_BIT | IO_PA01_BIT, false  );
00407   if ( status != CONTROLLER_OK )
00408     return status;
00409 
00410   Io_SetPeripheralA( IO_PA00 );
00411   Io_SetPeripheralA( IO_PA01 );
00412   Io_PioDisable( IO_PA00 );
00413   Io_PioDisable( IO_PA01 );
00414   
00415   // Create the queues
00416   Serial.receiveQueue = xQueueCreate( 100, 1 ); 
00417   Serial.transmitQueue = xQueueCreate( 100, 1 ); 
00418 
00419   // Disable interrupts
00420   AT91C_BASE_US0->US_IDR = (unsigned int) -1;
00421 
00422   // Timeguard disabled
00423   AT91C_BASE_US0->US_TTGR = 0;
00424 
00425   // Most of the detail setting is done in here
00426   // Also Resets TXRX and re-enables RX
00427   Serial_SetDetails();
00428 
00429   unsigned int mask = 0x1 << AT91C_ID_US0;    
00430                         
00431   /* Disable the interrupt on the interrupt controller */         
00432   AT91C_BASE_AIC->AIC_IDCR = mask ;                   
00433   /* Save the interrupt handler routine pointer and the interrupt priority */ 
00434   AT91C_BASE_AIC->AIC_SVR[ AT91C_ID_US0 ] = (unsigned int)Serial_Isr;     
00435   /* Store the Source Mode Register */                  
00436   AT91C_BASE_AIC->AIC_SMR[ AT91C_ID_US0 ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 4  ;       
00437   /* Clear the interrupt on the interrupt controller */         
00438   AT91C_BASE_AIC->AIC_ICCR = mask ;         
00439 
00440   AT91C_BASE_AIC->AIC_IECR = mask;
00441 
00442   AT91C_BASE_US0->US_IER = AT91C_US_RXRDY; 
00443 
00444 
00445   return CONTROLLER_OK;
00446 }

00453 int Serial_SetDetails( )
00454 {
00455   if ( Serial.users )
00456   {
00457      // Reset receiver and transmitter
00458     AT91C_BASE_US0->US_CR = AT91C_US_RSTRX | AT91C_US_RSTTX | AT91C_US_RXDIS | AT91C_US_TXDIS; 
00459 
00460     int baudValue; 
00461 
00462     // MCK is 47923200 for the Make Controller Kit
00463 
00464     // Calculate ( * 10 )
00465     baudValue = ( MCK * 10 ) / ( Serial.baud * 16 );
00466     // Round (and / 10)
00467     if ( ( baudValue % 10 ) >= 5 ) 
00468       baudValue = ( baudValue / 10 ) + 1; 
00469     else 
00470       baudValue /= 10;
00471 
00472     AT91C_BASE_US0->US_BRGR = baudValue; 
00473 
00474     AT91C_BASE_US0->US_MR = 
00475       ( ( Serial.hardwareHandshake ) ? AT91C_US_USMODE_HWHSH : AT91C_US_USMODE_NORMAL ) |
00476       ( AT91C_US_CLKS_CLOCK ) |
00477       ( ( ( Serial.bits - 5 ) << 6 ) & AT91C_US_CHRL ) |
00478       ( ( Serial.stopBits == 1 ) ? AT91C_US_NBSTOP_1_BIT : AT91C_US_NBSTOP_1_BIT ) |
00479       ( ( Serial.parity == 0 ) ? AT91C_US_PAR_NONE : ( ( Serial.parity == -1 ) ? AT91C_US_PAR_ODD : AT91C_US_PAR_EVEN ) );
00480       // 2 << 14; // this last thing puts it in loopback mode
00481 
00482 
00483     AT91C_BASE_US0->US_CR = AT91C_US_RXEN | AT91C_US_TXEN; 
00484   }
00485   return CONTROLLER_OK;
00486 }
