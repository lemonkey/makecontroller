00061 /** \defgroup AnalogIn Analog Inputs
00062   The AnalogIn subsystem converts 0-3.3V signals to 10-bit digital values.
00063   The analog to digital converters read incoming signals from 0 - 3.3V.  They are rated as 5V tolerant, 
00064   and indeed can momentarily withstand higher voltages, but will not return meaningful values for anything 
00065   above 3.3V.\n
00066   
00067   Converting the 0 - 1023 reading of the AnalogIn channel into a voltage is performed as follows:
00068     v = 3.3 * ( a / 1023.0 )
00069     where a is the AnalogIn value
00070 
00071   This is of course a floating point operation (slowish) using a division (slowish).  Fixed point 
00072   versions may be more suitable for some applications.  Where reduced accuracy is acceptable, the following 
00073   can be used to get the input as a percentage of 3.3V:
00074     p = ( 100 * a ) / 1023
00075 
00076   Initializing the controller's AnalogIn system is pretty involved (see AnalogIn_Init() in AnalogIn->c).  There are a lot of 
00077   different options - different converter speeds, DMA access, etc.  We've chosen something pretty simple here.
00078   More ambitious users may wish to alter the implementation.
00079 
00080   \todo Provide multi-channel conversion routines
00081 
00082   \ingroup Controller
00083   @{
00084 */

00132 /** 
00133   Read the value of an analog input.
00134   @param index An integer specifying which input (0-7).
00135   @return The value as an integer (0 - 1023).
00136 */
00137 int AnalogIn_GetValue( int index )
00138 {
00139   AnalogIn_SetActive( index, 1 );
00140   if ( index < 0 || index >= ANALOGIN_CHANNELS )
00141     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00142 
00143   if ( AnalogIn->channelUsers[ index ] < 1 )
00144   {
00145     int status = AnalogIn_Start( index );
00146     if ( status != CONTROLLER_OK )
00147       return status;
00148   }
00149 
00150   int value;
00151 
00152   if ( !xSemaphoreTake( AnalogIn->semaphore, 1000 ) )
00153     return -1;
00154 
00155   /* Third Step: Select the active channel */
00156   int mask = 1 << index; 
00157   AT91C_BASE_ADC->ADC_CHDR = ~mask;
00158   AT91C_BASE_ADC->ADC_CHER = mask;
00159   
00160   /* Fourth Step: Start the conversion */
00161   AT91C_BASE_ADC->ADC_CR = AT91C_ADC_START;
00162 
00163   /* Busy wait */
00164   // while ( !( AT91C_BASE_ADC->ADC_CHSR  & ( 1 << index ) ) );
00165 
00166   if ( !xSemaphoreTake( AnalogIn->doneSemaphore, 1000 ) )
00167     return -1;
00168 
00169   value = AT91C_BASE_ADC->ADC_LCDR & 0xFFFF;
00170 
00171   xSemaphoreGive( AnalogIn->semaphore );
00172 
00173   return value;
00174 }

00184 int AnalogIn_GetValueMulti( int mask, int values[] )
00185 {
00186   //AnalogIn_SetActive( 1 );
00187   if ( mask < 0 || mask > 255 ) // check the value is a valid 8-bit mask
00188     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00189 
00190   int i; // Is this the best way to make sure everything is started up properly?

00192   {
00193     if( mask >> i & 1 )
00194     {
00195       if ( AnalogIn->channelUsers[ i ] < 1 )
00196       {
00197         int status = AnalogIn_Start( i );
00198         if ( status != CONTROLLER_OK )
00199         return status;
00200       }
00201     }
00202   }
00203 
00204   if ( !xSemaphoreTake( AnalogIn->semaphore, 1000 ) )
00205     return -1;
00206 
00207   /* Third Step: Select the active channels */
00208   AT91C_BASE_ADC->ADC_CHDR = ~mask;
00209   AT91C_BASE_ADC->ADC_CHER = mask;
00210

00280 int AnalogIn_Start( int index )
00281 {
00282   if ( index < 0 || index >= ANALOGIN_CHANNELS )
00283     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00284   if ( AnalogIn->channelUsers[ index ]++ == 0 )
00285   {
00286     int status;
00287 
00288     // The lower four channel pins are shared with other subsystems, so no locking
00289     if ( index < 4 )
00290     {
00291       int io = AnalogIn_GetIo( index );
00292       status = Io_Start( io, false );
00293       if ( status != CONTROLLER_OK )
00294       {
00295         AnalogIn->channelUsers[ index ]--;
00296         return status;
00297       }
00298 
00299       Io_PullupDisable( io );
00300     }
00301 
00302     if ( AnalogIn->users++ == 0 )
00303     {
00304       AnalogIn_Init();  
00305     }
00306   }
00307   
00308   return CONTROLLER_OK;
00309 }

00311 int AnalogIn_Stop( int index )
00312 {
00313   if ( index < 0 || index >= ANALOGIN_CHANNELS )
00314     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00315   if ( AnalogIn->channelUsers[ index ] <= 0 )
00316     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00317   if ( --AnalogIn->channelUsers[ index ] == 0 )
00318   {
00319     if ( index < 4 )
00320     {
00321       int io = AnalogIn_GetIo( index );
00322       Io_Stop( io );
00323     }
00324     if ( --AnalogIn->users == 0 )
00325     {
00326       AnalogIn_Deinit();
00327     }
00328   }
00329   return CONTROLLER_OK;
00330 }

00332 int AnalogIn_Init()
00333 {
00334   // Enable the peripheral clock
00335   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_ADC;
00336 
00337   // Set up
00338   AT91C_BASE_ADC->ADC_MR =
00339        AT91C_ADC_TRGEN_DIS | // Hardware Trigger Disabled
00340     // AT91C_ADC_TRGEN_EN  | // Hardware Trigger Disabled
00341     //   AT91C_ADC_TRGSEL_ | // Hardware Trigger Disabled
00342     // AT91C_ADC_TRGSEL_TIOA0  | // Trigger Selection Don't Care
00343        AT91C_ADC_LOWRES_10_BIT | // 10 bit conversion
00344     // AT91C_ADC_LOWRES_8_BIT | // 8 bit conversion
00345        AT91C_ADC_SLEEP_NORMAL_MODE | // SLEEP
00346     // AT91C_ADC_SLEEP_MODE | // SLEEP
00347        ( ( 9 << 8 ) & AT91C_ADC_PRESCAL ) | // Prescale rate (8 bits)
00348        ( ( 127 << 16 ) & AT91C_ADC_STARTUP ) | // Startup rate
00349        ( ( 127 << 24 ) & AT91C_ADC_SHTIM ); // Sample and Hold Time
00350 
00351   //TODO: Will need to fine-tune these timings.
00352 
00353   // Do the OS stuff
00354 
00355   vSemaphoreCreateBinary( AnalogIn->semaphore );
00356 
00357   // Create the sempahore that will be used to wake the calling process up 
00358   vSemaphoreCreateBinary( AnalogIn->doneSemaphore );
00359   xSemaphoreTake( AnalogIn->doneSemaphore, 0 );
00360 
00361   // Initialize the interrupts
00362   // WAS AT91F_AIC_ConfigureIt( AT91C_ID_ADC, 3, AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL, ( void (*)( void ) ) AnalogInIsr );
00363   // Which is defined at the bottom of the AT91SAM7X256.h file
00364   unsigned int mask ;                         
00365                                       
00366   mask = 0x1 << AT91C_ID_ADC;   
00367                         
00368   /* Disable the interrupt on the interrupt controller */         
00369   AT91C_BASE_AIC->AIC_IDCR = mask ;                   
00370   /* Save the interrupt handler routine pointer and the interrupt priority */ 
00371   AT91C_BASE_AIC->AIC_SVR[ AT91C_ID_ADC ] = (unsigned int)AnalogInIsr;      
00372   /* Store the Source Mode Register */                  
00373   AT91C_BASE_AIC->AIC_SMR[ AT91C_ID_ADC ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 4  ;       
00374   /* Clear the interrupt on the interrupt controller */         
00375   AT91C_BASE_AIC->AIC_ICCR = mask ;         
00376 
00377   AT91C_BASE_ADC->ADC_IER = AT91C_ADC_DRDY; 
00378 
00379   AT91C_BASE_AIC->AIC_IECR = mask;
00380

00412 /** \defgroup AnalogInOSC Analog In - OSC

00416   \section devices Devices
00417   There are 8 Analog Inputs on the Make Application Board, numbered 0 - 7.
00418   
00419   \section properties Properties
00420   The Analog Ins have two properties:
00421   - value
00422   - active
00423 
00424   \par Value
00425   The \b value property corresponds to the incoming signal of an Analog In.
00426   The range of values you can expect to get back are from 0 - 1023.
00427   Because you can only ever \em read the value of an input, you'll never
