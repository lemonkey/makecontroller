00050 /**
00051   The structure used for a standard AT Command packet.
00052   - \b frameID is the ID for this packet that subsequent response/status messages can refer to.
00053   - \b command is the 2-character AT command.
00054   - \b parameters is a buffer that holds the value of the AT command.
00055  */
00056 typedef struct
00057 {
00058   uint8 frameID;
00059   uint8 command[ 2 ];
00060   uint8 parameters[ 97 ];
00061 } XBee_ATCommand;
00062 
00063 /** 
00064   The structure used for a response to a standard AT Command packet.
00065   - \b frameID is the ID that this response is referring to.
00066   - \b command is the 2-character AT command.
00067   - \b status is response value - 0 (OK) or 1 (ERROR).
00068   - \b value is a buffer that holds the value of the AT command response.
00069  */
00070 typedef struct
00071 {
00072   uint8 frameID;
00073   uint8 command[ 2 ];
00074   uint8 status;
00075   uint8 value[ 96 ];
00076 }  XBee_ATCommandResponse;
00077 
00078 /** 
00079   The structure used to transmit an XBee packet with a 64-bit destination address.
00080   - \b frameID is the ID for this packet that subsequent response/status messages can refer to.
00081   - \b destination is the 64-bit (8-byte) address of the destination.
00082   - \b options - 0x01 (disable ACK) or 0x04 (Send with Broadcast PAN ID).
00083   - \b data is a buffer that holds the value of the outgoing packet.
00084  */
00085 typedef struct
00086 {
00087   uint8 frameID;
00088   uint8 destination[ 8 ];
00089   uint8 options;
00090   uint8 data[ 90 ];
00091 }  XBee_TX64;
00092 
00093 /** 
00094   The structure used to transmit an XBee packet with a 16-bit destination address.
00095   - \b frameID is the ID for this packet that subsequent response/status messages can refer to.
00096   - \b destination is the 16-bit (2-byte) address of the destination.
00097   - \b options - 0x01 (disable ACK) or 0x04 (Send with Broadcast PAN ID).
00098   - \b data is a buffer that holds the value of the outgoing packet.
00099  */
00100 typedef struct
00101 {
00102   uint8 frameID;
00103   uint8 destination[ 2 ];
00104   uint8 options;
00105   uint8 data[ 96 ];
00106 }  XBee_TX16;
00107 
00108 /** 
00109   When a transmit request is completed, the module sends a transmit status message.
00110   - \b frameID is the ID that this response is referring to.
00111   - \b status can have values of:
00112     - 0 - success
00113     - 1 - No ACK received
00114     - 2 - CCA failure
00115     - 3 - Purged
00116  */
00117 typedef struct
00118 {
00119   uint8 frameID;
00120   uint8 status;
00121 }  XBee_TXStatus;
00122 
00123 /** 
00124   An incoming packet with a 64-bit address.
00125   - \b source is the 64-bit (8-byte) address of the sender.
00126   - \b rssi is the signal strength of the received message.
00127   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00128   - \b data is a buffer that holds the value of the incoming packet.
00129  */
00130 typedef struct
00131 {
00132   uint8 source[ 8 ];
00133   uint8 rssi;
00134   uint8 options;
00135   uint8 data[ 89 ];
00136 }  XBee_RX64;
00137 
00138 /**

00140   - \b source is the 16-bit (2-byte) address of the sender.
00141   - \b rssi is the signal strength of the received message.
00142   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00143   - \b data is a buffer that holds the value of the incoming packet.
00144  */

00153 /** 
00154   An incoming packet with IO data from a 64-bit address.
00155   - \b source is the 64-bit (8-byte) address of the sender.
00156   - \b rssi is the signal strength of the received message.
00157   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00158   - \b samples - the number of samples in this packet.
00159   - \b channelIndicators - bit mask indicating which channels have been sampled.
00160   - \b data is a buffer that holds the IO values as indicated by \b channelIndicators.
00161  */
00162 typedef struct
00163 {
00164   uint8 source[ 8 ];
00165   uint8 rssi;
00166   uint8 options;
00167   uint8 samples;
00168   uint8 channelIndicators[ 2 ];
00169   uint8 data[ 86 ];
00170 }  XBee_IO64;
00171 
00172 /** 
00173   An incoming packet with IO data from a 16-bit address.
00174   - \b source is the 16-bit (2-byte) address of the sender.
00175   - \b rssi is the signal strength of the received message.
00176   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00177   - \b samples - the number of samples in this packet.
00178   - \b channelIndicators - bit mask indicating which channels have been sampled.
00179   - \b data is a buffer that holds the IO values as indicated by \b channelIndicators.

00185   uint8 options;
00186   uint8 samples;
00187   uint8 channelIndicators[ 2 ];
00188   uint8 data[ 92 ];
00189 }  XBee_IO16;
