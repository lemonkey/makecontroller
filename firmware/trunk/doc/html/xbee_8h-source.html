00058 /**
00059   The structure used for a standard AT Command packet.
00060   - \b frameID is the ID for this packet that subsequent response/status messages can refer to.
00061   - \b command is the 2-character AT command.
00062   - \b parameters is a buffer that holds the value of the AT command.
00063  */
00064 typedef struct
00065 {
00066   uint8 frameID;
00067   uint8 command[ 2 ];
00068   uint8 parameters[ 97 ];
00069 } XBee_ATCommand;
00070 
00071 /** 
00072   The structure used for a response to a standard AT Command packet.
00073   - \b frameID is the ID that this response is referring to.
00074   - \b command is the 2-character AT command.
00075   - \b status is response value - 0 (OK) or 1 (ERROR).
00076   - \b value is a buffer that holds the value of the AT command response.
00077  */
00078 typedef struct
00079 {
00080   uint8 frameID;
00081   uint8 command[ 2 ];
00082   uint8 status;
00083   uint8 value[ 96 ];
00084 }  XBee_ATCommandResponse;
00085 
00086 /** 
00087   The structure used to transmit an XBee packet with a 64-bit destination address.
00088   - \b frameID is the ID for this packet that subsequent response/status messages can refer to.
00089   - \b destination is the 64-bit (8-byte) address of the destination.
00090   - \b options - 0x01 (disable ACK) or 0x04 (Send with Broadcast PAN ID).
00091   - \b data is a buffer that holds the value of the outgoing packet.
00092  */
00093 typedef struct
00094 {
00095   uint8 frameID;
00096   uint8 destination[ 8 ];
00097   uint8 options;
00098   uint8 data[ 90 ];
00099 }  XBee_TX64;
00100 
00101 /** 
00102   The structure used to transmit an XBee packet with a 16-bit destination address.
00103   - \b frameID is the ID for this packet that subsequent response/status messages can refer to.
00104   - \b destination is the 16-bit (2-byte) address of the destination.
00105   - \b options - 0x01 (disable ACK) or 0x04 (Send with Broadcast PAN ID).
00106   - \b data is a buffer that holds the value of the outgoing packet.
00107  */
00108 typedef struct
00109 {
00110   uint8 frameID;
00111   uint8 destination[ 2 ];
00112   uint8 options;
00113   uint8 data[ 96 ];
00114 }  XBee_TX16;
00115 
00116 /** 
00117   When a transmit request is completed, the module sends a transmit status message.
00118   - \b frameID is the ID that this response is referring to.
00119   - \b status can have values of:
00120     - 0 - success
00121     - 1 - No ACK received
00122     - 2 - CCA failure
00123     - 3 - Purged
00124  */
00125 typedef struct
00126 {
00127   uint8 frameID;
00128   uint8 status;
00129 }  XBee_TXStatus;

00131 /** 
00132   An incoming packet with a 64-bit address.
00133   - \b source is the 64-bit (8-byte) address of the sender.
00134   - \b rssi is the signal strength of the received message.
00135   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00136   - \b data is a buffer that holds the value of the incoming packet.
00137  */
00138 typedef struct

00140   uint8 source[ 8 ];
00141   uint8 rssi;
00142   uint8 options;
00143   uint8 data[ 89 ];
00144 }  XBee_RX64;

00161 /** 
00162   An incoming packet with IO data from a 64-bit address.
00163   - \b source is the 64-bit (8-byte) address of the sender.
00164   - \b rssi is the signal strength of the received message.
00165   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00166   - \b samples - the number of samples in this packet.
00167   - \b channelIndicators - bit mask indicating which channels have been sampled.
00168   - \b data is a buffer that holds the IO values as indicated by \b channelIndicators.
00169  */
00170 typedef struct
00171 {
00172   uint8 source[ 8 ];
00173   uint8 rssi;
00174   uint8 options;
00175   uint8 samples;
00176   uint8 channelIndicators[ 2 ];
00177   uint8 data[ 86 ];
00178 }  XBee_IO64;
00179 
00180 /** 
00181   An incoming packet with IO data from a 16-bit address.
00182   - \b source is the 16-bit (2-byte) address of the sender.
00183   - \b rssi is the signal strength of the received message.
00184   - \b options - bit 1 is Address Broadcast, bit 2 is PAN broadcast.  Other bits reserved.
00185   - \b samples - the number of samples in this packet.
00186   - \b channelIndicators - bit mask indicating which channels have been sampled.
00187   - \b data is a buffer that holds the IO values as indicated by \b channelIndicators.
00188  */
00189 typedef struct

00192   uint8 rssi;
00193   uint8 options;
00194   uint8 samples;
00195   uint8 channelIndicators[ 2 ];
00196   uint8 data[ 92 ];
00197 }  XBee_IO16;
