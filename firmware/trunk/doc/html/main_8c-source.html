00102 int main( void )
00103 {
00104   /* Setup the ports. */
00105   prvSetupHardware();
00106 
00107   /* Create the make task. */
00108   xTaskCreate( MakeStarterTask, "Make", 300, NULL, 4, NULL );
00109 
00110   /*NOTE : Tasks run in system mode and the scheduler runs in Supervisor mode.
00111   The processor MUST be in supervisor mode when vTaskStartScheduler is 
00112   called.  The demo applications included in the FreeRTOS.org download switch
00113   to supervisor mode prior to main being called.  If you are not using one of
00114   these demo application projects then ensure Supervisor mode is used here. */
00115   vTaskStartScheduler();
00116 
00117   /* Should never get here! */
00118   return 0;
00119 }

00123 static void prvSetupHardware( void )
00124 {
00125   /* When using the JTAG debugger the hardware is not always initialised to
00126   the correct default state.  This line just ensures that this does not
00127   cause all interrupts to be masked at the start. */
00128   AT91C_BASE_AIC->AIC_EOICR = 0;
00129   
00130   /* Most setup is performed by the low level init function called from the
00131   startup asm file.*/
00132 
00133   // EEPROM DISABLE
00134   #if ( CONTROLLER_VERSION == 90 )
00135     AT91C_BASE_PIOB->PIO_PER = 1 << 17; // Set PB17 - the EEPROM ~enable - in PIO mode
00136     AT91C_BASE_PIOB->PIO_OER = 1 << 17; // Configure in Output
00137     AT91C_BASE_PIOB->PIO_SODR = 1 << 17; // Set Output
00138   #endif
00139   #if ( CONTROLLER_VERSION == 95 || CONTROLLER_VERSION == 100 )
00140     AT91C_BASE_PIOA->PIO_PER = 1 << 9; // Set PA9 - the EEPROM ~enable - in PIO mode
00141     AT91C_BASE_PIOA->PIO_OER = 1 << 9; // Configure in Output
00142     AT91C_BASE_PIOA->PIO_SODR = 1 << 9; // Set Output
00143   #endif
00144 
00145   // CAN DISABLE
00146   #if ( CONTROLLER_VERSION == 90 )
00147     AT91C_BASE_PIOB->PIO_PER = 1 << 16; // Set PB16 - the CAN ~enable - in PIO mode
00148     AT91C_BASE_PIOB->PIO_OER = 1 << 16; // Configure in Output
00149     AT91C_BASE_PIOB->PIO_SODR = 1 << 16; // Set Output
00150   #endif
00151   #if ( CONTROLLER_VERSION == 95  || CONTROLLER_VERSION == 100 )
00152     AT91C_BASE_PIOA->PIO_PER = 1 << 7; // Set PA7 - the CAN ~enable - in PIO mode
00153     AT91C_BASE_PIOA->PIO_OER = 1 << 7; // Configure in Output
00154     AT91C_BASE_PIOA->PIO_SODR = 1 << 7; // Set Output
00155   #endif
00156 
00157   /* Enable the peripheral clock. */
00158   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_PIOA;
00159   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_PIOB;
00160   
00161   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_EMAC;
00162   
00163   /* Turn the USB line into an input, kill the pull up */
00164   #if ( CONTROLLER_VERSION == 90 )
00165     AT91C_BASE_PIOB->PIO_PER = 1 << 10; 
00166     AT91C_BASE_PIOB->PIO_ODR = 1 << 10;
00167     AT91C_BASE_PIOB->PIO_PPUDR = 1 << 10;
00168   #endif
00169   #if ( CONTROLLER_VERSION == 95 || CONTROLLER_VERSION == 100 )
00170     AT91C_BASE_PIOA->PIO_PER = 1 << 10; 
00171     AT91C_BASE_PIOA->PIO_ODR = 1 << 10;
00172     AT91C_BASE_PIOA->PIO_PPUDR = 1 << 10;
00173   #endif
00174 
00175   /* Setup the PIO for the USB pull up resistor. */
00176   /* Start low: no USB */
00177   #if ( CONTROLLER_VERSION == 90 )
00178     AT91C_BASE_PIOB->PIO_PER = AT91C_PIO_PB11;
00179     AT91C_BASE_PIOB->PIO_OER = AT91C_PIO_PB11;
00180     AT91C_BASE_PIOB->PIO_SODR = AT91C_PIO_PB11;
00181   #endif
00182   #if ( CONTROLLER_VERSION == 95 || CONTROLLER_VERSION == 100 )
00183     AT91C_BASE_PIOA->PIO_PER = AT91C_PIO_PA11;
00184     AT91C_BASE_PIOA->PIO_OER = AT91C_PIO_PA11;
00185     AT91C_BASE_PIOA->PIO_SODR = AT91C_PIO_PA11;
00186   #endif
00187 }
