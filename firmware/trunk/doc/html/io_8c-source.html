00070 /** \defgroup Io
00071   A mechanism to manage the 32 parallel io lines on the controller.
00072 
00073   Control of the SAM7X's IO lines is more complex than most microcontrollers.  All 
00074   lines may be inputs or outputs, have pull-ups or not, have a glitch filter enabled 
00075   or not, and so on.  In addition all IO pins serve at least double and sometimes 
00076   triple duty, being general IO lines and also being IO lines for one or more of 
00077   the controller's many on-chip peripherals.  All these functions can be controlled
00078   from the Io subsystem.
00079 
00080   This API can be used in two ways - either one line at a time or many at a time.  
00081   Functions supporting the latter are suffixed with "Bits".  So the one-at-a-time 
00082   mechanism for setting an IO to true is called Io_SetTrue().  It takes an index and
00083   the ones for setting
00084   a number of lines is Io_SetTrueBits( ).
00085 
00086   The pattern of use is as follows:  prior to using an IO line, Call Io_Start() on it, the 
00087   return value indicates if the start operation was successful.  After this, call any 
00088   of the other Io functions.
00089 
00090   \todo Consider changing the api slightly to permit the immediate use of the io
00091         without starting it.
00092   \todo Add glitch filter control
00093   \todo Implement more bits-style functions
00094   \ingroup Controller
00095   @{
00096 */

00098 int Io_Start( int index, bool lock )
00099 {
00100   if ( index < 0 || index > IO_PIN_COUNT )
00101     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00102   IoPin* p = &Io.pins[ index ];
00103   // If it's not locked at all and 
00104   //    either no lock is requested or no a lock is requested but no other are using the io and
00105   //    there are not a huge number of users
00106   if ( !p->lock && ( !lock || ( lock && p->users == 0 ) ) && ( p->users < IO_USERS_MAX - 2 ) )
00107   {
00108     if ( Io.users == 0 )
00109       Io_Init();
00110    
00111     // Lock it up.
00112     Io.users++;
00113     p->users++;
00114     if ( lock )
00115       p->lock = true;
00116 
00117     // Output Pair 1
00118     // Switch to PIO
00119     // AT91C_BASE_PIOA->PIO_PER = AT91C_PIO_PA26;
00120     // AT91C_BASE_PIOB->PIO_PER = AT91C_PIO_PB23;
00121     
00122     return CONTROLLER_OK;
00123   }
00124   else
00125     return CONTROLLER_ERROR_TOO_MANY_USERS;
00126 }

00128 int Io_Stop( int index )
00129 {
00130   if ( index < 0 || index > IO_PIN_COUNT )
00131     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00132   IoPin* p = &Io.pins[ index ];
00133   if ( p->users > 0 )
00134   {
00135     Io.users--;
00136     p->users--;
00137     p->lock = false;
00138 
00139     if ( Io.users == 0 )
00140       Io_Deinit();
00141 
00142     return CONTROLLER_OK;
00143   }
00144   else
00145     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00146 }

00148 int Io_StartBits( longlong bits, bool lock )
00149 {
00150   IoPin* p = &Io.pins[ 0 ];
00151   int i = 0;
00152   int failure = -1;
00153   ullong mask = 1LL;
00154   for ( i = 0; i < 64; i++ )
00155   {
00156     if ( bits & mask )
00157     {
00158       if ( !p->lock && ( !lock || ( lock && p->users == 0 ) ) && p->users < IO_USERS_MAX - 2 )
00159       {
00160         if ( Io.users == 0 )
00161           Io_Init();
00162         if ( lock )
00163           p->lock = true;
00164         p->users++;
00165         Io.users++;
00166       }
00167       else
00168       {
00169         failure = i;
00170         break;
00171       }
00172     }
00173     mask <<= 1;
00174     p++;
00175   }
00176 
00177   if ( failure != -1 )
00178   {
00179     IoPin* p = &Io.pins[ 0 ];
00180     mask = 1LL;
00181     for ( i = 0; i < failure; i++ )
00182     {
00183       if ( bits & mask )
00184       {
00185         if ( lock )
00186           p->lock = false;
00187         p->users--;
00188         Io.users--;
00189         if ( Io.users == 0 )
00190           Io_Deinit();
00191       }
00192       mask <<= 1;
00193       p++;
00194     }
00195     return CONTROLLER_ERROR_CANT_LOCK;
00196   }
00197 
00198   return CONTROLLER_OK;
00199 }

00606   \verbatim /io/4/value 1\endverbatim

00610   \verbatim /io/4/value\endverbatim

00612   \par Active

00614   \verbatim /io/0/active \endverbatim
