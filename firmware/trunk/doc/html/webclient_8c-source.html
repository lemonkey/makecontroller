00028 /** \defgroup webclient Web Client
00029   A very simple web client for HTTP operations.
00030 
00031   The web client system allows the Make Controller to get/post data to a webserver.  This
00032   makes it straightforward to use the Make Controller as a source of data for your web apps.
00033 
00034   There's currently not a method provided for name resolution - you can always ping the 
00035   server you want to communicate with to see its IP address, and just use that.
00036 
00037   \todo Get a DNS solution together.
00038 
00039   \ingroup Controller
00040   @{
00041 */
00042 
00043 /** 
00044   Performs an HTTP GET operation to the path at the address / port specified.  
00045   The result is returned in the specified buffer.
00046   @param address The IP address of the server to get from.  Usually created using the IP_ADDRESS( ) macro.
00047   @param port The port to connect on.  Usually 80 for HTTP.
00048   @param path The path on the server to connect to.
00049   @param buffer A pointer to the buffer read back into.  
00050   @param buffer_size An integer specifying the actual size of the buffer.
00051   @return the number of bytes read, or < 0 on error.
00052 
00053   \par Example
00054   \code
00055   int addr = IP_ADDRESS( 72, 249, 53, 185); // makingthings.com is 72.249.53.185
00056   int bufLength = 100;
00057   char myBuffer[bufLength];
00058   int getSize = WebClient_Get( addr, 80, "/test/path", myBuffer, bufLength );
00059   \endcode
00060   Now we should have the results of the HTTP GET from \b www.makingthings.com/test/path in \b myBuffer.
00061 */
00062 int WebClient_Get( int address, int port, char* path, char* buffer, int buffer_size )
00063 {
00064   char b[ WEBCLIENT_INTERNAL_BUFFER_SIZE ];
00065   int buffer_read = 0;
00066   void* s = Socket( address, port );  
00067   if ( s != NULL )
00068   {
00069     int send_len = snprintf( b, WEBCLIENT_INTERNAL_BUFFER_SIZE, "GET %s HTTP/1.1\r\n\r\n", path );
00070     if ( send_len > WEBCLIENT_INTERNAL_BUFFER_SIZE )
00071     {
00072       SocketClose( s );
00073       return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00074     }
00075 
00076     SocketWrite( s, b, send_len );
00077     int content_length = 0;
00078     
00079     int buffer_length;
00080     while ( ( buffer_length = SocketReadLine( s, b, WEBCLIENT_INTERNAL_BUFFER_SIZE ) ) )
00081     {
00082       if ( strncmp( b, "\r\n", 2 ) == 0 )
00083         break;
00084       if ( strncmp( b, "Content-Length", 14 ) == 0 )
00085         content_length = atoi( &b[ 15 ] );
00086     }
00087     
00088     if ( content_length > 0 && buffer_length > 0 )
00089     {
00090       char* bp = buffer;
00091       while ( ( buffer_length = SocketRead( s, bp, buffer_size - buffer_read ) ) )
00092       {
00093         buffer_read += buffer_length;
00094         bp += buffer_read;
00095         if ( buffer_read >= content_length )
00096           break;
00097       }
00098     }
00099           
00100     SocketClose( s );
00101     return buffer_read;
00102   }
00103   else
00104     return CONTROLLER_ERROR_BAD_ADDRESS;
00105 }

00107 /** 
00108   Performs an HTTP POST operation to the path at the address / port specified.  The actual post contents 
00109   are found read from a given buffer and the result is returned in the same buffer.
00110   @param address The IP address of the server to post to.
00111   @param port The port on the server you're connecting to. Usually 80 for HTTP.
00112   @param path The path on the server to post to.
00113   @param buffer A pointer to the buffer to write from and read back into.  
00114   @param data_length An integer specifying the number of bytes to write.
00115   @param buffer_size An integer specifying the actual size of the buffer.
00116   @return status.
00117 
00118   \par Example
00119   \code
00120   // we'll post a test message to www.makingthings.com/post/path
00121   int addr = IP_ADDRESS( 72, 249, 53, 185); // makingthings.com is 72.249.53.185
00122   int bufLength = 100;
00123   char myBuffer[bufLength];
00124   sprintf( myBuffer, "A test message to post" );
00125   int result = WebClient_Post( addr, 80, "/post/path", myBuffer, 
00126                                 strlen("A test message to post"), bufLength );
00127   \endcode
00128 */
00129 int WebClient_Post( int address, int port, char* path, char* buffer, int data_length, int buffer_size )
00130 {
00131   char b[ WEBCLIENT_INTERNAL_BUFFER_SIZE ];
00132   int buffer_read = 0;
00133   int wrote = 0;
00134   void* s = Socket( address, port );  
00135   if ( s != NULL )
00136   { 
00137     int send_len = snprintf( b, WEBCLIENT_INTERNAL_BUFFER_SIZE, 
00138                                 "POST %s HTTP/1.1\r\nContent-Type: text/plain\r\nContent-Length: %d\r\n\r\n", 
00139                                 path, data_length );
00140     if ( send_len > WEBCLIENT_INTERNAL_BUFFER_SIZE )
00141     {
00142       SocketClose( s );
00143       return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00144     }
00145 
00146     wrote = SocketWrite( s, b, send_len );
00147     if ( wrote == 0 )
00148     {
00149       SocketClose( s );
00150       return CONTROLLER_ERROR_WRITE_FAILED;
00151     }
00152 
00153     SocketWrite( s, buffer, data_length );
00154     
00155     int content_length = 0;
00156     int b_len;
00157     while ( ( b_len = SocketReadLine( s, b, WEBCLIENT_INTERNAL_BUFFER_SIZE ) ) )
00158     {
00159       if ( strncmp( b, "\r\n", 2 ) == 0 )
00160         break;
00161       if ( strncmp( b, "Content-Length", 14 ) == 0 )
00162         content_length = atoi( &b[ 16 ] );
00163     }
00164           
00165     if ( content_length > 0 && b_len > 0 )
00166     {
00167       char* bp = buffer;
00168       while ( ( b_len = SocketRead( s, bp, buffer_size - buffer_read ) ) )
00169       {
00170         buffer_read += b_len;
00171         bp += buffer_read;
00172         if ( buffer_read >= content_length )
00173           break;
00174       }
00175     }          
00176 
00177     SocketClose( s );
00178     return buffer_read;
00179   }
00180   else
00181     return CONTROLLER_ERROR_BAD_ADDRESS;
00182 }
