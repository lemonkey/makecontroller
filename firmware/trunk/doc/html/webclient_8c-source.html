00028 /** 
00029   Performs an HTTP GET operation to the path at the address / port specified.  The result 
00030   is returned in the specified buffer.
00031   @param buffer A pointer to the buffer read back into.  
00032   @param buffer_size An integer specifying the actual size of the buffer.
00033   @return status.
00034 */
00035 int WebClient_Get( int address, int port, char* path, char* buffer, int buffer_size )
00036 {
00037   char b[ WEBCLIENT_INTERNAL_BUFFER_SIZE ];
00038   int buffer_read = 0;
00039   void* s = Socket( address, port );  
00040   if ( s != NULL )
00041   {
00042     int send_len = snprintf( b, WEBCLIENT_INTERNAL_BUFFER_SIZE, "GET %s HTTP/1.1\r\n\r\n", path );
00043     if ( send_len > WEBCLIENT_INTERNAL_BUFFER_SIZE )
00044     {
00045       SocketClose( s );
00046       return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00047     }
00048 
00049     AppLed_SetState( 1, 1 );
00050     SocketWrite( s, b, send_len );
00051     int content_length = 0;
00052     
00053     AppLed_SetState( 2, 1 );
00054     int buffer_length;
00055     while ( ( buffer_length = SocketReadLine( s, b, WEBCLIENT_INTERNAL_BUFFER_SIZE ) ) )
00056     {
00057       if ( strncmp( b, "\r\n", 2 ) == 0 )
00058         break;
00059       if ( strncmp( b, "Content-Length", 14 ) == 0 )
00060         content_length = atoi( &b[ 15 ] );
00061     }
00062     
00063     if ( content_length > 0 && buffer_length > 0 )
00064     {
00065       AppLed_SetState( 3, 1 );
00066       char* bp = buffer;
00067       while ( ( buffer_length = SocketRead( s, bp, buffer_size - buffer_read ) ) )
00068       {
00069         buffer_read += buffer_length;
00070         bp += buffer_read;
00071         if ( buffer_read >= content_length )
00072           break;
00073       }
00074     }
00075           
00076     SocketClose( s );
00077 
00078     AppLed_SetState( 1, 0 );
00079     AppLed_SetState( 2, 0 );
00080     AppLed_SetState( 3, 0 );
00081    
00082     return buffer_read;
00083   }
00084   else
00085     return CONTROLLER_ERROR_BAD_ADDRESS;
00086 }

00088 /** 
00089   Performs an HTTP POST operation to the path at the address / port specified.  The actual post contents 
00090   are found in buffer and the result is returned in buffer.
00091   @param buffer A pointer to the buffer to write from and read back into.  
00092   @param buffer_length An integer specifying the number of bytes to write.
00093   @param buffer_size An integer specifying the actual size of the buffer.

00105                                 "POST %s HTTP/1.1\r\nContent-Type: text/plain\r\nContent-Length: %d\r\n\r\n", 
00106                                 path, buffer_length );
00107     if ( send_len > WEBCLIENT_INTERNAL_BUFFER_SIZE )
00108     {
