00033 void DisableFIQFromThumb( void )
00034 {
00035   asm volatile ( "STMDB SP!, {R0}" );   /* Push R0.                 */
00036   asm volatile ( "MRS   R0, CPSR" );    /* Get CPSR.                */
00037   asm volatile ( "ORR   R0, R0, #0x40" ); /* Disable FIQ.           */
00038   asm volatile ( "MSR   CPSR, R0" );    /* Write back modified value.       */
00039   asm volatile ( "LDMIA SP!, {R0}" );   /* Pop R0.                  */
00040   asm volatile ( "BX    R14" );       /* Return back to thumb.          */
00041 }
00042     
00043 void EnableFIQFromThumb( void )
00044 {
00045   asm volatile ( "STMDB SP!, {R0}" );   /* Push R0.                 */  
00046   asm volatile ( "MRS   R0, CPSR" );    /* Get CPSR.                */  
00047   asm volatile ( "BIC   R0, R0, #0x40" ); /* Enable FIQ.              */  
00048   asm volatile ( "MSR   CPSR, R0" );    /* Write back modified value.       */  
00049   asm volatile ( "LDMIA SP!, {R0}" );   /* Pop R0.                  */
00050   asm volatile ( "BX    R14" );       /* Return back to thumb.          */
00051 }

00065 void FastTimer_Isr( void )
00066 {
00067   int status = AT91C_BASE_TC2->TC_SR;
00068   if ( status & AT91C_TC_CPCS )
00069   {
00070     FastTimer.servicing = true;
00071 
00072     //AT91C_BASE_TC2->TC_CCR = AT91C_TC_CLKDIS;
00073 
00074     // make sure there's not another IRQ while we're processing
00075     timeReference = AT91C_BASE_TC2->TC_RC;
00076     AT91C_BASE_TC2->TC_RC = 0xFF00;
00077 
00078 #ifdef FASTIRQ_MONITOR_IO
00079     Io_SetTrue( FASTIRQ_MONITOR_IO );
00080 #endif
00081 
00082 #ifdef FASTIRQ_STATS
00083     int startCount = AT91C_BASE_TC2->TC_CV;
00084 
00085     // moved outside for debugging
00086     //int jitter;
00087     jitter = AT91C_BASE_TC2->TC_CV;
00088 
00089     if ( ++FastTimer.count == 1000 )
00090     {
00091       // need to not do division here... takes too long
00092       //FastTimer.jitterTotal = FastTimer.jitterTotal / FastTimer.count;
00093       FastTimer.jitterTotal = 0;
00094       FastTimer.jitterMax = 0;
00095       // need to not do division here... takes too long
00096       // FastTimer.durationTotal = FastTimer.durationTotal / FastTimer.count;
00097       FastTimer.durationTotal = 0;
00098       FastTimer.durationMax = 0;
00099       FastTimer.count = 1;
00100     }
00101     
00102     FastTimer.jitterTotal += jitter;
00103         
00104     if ( jitter > FastTimer.jitterMax )
00105       FastTimer.jitterMax = jitter;
00106     if ( jitter > FastTimer.jitterMaxAllDay )
00107       FastTimer.jitterMaxAllDay = jitter;
00108 
00109 #endif
00110 
00111     //FastTimerEntry* te = FastTimer.first;
00112     // Use this during debuggin
00113     te = FastTimer.first;
00114 
00115     FastTimer.next = NULL;
00116     FastTimer.previous = NULL;
00117     FastTimer.nextTime = 0xFF00;
00118     int removed = false;
00119     // timeReference = AT91C_BASE_TC2->TC_CV;
00120 
00121     while ( te != NULL )
00122     {
00123       FastTimer.next = te->next;
00124       te->timeCurrent -= ( timeReference + AT91C_BASE_TC2->TC_CV );
00125       if ( te->timeCurrent <= FASTTIMER_MINCOUNT )
00126       {
00127         // Watch out for gross errors
00128         //if ( te->timeCurrent < -FASTTIMER_MINCOUNT)
00129         //  te->timeCurrent = -FASTTIMER_MINCOUNT;
00130 
00131         if ( te->repeat )
00132         {
00133           te->timeCurrent += te->timeInitial;
00134         }
00135         else
00136         {
00137           // remove it if necessary (do this first!)
00138           if ( FastTimer.previous == NULL )
00139             FastTimer.first = FastTimer.next;
00140           else
00141             FastTimer.previous->next = FastTimer.next;  
00142           removed = true;
00143         }
00144 
00145         if ( te->callback != NULL )
00146         {
00147           // in this callback, the callee is free to add and remove any members of this list
00148           // which might effect the first, next and previous pointers
00149           // so don't assume any of those local variables are good anymore
00150           (*te->callback)( te->id );
00151         }
00152       }
00153       
00154       // note that this has to be better than this ultimately - since
00155       // the callback routine can remove the entry without letting us know
00156       // at all.
00157       if ( !removed )
00158       {
00159         if ( FastTimer.nextTime == -1 || te->timeCurrent < FastTimer.nextTime )
00160         {
00161           FastTimer.nextTime = te->timeCurrent;
00162         }
00163         FastTimer.previous = te;
00164       }
00165 
00166       te = FastTimer.next;
00167     }
00168 
00169     if ( FastTimer.first != NULL )
00170     {
00171       // Make sure it's not too big
00172       if ( FastTimer.nextTime > 0xFF00 )
00173         FastTimer.nextTime = 0xFF00;
00174       // Make sure it's not too small
00175       if ( FastTimer.nextTime < AT91C_BASE_TC2->TC_CV + 20 )
00176         FastTimer.nextTime = AT91C_BASE_TC2->TC_CV + 20;
00177       AT91C_BASE_TC2->TC_RC = FastTimer.nextTime;
00178     }
00179     else
00180     {
00181       AT91C_BASE_TC2->TC_CCR = AT91C_TC_CLKDIS;
00182       FastTimer.running = false;
00183     }
00184 
00185 #ifdef FASTIRQ_STATS    
00186     int duration = AT91C_BASE_TC2->TC_CV - startCount;
00187     FastTimer.durationTotal += duration;  
00188     if ( duration > FastTimer.durationMax )
00189       FastTimer.durationMax = duration;
00190     if ( duration > FastTimer.durationMaxAllDay )
00191       FastTimer.durationMaxAllDay = duration;
00192 #endif
00193 
00194 #ifdef FASTIRQ_MONITOR_IO    
00195     Io_SetFalse( FASTIRQ_MONITOR_IO );
00196 #endif
00197 
00198     // AT91C_BASE_TC2->TC_CCR = AT91C_TC_CLKEN | AT91C_TC_SWTRG;
00199     FastTimer.servicing = false;
00200   }
00201 
00202   /* Clear AIC to complete ISR processing */
00203   AT91C_BASE_AIC->AIC_EOICR = 0;
00204 
00205 }
