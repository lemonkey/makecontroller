00034 /* 
00035   BASIC INTERRUPT DRIVEN DRIVER FOR USB. 
00036 
00037   This file contains all the usb components that must be compiled
00038   to ARM mode.  The components that can be compiled to either ARM or THUMB
00039   mode are contained in USB-CDC.c.
00040 
00041 */
00042 
00043 /* Scheduler includes. */
00044 #include "FreeRTOS.h"
00045 #include "task.h"
00046 #include "queue.h"
00047 
00048 /* Demo application includes. */
00049 #include "Board.h"
00050 // MakingThings: renamed
00051 #include "usb_internal.h"
00052 #include "USB-CDC.h"
00053 
00054 #define usbINT_CLEAR_MASK (AT91C_UDP_TXCOMP | AT91C_UDP_STALLSENT | AT91C_UDP_RXSETUP | AT91C_UDP_RX_DATA_BK0 | AT91C_UDP_RX_DATA_BK1 )
00055 /*-----------------------------------------------------------*/
00056 
00057 /* Messages and queue used to communicate between the ISR and the USB task. */
00058 static xISRStatus xISRMessages[ usbQUEUE_LENGTH + 1 ];
00059 extern xQueueHandle xUSBInterruptQueue;
00060 /*-----------------------------------------------------------*/
00061 
00062 /* The ISR can cause a context switch so is declared naked. */
00063 void vUSB_ISR( void ) __attribute__ ((naked));
00064 
00065 /*-----------------------------------------------------------*/
00066 
00067 
00068 void vUSB_ISR( void )
00069 {
00070   /* This ISR can cause a context switch.  Therefore a call to the 
00071   portENTER_SWITCHING_ISR() macro is made.  This must come BEFORE any 
00072   stack variable declarations. */
00073   portENTER_SWITCHING_ISR();
00074 
00075   /* Now variables can be declared. */
00076   portCHAR cTaskWokenByPost = pdFALSE; 
00077   static volatile unsigned portLONG ulNextMessage = 0;
00078   xISRStatus *pxMessage;
00079   unsigned portLONG ulRxBytes;
00080   unsigned portCHAR ucFifoIndex;
00081 
00082     /* Use the next message from the array. */
00083   pxMessage = &( xISRMessages[ ( ulNextMessage & usbQUEUE_LENGTH ) ] );
00084   ulNextMessage++;
00085 
00086     /* Save UDP ISR state for task-level processing. */
00087   pxMessage->ulISR = AT91C_BASE_UDP->UDP_ISR;
00088   pxMessage->ulCSR0 = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ];
00089 
00090     /* Clear interrupts from ICR. */
00091   AT91C_BASE_UDP->UDP_ICR = AT91C_BASE_UDP->UDP_IMR | AT91C_UDP_ENDBUSRES;
00092   
00093     
00094   /* Process incoming FIFO data.  Must set DIR (if needed) and clear RXSETUP 
00095   before exit. */
00096 
00097     /* Read CSR and get incoming byte count. */
00098   ulRxBytes = ( pxMessage->ulCSR0 >> 16 ) & usbRX_COUNT_MASK;
00099   
00100   /* Receive control transfers on endpoint 0. */
00101   if( pxMessage->ulCSR0 & ( AT91C_UDP_RXSETUP | AT91C_UDP_RX_DATA_BK0 ) )
00102   {
00103     /* Save FIFO data buffer for either a SETUP or DATA stage */
00104     for( ucFifoIndex = 0; ucFifoIndex < ulRxBytes; ucFifoIndex++ )
00105     {
00106       pxMessage->ucFifoData[ ucFifoIndex ] = AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_0 ];
00107     }
00108 
00109     /* Set direction for data stage.  Must be done before RXSETUP is 
00110     cleared. */
00111     if( ( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_RXSETUP ) )
00112     {
00113       if( ulRxBytes && ( pxMessage->ucFifoData[ usbREQUEST_TYPE_INDEX ] & 0x80 ) )
00114       {
00115         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] |= AT91C_UDP_DIR;
00116 
00117         /* Might not be wise in an ISR! */
00118         while( !(AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_DIR) );
00119       }
00120 
00121       /* Clear RXSETUP */
00122       AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] &= ~AT91C_UDP_RXSETUP;
00123 
00124       /* Might not be wise in an ISR! */
00125       while ( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_RXSETUP );
00126     }
00127     else
00128     {
00129        /* Clear RX_DATA_BK0 */
00130        AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] &= ~AT91C_UDP_RX_DATA_BK0;
00131 
00132        /* Might not be wise in an ISR! */
00133        while ( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_RX_DATA_BK0 );
00134     }
00135   }
00136   
00137   /* If we received data on endpoint 1, disable its interrupts until it is 
00138   processed in the main loop */
00139   if( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] & ( AT91C_UDP_RX_DATA_BK0 | AT91C_UDP_RX_DATA_BK1 ) )
00140   {
00141     AT91C_BASE_UDP->UDP_IDR = AT91C_UDP_EPINT1;
00142   }
00143   
00144   AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] &= ~( AT91C_UDP_TXCOMP | AT91C_UDP_STALLSENT );
00145      
00146   /* Clear interrupts for the other endpoints, retain data flags for endpoint 
00147   1. */
00148   AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] &= ~( AT91C_UDP_TXCOMP | AT91C_UDP_STALLSENT | AT91C_UDP_RXSETUP );
00149   AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] &= ~usbINT_CLEAR_MASK;
00150   AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_3 ] &= ~usbINT_CLEAR_MASK;
00151 
00152   /* Post ISR data to queue for task-level processing */
00153   cTaskWokenByPost = xQueueSendFromISR( xUSBInterruptQueue, &pxMessage, cTaskWokenByPost );
00154 
00155   /* Clear AIC to complete ISR processing */
00156   AT91C_BASE_AIC->AIC_EOICR = 0;
00157 
00158   /* Do a task switch if needed */
00159   portEXIT_SWITCHING_ISR( cTaskWokenByPost )
00160 }
