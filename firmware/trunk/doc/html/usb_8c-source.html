00054 /**
00055   Sets whether the Usb subsystem is active.
00056   @param state An integer specifying the active state of the Usb system - 1 (on) or 0 (off).
00057   @return Zero on success.
00058 */
00059 int Usb_SetActive( int state )
00060 {
00061   if ( state )
00062   {
00063     if ( Usb_Users++ == 0 )
00064     {
00065       /* Create the USB task.*/
00066       TaskCreate(  vUSBCDCTask, "USB", mainUSB_TASK_STACK, NULL, mainUSB_PRIORITY );
00067 
00068       while ( !Usb_Running )
00069         Sleep( 100 );
00070     }
00071   }
00072   else
00073   {
00074     if ( Usb_Users )
00075     {
00076       if ( --Usb_Users == 0 )
00077       {
00078         /* kill the task, someday */
00079       }
00080     }
00081   }
00082   return CONTROLLER_OK;
00083 }

00095 /** 
00096   Get a character from the USB port.  If the USB subsystem is not already initialized,
00097   a call to this function will do so.  If this is the case, this first call may take a little
00098   while (1 second?) to return.  The return value is an integer to accomodate full 8 bit
00099   communication and still provide a mechanism to return "no character" - the -1.
00100   @param timeout an integer specifying, in milliseconds, how long to wait for a character before quitting
00101   @return the latest character from the USB port, or -1 if there was none
00102 */
00103 int Usb_Get( int timeout )
00104 {
00105   if ( !Usb_Users )
00106     Usb_SetActive( 1 );
00107 
00108   return cUSBGetByte( timeout / TICKRATE );
00109 }

00127 // SLIP codes
00128 #define END             0300    // indicates end of packet 
00129 #define ESC             0333    // indicates byte stuffing 
00130 #define ESC_END         0334    // ESC ESC_END means END data byte 
00131 #define ESC_ESC         0335    // ESC ESC_ESC means ESC data byte
00132 
00133 /**
00134  */
00135 int Usb_SlipSend( char* buffer, int length )
00136 {
00137  Usb_Put( END ); // Flush out any spurious data that may have accumulated
00138 
00139   while( length-- )
00140   {
00141     switch(*buffer)
00142     {
00143       // if it's the same code as an END character, we send a special 
00144       //two character code so as not to make the receiver think we sent an END
00145       case END:
00146         Usb_Put( ESC );
00147         Usb_Put( ESC_END );
00148         break;
00149         
00150         // if it's the same code as an ESC character, we send a special 
00151         //two character code so as not to make the receiver think we sent an ESC
00152       case ESC:
00153         Usb_Put( ESC );
00154         Usb_Put( ESC_ESC );
00155         break;
00156         //otherwise, just send the character
00157       default:
00158         Usb_Put( *buffer );
00159     }
00160     buffer++;
00161   }
00162   
00163   // tell the receiver that we're done sending the packet
00164   Usb_Put( END );
00165 
00166   return CONTROLLER_OK;
00167 }

00171 int Usb_SlipReceive( char* buffer, int length )
00172 {
00173   int started = 0;
00174   int count = 0;
00175   char* bp = buffer;
00176 
00177   while ( count < length )
00178   {
00179     int incoming = Usb_Get( 1000 );
00180     if ( incoming >= 0 )
00181     {
00182       switch( incoming )
00183       {
00184         case END:
00185           if( started && count ) // it was the END byte
00186           { 
00187             // We're done now if we had received any characters
00188             return count;
00189           } 
00190           else
00191           {
00192             // skipping all starting END bytes
00193             started = true;
00194           }
00195           break;          
00196         case ESC:
00197           // if it's the same code as an ESC character, get another character,
00198           // then figure out what to store in the packet based on that.
00199           incoming = Usb_Get( 1000 );
00200           if( incoming >= 0 )
00201           {
00202             switch( incoming )
00203             {
00204               case ESC_END:
00205                 incoming = END;
00206                 break;
00207               case ESC_ESC:
00208                 incoming = ESC;
00209                 break;
00210             }
00211           }
00212           // no break here, just stick it in the packet   
00213         default:
00214           *bp++ = incoming;
00215           count++;
00216       }
00217     }
00218     else
00219       Sleep( 10 );
00220   }
00221   return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00222 }
