00025 /** \defgroup XBee
00026   Communicate with XBee (Zigbee) wireless modules via the Make Controller's serial port.
00027   
00028   XBee modules from \b MaxStream are small, cheap ($19 each), wireless \b RF (radio frequency) modules that
00029   can easily be used with the Make Controller Kit to create projects that require wireless communication.
00030   XBee modules are <b>ZigBee/IEEE 802.15.4</b> compliant and can operate in several modes:
00031   - Transparent serial port.  Messages in one side magically end up at the other endpoint.
00032   - AT command mode.  Send traditional AT commands to configure the module itself (as opposed to having
00033   the data go straight through via serial mode.
00034   - Packet (API) mode.  Lower level communication that doesn't have to wait for the module to be in
00035   AT command mode.  Check the \ref XBeePacketTypes for a description of how these packets are laid out.
00036 
00037   The general idea is that you have one XBee module connected directly to your Make Controller Kit, and then
00038   any number of other XBee modules that can communicate with it in order to get information to and from the
00039   Make Controller.
00040 
00041   XBee modules also have some digital and analog I/O right on them, which means you can directly connect
00042   sensors to the XBee modules which will both read the values and send them wirelessly.  Check the XBee doc
00043   for the appropriate commands to send in order to set this up.
00044 
00045   \ingroup Controller
00046   @{
00047 */

00049 /** 
00050   Controls the active state of the \b XBee subsystem
00051   @param state Whether this subsystem is active or not
00052   @return Zero on success.
00053 */
00054 int XBee_SetActive( int state )
00055 {
00056   if ( state != 0 && XBee == NULL ) 
00057   {
00058     XBee = Malloc( sizeof( XBee_ ) );
00059     if( XBee == NULL )
00060       return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00061     
00062     XBee->mode = XB_IDLE;
00063     if( CONTROLLER_OK != Serial_SetActive( 1 ) )
00064     {
00065       if( XBee != NULL )
00066       {
00067         Free( XBee );
00068         XBee = 0;
00069       }
00070       return CONTROLLER_ERROR_SUBSYSTEM_INACTIVE;
00071     }
00072   }
00073   if ( state == 0 && XBee != NULL ) 
00074   {
00075     Free( XBee );
00076     XBee = 0;
00077   }
00078   return CONTROLLER_OK;
00079 }

00090 /** 
00091   Receive an incoming XBee packet.
00092   This function will not block, and will return as soon as there's no more incoming data or
00093   the packet is fully received.  If the packet was not fully received, call the function repeatedly
00094   until it is.
00095 
00096   Clear out a packet before reading into it with a call to XBee_InitPacket( )
00097   @param packet The XBeePacket to receive into.
00098   @return 1 if a complete packet has been received, 0 if not.
00099   @see XBee_SetPacketApiMode( )
00100 
00101   \par Example
00102   \code
00103   // we're inside a task here...
00104   XBeePacket myPacket;
00105   XBee_InitPacket( &myPacket );
00106   while( 1 )
00107   {
00108     if( XBee_GetPacket( &myPacket ) )
00109     {
00110       // process the new packet
00111       XBee_InitPacket( &myPacket ); // then clear it out before reading again
00112     }
00113     Sleep( 10 );
00114   }
00115   \endcode
00116 */
00117 int XBee_GetPacket( XBeePacket* packet )
00118 {
00119   if( CONTROLLER_OK != XBee_SetActive( 1 ) )
00120     return CONTROLLER_ERROR_SUBSYSTEM_INACTIVE;
00121   
00122   while( Serial_GetReadable( ) )
00123   {
00124     int newChar = Serial_GetChar( );
00125     if( newChar == -1 )
00126       break;
00127 
00128     switch( packet->rxState )
00129     {
00130       case XBEE_PACKET_RX_START:
00131         if( newChar == XBEE_PACKET_STARTBYTE )
00132           packet->rxState = XBEE_PACKET_RX_LENGTH_1;
00133         break;
00134       case XBEE_PACKET_RX_LENGTH_1:
00135         packet->length = newChar << 8;
00136         packet->rxState = XBEE_PACKET_RX_LENGTH_2;
00137         break;
00138       case XBEE_PACKET_RX_LENGTH_2:
00139         packet->length += newChar;
00140         packet->rxState = XBEE_PACKET_RX_PAYLOAD;
00141         break;
00142       case XBEE_PACKET_RX_PAYLOAD:
00143         if( packet->index++ < packet->length )
00144           *packet->dataPtr++ = newChar;
00145         else
00146           packet->rxState = XBEE_PACKET_RX_CRC;
00147         packet->crc += newChar;
00148         break;
00149       case XBEE_PACKET_RX_CRC:
00150         // TODO: read the CRC here, not in the previous state
00151         packet->rxState = XBEE_PACKET_RX_START;
00152         return (packet->crc == 0xFF) ? 1 : 0;
00153     }
00154   }
00155   return 0;
00156 }

00158 /** 
00159   Send an XBee packet.
00160 
00161   Check the possible \ref XBeePacketTypes that can be sent, and populate the packet structures
00162   appropriately before sending them.
00163   @param packet The XBeePacket to send.
00164   @param datalength The length of the actual data being sent (not including headers, options, etc.)
00165   @return Zero on success.
00166   @see XBee_SetPacketApiMode( )
00167 
00168   \par Example
00169   \code
00170   XBeePacket myPacket;
00171   packet->apiId = XBEE_COMM_TX16; // we're going to send a packet with a 16-bit address
00172   packet->tx16.frameID = 0x00;
00173   packet->tx16.destination[0] = 0xFF; // 0xFFFF is the broadcast address
00174   packet->tx16.destination[1] = 0xFF;
00175   packet->tx16.options = 0x00; // no options
00176   packet->tx16.data[0] = 'A'; // now pack in the data
00177   packet->tx16.data[1] = 'B';
00178   packet->tx16.data[2] = 'C';
00179   // finally, send the packet indicating that we're sending 3 bytes of data - "ABC"
00180   XBee_SendPacket( &myPacket, 3 ); 
00181   \endcode
00182 */
00183 int XBee_SendPacket( XBeePacket* packet, int datalength )
00184 {
00185   if( CONTROLLER_OK != XBee_SetActive( 1 ) )
00186     return CONTROLLER_ERROR_SUBSYSTEM_INACTIVE;
00187   
00188   Serial_SetChar( XBEE_PACKET_STARTBYTE );
00189   int size = datalength;
00190   switch( packet->apiId )
00191   {
00192     case XBEE_COMM_RX64: //account for apiId, 8 bytes source address, signal strength, and options
00193     case XBEE_COMM_TX64: //account for apiId, frameId, 8 bytes destination, and options
00194       size += 11;
00195       break;
00196     case XBEE_COMM_RX16: //account for apiId, 2 bytes source address, signal strength, and options
00197     case XBEE_COMM_TX16: //account for apiId, frameId, 2 bytes destination, and options
00198     case XBEE_COMM_ATCOMMANDRESPONSE: // account for apiId, frameID, 2 bytes AT cmd, 1 byte status
00199       size += 5; 
00200       break;
00201     case XBEE_COMM_TXSTATUS:
00202       size = 3; // explicitly set this, since there's no data afterwards
00203       break;
00204     case XBEE_COMM_ATCOMMAND: // account for apiId, frameID, 2 bytes AT command
00205     case XBEE_COMM_ATCOMMANDQ: // same
00206       size += 4;
00207       break;
00208     default:
00209       size = 0;
00210       break;
00211   }
00212 
00213   Serial_SetChar( (size >> 8) & 0xFF ); // send the most significant bit
00214   Serial_SetChar( size & 0xFF ); // then the LSB
00215   packet->crc = 0; // just in case it hasn't been initialized.
00216   uint8* p = (uint8*)packet;
00217   while( size-- )
00218   {
00219     Serial_SetChar( *p );
00220     packet->crc += *p++;
00221   }
00222   Serial_SetChar( 0xFF - packet->crc );
00223   return CONTROLLER_OK;
00224 }

00241 /** 
00242   Set a module into AT command mode.  
00243   Because XBee modules need to wait 1 second after sending the command sequence before they're 
00244   ready to receive any AT commands, this function will block for about a second.
00245   @return An integer specifying the active state - 1 (active) or 0 (inactive).
00246 */
00247 void XBee_SetPacketApiMode( )
00248 {
00249   if( CONTROLLER_OK != XBee_SetActive( 1 ) )
00250     return;
00251   
00252   char buf[50];
00253   snprintf( buf, 50, "+++" ); // enter command mode
00254   Serial_Write( (uchar*)buf, strlen(buf), 0 );
00255   Sleep( 1025 ); // have to wait one second after +++ to actually get set to receive in AT mode
00256   
00257   snprintf( buf, 50, "ATAP %x,CN\r", 1 ); // turn API mode on, and leave command mode
00258   Serial_Write( (uchar*)buf, strlen(buf), 0 );
00259 }

00261 /** 
00262   A convenience function for creating an AT command packet.
00263   Because XBee modules need to wait 1 second after sending the command sequence before they're 
00264   ready to receive any AT commands, this function will block for about a second.
00265   @param packet The XBeePacket to create.
00266   @param frameID The frame ID for this packet that subsequent response/status messages can refer to.
00267   @param cmd The 2-character AT command.
00268   @param params A pointer to the buffer containing the data to be sent.
00269   @param datalength The number of bytes to send from the params buffer.
00270   @return An integer specifying the active state - 1 (active) or 0 (inactive).
00271   
00272   \par Example
00273   \code
00274   XBeePacket txPacket;
00275   uint8 params[5];
00276   params[0] = 0x14; // only 1 byte of data in this case
00277   XBee_CreateATCommandPacket( &txPacket, 0, "IR", params, 1 );
00278   XBee_SendPacket( &txPacket, 1 );
00279   \endcode
00280 */
00281 void XBee_CreateATCommandPacket( XBeePacket* packet, uint8 frameID, char* cmd, uint8* params, int datalength )
00282 {
00283   packet->apiId = XBEE_COMM_ATCOMMAND;
00284   packet->atCommand.frameID = frameID;
00285   uint8* p = packet->atCommand.command;
00286   *p++ = *cmd++;
00287   *p++ = *cmd++;
00288   p = packet->atCommand.parameters;
00289   while( datalength-- )
00290     *p++ = *params++;
00291 }

00293 int XBee_GetIOValues( XBeePacket* packet, int *inputs )
00294 {
00295   if( packet->apiId == XBEE_COMM_IO16 )
00296   {
00297     int i;
00298     static bool enabled;
00299     int digitalins = 0;
00300     uint8* p = packet->io16.data;
00301     int channelIndicators = (packet->io16.channelIndicators[0] << 0x08) | packet->io16.channelIndicators[1];
00302     for( i = 0; i < XBEE_INPUTS; i++ )
00303     {
00304       enabled = channelIndicators & 1;
00305       channelIndicators >>= 1;
00306       if( i < 9 ) // digital ins
00307       {
00308         if( enabled )
00309         {
00310           if( !digitalins )
00311           {
00312             int dig0 = *p++ << 0x08;
00313             digitalins = dig0 | *p++;
00314           }
00315           inputs[i] = ((digitalins >> i) & 1) * 1023;
00316         }
00317         else
00318           inputs[i] = 0;
00319       }
00320       else // analog ins
00321       {
00322         if( enabled )
00323         {
00324           int ain_msb = *p++ << 0x08;
00325           inputs[i-9] = ain_msb | *p++;
00326         }
00327         else
00328           inputs[i] = 0;
00329       }
00330     }
00331   }
00332   return 0;
00333 }
