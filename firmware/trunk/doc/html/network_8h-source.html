00027 // lwIP socket addresses are formatted this way...
00028 #define IP_ADDRESS( a, b, c, d ) ( ( (int)a << 24 ) + ( (int)b << 16 ) + ( (int)c << 8 ) + (int)d )
00029 #define IP_ADDRESS_A( address )  ( ( (int)address >> 24 ) & 0xFF )
00030 #define IP_ADDRESS_B( address )  ( ( (int)address >> 16 ) & 0xFF ) 
00031 #define IP_ADDRESS_C( address )  ( ( (int)address >>  8 ) & 0xFF )
00032 #define IP_ADDRESS_D( address )  ( ( (int)address       ) & 0xFF )
00033 // the lwIP netif structure formats the address the other way round...weird.
00034 #define NETIF_IP_ADDRESS( a, b, c, d ) ( ( (int)d << 24 ) + ( (int)c << 16 ) + ( (int)b << 8 ) + (int)a )
00035 #define NETIF_IP_ADDRESS_D( address )  ( ( (int)address >> 24 ) & 0xFF )
00036 #define NETIF_IP_ADDRESS_C( address )  ( ( (int)address >> 16 ) & 0xFF ) 
00037 #define NETIF_IP_ADDRESS_B( address )  ( ( (int)address >>  8 ) & 0xFF )
00038 #define NETIF_IP_ADDRESS_A( address )  ( ( (int)address       ) & 0xFF )
