00230 /* See the header file for descriptions of public functions. */
00231 portLONG lEMACSend( portCHAR *pcFrom, unsigned portLONG ulLength, portLONG lEndOfFrame )
00232 {
00233 static unsigned portBASE_TYPE uxTxBufferIndex = 0;
00234 portBASE_TYPE xWaitCycles = 0;
00235 portLONG lReturn = pdPASS;
00236 portCHAR *pcBuffer;
00237 unsigned portLONG ulLastBuffer, ulDataBuffered = 0, ulDataRemainingToSend, ulLengthToSend;
00238 
00239   /* If the length of data to be transmitted is greater than each individual
00240   transmit buffer then the data will be split into more than one buffer.
00241   Loop until the entire length has been buffered. */
00242   while( ulDataBuffered < ulLength )
00243   {
00244     /* Is a buffer available? */
00245     while( !( xTxDescriptors[ uxTxBufferIndex ].U_Status.status & AT91C_TRANSMIT_OK ) )
00246     {
00247       /* There is no room to write the Tx data to the Tx buffer.  Wait a
00248       short while, then try again. */
00249       xWaitCycles++;
00250       if( xWaitCycles > emacMAX_WAIT_CYCLES )
00251       {
00252         /* Give up. */
00253         lReturn = pdFAIL;
00254         break;
00255       }
00256       else
00257       {
00258         vTaskDelay( emacBUFFER_WAIT_DELAY );
00259       }
00260     }
00261   
00262     /* lReturn will only be pdPASS if a buffer is available. */
00263     if( lReturn == pdPASS )
00264     {
00265       portENTER_CRITICAL();
00266       {
00267         /* Get the address of the buffer from the descriptor, then copy 
00268         the data into the buffer. */
00269         pcBuffer = ( portCHAR * ) xTxDescriptors[ uxTxBufferIndex ].addr;
00270 
00271         /* How much can we write to the buffer? */
00272         ulDataRemainingToSend = ulLength - ulDataBuffered;
00273         if( ulDataRemainingToSend <= ETH_TX_BUFFER_SIZE )
00274         {
00275           /* We can write all the remaining bytes. */
00276           ulLengthToSend = ulDataRemainingToSend;
00277         }
00278         else
00279         {
00280           /* We can not write more than ETH_TX_BUFFER_SIZE in one go. */
00281           ulLengthToSend = ETH_TX_BUFFER_SIZE;
00282         }
00283 
00284         /* Copy the data into the buffer. */
00285         memcpy( ( void * ) pcBuffer, ( void * ) &( pcFrom[ ulDataBuffered ] ), ulLengthToSend );
00286         ulDataBuffered += ulLengthToSend;
00287 
00288         /* Is this the last data for the frame? */
00289         if( lEndOfFrame && ( ulDataBuffered >= ulLength ) )
00290         {
00291           /* No more data remains for this frame so we can start the 
00292           transmission. */
00293           ulLastBuffer = AT91C_LAST_BUFFER;
00294         }
00295         else
00296         {
00297           /* More data to come for this frame. */
00298           ulLastBuffer = 0;
00299         }
00300   
00301         /* Fill out the necessary in the descriptor to get the data sent, 
00302         then move to the next descriptor, wrapping if necessary. */
00303         if( uxTxBufferIndex >= ( NB_TX_BUFFERS - 1 ) )
00304         {       
00305           xTxDescriptors[ uxTxBufferIndex ].U_Status.status =   ( ulLengthToSend & ( unsigned portLONG ) AT91C_LENGTH_FRAME )
00306                                       | ulLastBuffer
00307                                       | AT91C_TRANSMIT_WRAP;
00308           uxTxBufferIndex = 0;
00309         }
00310         else
00311         {
00312           xTxDescriptors[ uxTxBufferIndex ].U_Status.status =   ( ulLengthToSend & ( unsigned portLONG ) AT91C_LENGTH_FRAME )
00313                                       | ulLastBuffer;
00314           uxTxBufferIndex++;
00315         }
00316   
00317         /* If this is the last buffer to be sent for this frame we can
00318         start the transmission. */
00319         if( ulLastBuffer )
00320         {
00321           AT91C_BASE_EMAC->EMAC_NCR |= AT91C_EMAC_TSTART;
00322         }
00323       }
00324       portEXIT_CRITICAL();
00325     }
00326     else
00327     {
00328       break;
00329     }
00330   }
00331 
00332   return lReturn;
00333 }

00336 /* See the header file for descriptions of public functions. */
00337 unsigned portLONG ulEMACInputLength( void )
00338 {
00339 register unsigned portLONG ulIndex, ulLength = 0;
00340 
00341   /* Skip any fragments.  We are looking for the first buffer that contains
00342   data and has the SOF (start of frame) bit set. */
00343   while( ( xRxDescriptors[ ulNextRxBuffer ].addr & AT91C_OWNERSHIP_BIT ) && !( xRxDescriptors[ ulNextRxBuffer ].U_Status.status & AT91C_SOF ) )
00344   {
00345     /* Ignoring this buffer.  Mark it as free again. */
00346     xRxDescriptors[ ulNextRxBuffer ].addr &= ~( AT91C_OWNERSHIP_BIT );    
00347     ulNextRxBuffer++;
00348     if( ulNextRxBuffer >= NB_RX_BUFFERS )
00349     {
00350       ulNextRxBuffer = 0;
00351     }
00352   }
00353 
00354   /* We are going to walk through the descriptors that make up this frame, 
00355   but don't want to alter ulNextRxBuffer as this would prevent vEMACRead()
00356   from finding the data.  Therefore use a copy of ulNextRxBuffer instead. */
00357   ulIndex = ulNextRxBuffer;
00358 
00359   /* Walk through the descriptors until we find the last buffer for this 
00360   frame.  The last buffer will give us the length of the entire frame. */
00361   while( ( xRxDescriptors[ ulIndex ].addr & AT91C_OWNERSHIP_BIT ) && !ulLength )
00362   {
00363     ulLength = xRxDescriptors[ ulIndex ].U_Status.status & emacRX_LENGTH_FRAME;
00364 
00365     /* Increment to the next buffer, wrapping if necessary. */
00366     ulIndex++;
00367     if( ulIndex >= NB_RX_BUFFERS )
00368     {
00369       ulIndex = 0;
00370     }
00371   }
00372 
00373   return ulLength;
00374 }

00377 /* See the header file for descriptions of public functions. */
00378 void vEMACRead( portCHAR *pcTo, unsigned portLONG ulSectionLength, unsigned portLONG ulTotalFrameLength )
00379 {
00380 static unsigned portLONG ulSectionBytesReadSoFar = 0, ulBufferPosition = 0, ulFameBytesReadSoFar = 0;
00381 static portCHAR *pcSource;
00382 register unsigned portLONG ulBytesRemainingInBuffer, ulRemainingSectionBytes;
00383 
00384   /* Read ulSectionLength bytes from the Rx buffers.  This is not necessarily any
00385   correspondence between the length of our Rx buffers, and the length of the
00386   data we are returning or the length of the data being requested.  Therefore, 
00387   between calls  we have to remember not only which buffer we are currently 
00388   processing, but our position within that buffer.  This would be greatly
00389   simplified if PBUF_POOL_BUFSIZE could be guaranteed to be greater than
00390   the size of each Rx buffer, and that memory fragmentation did not occur.
00391   
00392   This function should only be called after a call to ulEMACInputLength().
00393   This will ensure ulNextRxBuffer is set to the correct buffer. */
00394 
00395 
00396 
00397   /* vEMACRead is called with pcTo set to NULL to indicate that we are about
00398   to read a new frame.  Any fragments remaining in the frame we were 
00399   processing during the last call should be dropped. */
00400   if( pcTo == NULL )
00401   {
00402     /* How many bytes are indicated as being in this buffer?  If none then
00403     the buffer is completely full and the frame is contained within more
00404     than one buffer. */
00405 
00406     /* Reset our state variables ready for the next read from this buffer. */
00407         pcSource = ( portCHAR * )( xRxDescriptors[ ulNextRxBuffer ].addr & emacADDRESS_MASK );
00408         ulFameBytesReadSoFar = ( unsigned portLONG ) 0;
00409     ulBufferPosition = ( unsigned portLONG ) 0;
00410   }
00411   else
00412   {
00413     /* Loop until we have obtained the required amount of data. */
00414         ulSectionBytesReadSoFar = 0;
00415     while( ulSectionBytesReadSoFar < ulSectionLength )
00416     {
00417       /* We may have already read some data from this buffer.  How much
00418       data remains in the buffer? */
00419       ulBytesRemainingInBuffer = ( ETH_RX_BUFFER_SIZE - ulBufferPosition );
00420 
00421       /* How many more bytes do we need to read before we have the 
00422       required amount of data? */
00423             ulRemainingSectionBytes = ulSectionLength - ulSectionBytesReadSoFar;
00424 
00425       /* Do we want more data than remains in the buffer? */
00426       if( ulRemainingSectionBytes > ulBytesRemainingInBuffer )
00427       {
00428         /* We want more data than remains in the buffer so we can 
00429         write the remains of the buffer to the destination, then move
00430         onto the next buffer to get the rest. */
00431         memcpy( &( pcTo[ ulSectionBytesReadSoFar ] ), &( pcSource[ ulBufferPosition ] ), ulBytesRemainingInBuffer );
00432         ulSectionBytesReadSoFar += ulBytesRemainingInBuffer;
00433                 ulFameBytesReadSoFar += ulBytesRemainingInBuffer;
00434 
00435         /* Mark the buffer as free again. */
00436         xRxDescriptors[ ulNextRxBuffer ].addr &= ~( AT91C_OWNERSHIP_BIT );
00437 
00438         /* Move onto the next buffer. */
00439         ulNextRxBuffer++;
00440         if( ulNextRxBuffer >= NB_RX_BUFFERS )
00441         {
00442           ulNextRxBuffer = ( unsigned portLONG ) 0;
00443         }
00444 
00445         /* Reset the variables for the new buffer. */
00446         pcSource = ( portCHAR * )( xRxDescriptors[ ulNextRxBuffer ].addr & emacADDRESS_MASK );
00447         ulBufferPosition = ( unsigned portLONG ) 0;
00448       }
00449       else
00450       {
00451         /* We have enough data in this buffer to send back.  Read out
00452         enough data and remember how far we read up to. */
00453         memcpy( &( pcTo[ ulSectionBytesReadSoFar ] ), &( pcSource[ ulBufferPosition ] ), ulRemainingSectionBytes );
00454 
00455         /* There may be more data in this buffer yet.  Increment our 
00456         position in this buffer past the data we have just read. */
00457         ulBufferPosition += ulRemainingSectionBytes;
00458         ulSectionBytesReadSoFar += ulRemainingSectionBytes;
00459                 ulFameBytesReadSoFar += ulRemainingSectionBytes;
00460 
00461         /* Have we now finished with this buffer? */
00462         if( ( ulBufferPosition >= ETH_RX_BUFFER_SIZE ) || ( ulFameBytesReadSoFar >= ulTotalFrameLength ) )
00463         {
00464           /* Mark the buffer as free again. */
00465           xRxDescriptors[ ulNextRxBuffer ].addr &= ~( AT91C_OWNERSHIP_BIT );
00466 
00467           /* Move onto the next buffer. */
00468           ulNextRxBuffer++;
00469           if( ulNextRxBuffer >= NB_RX_BUFFERS )
00470           {
00471             ulNextRxBuffer = 0;
00472           }
00473   
00474           pcSource = ( portCHAR * )( xRxDescriptors[ ulNextRxBuffer ].addr & emacADDRESS_MASK );
00475           ulBufferPosition = 0;
00476         }
00477       }
00478     }
00479   }
00480 }

00484 xSemaphoreHandle xEMACInit( void )
00485 {
00486   /* Code supplied by Atmel -------------------------------*/
00487 
00488   /* Disable pull up on RXDV => PHY normal mode (not in test mode),
00489   PHY has internal pull down. */
00490   AT91C_BASE_PIOB->PIO_PPUDR = 1 << 15;
00491 
00492   #if USE_RMII_INTERFACE != 1
00493       /* PHY has internal pull down : set MII mode. */
00494       AT91C_BASE_PIOB->PIO_PPUDR = 1 << 16;
00495   #endif
00496 
00497   /* MAKINGTHINGS: ADDITION */
00498   /* Clear PB18 <=> PHY power up. */
00499   #if ( CONTROLLER_VERSION == 50 || CONTROLLER_VERSION == 95 || CONTROLLER_VERSION == 100 )
00500     AT91C_BASE_PIOB->PIO_PER = 1 << 18;
00501   AT91C_BASE_PIOB->PIO_OER = 1 << 18;
00502   AT91C_BASE_PIOB->PIO_CODR = 1 << 18;
00503   #endif
00504   #if ( CONTROLLER_VERSION == 90 )
00505     /* PB13 <=> PHY powerdown - so clearing it is power up. */
00506     AT91C_BASE_PIOB->PIO_PER = 1 << 13; 
00507     AT91C_BASE_PIOB->PIO_OER = 1 << 13;
00508     AT91C_BASE_PIOB->PIO_CODR = 1 << 13;
00509   #endif
00510 
00511   /* After PHY power up, hardware reset. */
00512   AT91C_BASE_RSTC->RSTC_RMR = emacRESET_KEY | emacRESET_LENGTH;
00513   AT91C_BASE_RSTC->RSTC_RCR = emacRESET_KEY | AT91C_RSTC_EXTRST;
00514 
00515   /* Wait for hardware reset end. */
00516   while( !( AT91C_BASE_RSTC->RSTC_RSR & AT91C_RSTC_NRSTL ) )
00517   {
00518     __asm volatile ( "NOP" );
00519   }
00520     __asm volatile ( "NOP" );
00521 
00522   /* Setup the pins. */
00523   AT91C_BASE_PIOB->PIO_ASR = emacPERIPHERAL_A_SETUP;
00524   AT91C_BASE_PIOB->PIO_PDR = emacPERIPHERAL_A_SETUP;
00525 
00526   /* Enable com between EMAC PHY.
00527 
00528   Enable management port. */
00529   AT91C_BASE_EMAC->EMAC_NCR |= AT91C_EMAC_MPE;  
00530 
00531   /* MDC = MCK/32. */
00532   AT91C_BASE_EMAC->EMAC_NCFGR |= ( 2 ) << 10; 
00533 
00534   /* Wait for PHY auto init end (rather crude delay!). */
00535   vTaskDelay( emacPHY_INIT_DELAY );
00536 
00537   /* PHY configuration. */
00538   #if USE_RMII_INTERFACE != 1
00539   {
00540     unsigned portLONG ulControl;
00541 
00542     /* PHY has internal pull down : disable MII isolate. */
00543     vReadPHY( AT91C_PHY_ADDR, MII_BMCR, &ulControl );
00544     vReadPHY( AT91C_PHY_ADDR, MII_BMCR, &ulControl );
00545     ulControl &= ~BMCR_ISOLATE;
00546     vWritePHY( AT91C_PHY_ADDR, MII_BMCR, ulControl );
00547   }
00548   #endif
00549 
00550   /* Disable management port again. */
00551   AT91C_BASE_EMAC->EMAC_NCR &= ~AT91C_EMAC_MPE;
00552 
00553   #if USE_RMII_INTERFACE != 1
00554     /* Enable EMAC in MII mode, enable clock ERXCK and ETXCK. */
00555     AT91C_BASE_EMAC->EMAC_USRIO = AT91C_EMAC_CLKEN ;
00556   #else
00557     /* Enable EMAC in RMII mode, enable RMII clock (50MHz from oscillator
00558     on ERFCK). */
00559     AT91C_BASE_EMAC->EMAC_USRIO = AT91C_EMAC_RMII | AT91C_EMAC_CLKEN ;
00560   #endif
00561 
00562   /* End of code supplied by Atmel ------------------------*/
00563 
00564   /* Setup the buffers and descriptors. */
00565   prvSetupDescriptors();
00566   
00567   /* Load our MAC address into the EMAC. */
00568   prvSetupMACAddress();
00569 
00570   /* Are we connected? */
00571   if( prvProbePHY() )
00572   {
00573     /* Enable the interrupt! */
00574     portENTER_CRITICAL();
00575     {
00576       prvSetupEMACInterrupt();
00577       vPassEMACSemaphore( xSemaphore );
00578     }
00579     portEXIT_CRITICAL();
00580   }
00581 
00582   return xSemaphore;
00583 }

00586 /* See the header file for descriptions of public functions. */
00587 void vClearEMACTxBuffer( void )
00588 {
00589 static unsigned portBASE_TYPE uxNextBufferToClear = 0;
00590 
00591   /* Called on Tx interrupt events to reset the AT91C_TRANSMIT_OK bit in each 
00592   Tx buffer within the frame just transmitted.  This marks all the buffers
00593   as available again.
00594 
00595   The first buffer in the frame should have the bit set automatically. */
00596   if( xTxDescriptors[ uxNextBufferToClear ].U_Status.status & AT91C_TRANSMIT_OK )
00597   {
00598     /* Loop through the other buffers in the frame. */
00599     while( !( xTxDescriptors[ uxNextBufferToClear ].U_Status.status & AT91C_LAST_BUFFER ) )
00600     {
00601       uxNextBufferToClear++;
00602 
00603       if( uxNextBufferToClear >= NB_TX_BUFFERS )
00604       {
00605         uxNextBufferToClear = 0;
00606       }
00607 
00608       xTxDescriptors[ uxNextBufferToClear ].U_Status.status |= AT91C_TRANSMIT_OK;
00609     }
00610 
00611     /* Start with the next buffer the next time a Tx interrupt is called. */
00612     uxNextBufferToClear++;
00613 
00614     /* Do we need to wrap back to the first buffer? */
00615     if( uxNextBufferToClear >= NB_TX_BUFFERS )
00616     {
00617       uxNextBufferToClear = 0;
00618     }
00619   }
00620 }

00623 static void prvSetupDescriptors(void)
00624 {
00625 unsigned portBASE_TYPE xIndex;
00626 unsigned portLONG ulAddress;
00627 
00628   /* Initialise xRxDescriptors descriptor. */
00629   for( xIndex = 0; xIndex < NB_RX_BUFFERS; ++xIndex )
00630   {
00631     /* Calculate the address of the nth buffer within the array. */
00632     ulAddress = ( unsigned portLONG )( pcRxBuffer + ( xIndex * ETH_RX_BUFFER_SIZE ) );
00633 
00634     /* Write the buffer address into the descriptor.  The DMA will place
00635     the data at this address when this descriptor is being used.  Mask off
00636     the bottom bits of the address as these have special meaning. */
00637     xRxDescriptors[ xIndex ].addr = ulAddress & emacADDRESS_MASK;
00638   } 
00639 
00640   /* The last buffer has the wrap bit set so the EMAC knows to wrap back
00641   to the first buffer. */
00642   xRxDescriptors[ NB_RX_BUFFERS - 1 ].addr |= emacRX_WRAP_BIT;
00643 
00644   /* Initialise xTxDescriptors. */
00645   for( xIndex = 0; xIndex < NB_TX_BUFFERS; ++xIndex )
00646   {
00647     /* Calculate the address of the nth buffer within the array. */
00648     ulAddress = ( unsigned portLONG )( pcTxBuffer + ( xIndex * ETH_TX_BUFFER_SIZE ) );
00649 
00650     /* Write the buffer address into the descriptor.  The DMA will read
00651     data from here when the descriptor is being used. */
00652     xTxDescriptors[ xIndex ].addr = ulAddress & emacADDRESS_MASK;
00653     xTxDescriptors[ xIndex ].U_Status.status = AT91C_TRANSMIT_OK;
00654   } 
00655 
00656   /* The last buffer has the wrap bit set so the EMAC knows to wrap back
00657   to the first buffer. */
00658   xTxDescriptors[ NB_TX_BUFFERS - 1 ].U_Status.status = AT91C_TRANSMIT_WRAP | AT91C_TRANSMIT_OK;
00659 
00660   /* Tell the EMAC where to find the descriptors. */
00661   AT91C_BASE_EMAC->EMAC_RBQP = ( unsigned portLONG ) xRxDescriptors;
00662   AT91C_BASE_EMAC->EMAC_TBQP = ( unsigned portLONG ) xTxDescriptors;
00663   
00664   /* Clear all the bits in the receive status register. */
00665   AT91C_BASE_EMAC->EMAC_RSR = ( AT91C_EMAC_OVR | AT91C_EMAC_REC | AT91C_EMAC_BNA );
00666 
00667   /* Enable the copy of data into the buffers, ignore broadcasts, 
00668   and don't copy FCS. */
00669   AT91C_BASE_EMAC->EMAC_NCFGR |= ( AT91C_EMAC_CAF | AT91C_EMAC_NBC | AT91C_EMAC_DRFCS);
00670 
00671   /* Enable Rx and Tx, plus the stats register. */
00672   AT91C_BASE_EMAC->EMAC_NCR |= ( AT91C_EMAC_TE | AT91C_EMAC_RE | AT91C_EMAC_WESTAT );
00673 }

00677 {
00678   /* Must be written SA1L then SA1H. */
00679   /* MakingThings: Switch variables to load from */
00680   AT91C_BASE_EMAC->EMAC_SA1L =  ( ( unsigned portLONG ) emacETHADDR3 << 24 ) |
00681                   ( ( unsigned portLONG ) emacETHADDR2 << 16 ) |
00682                   ( ( unsigned portLONG ) emacETHADDR1 << 8  ) |
00683                   emacETHADDR0;
00684 
00685   AT91C_BASE_EMAC->EMAC_SA1H =  ( ( unsigned portLONG ) emacETHADDR5 << 8 ) |
00686                   emacETHADDR4;

00690 static void prvSetupEMACInterrupt( void )
00691 {
00692   /* Create the semaphore used to trigger the EMAC task. */
00693   vSemaphoreCreateBinary( xSemaphore );
00694   if( xSemaphore )
00695   {
00696     /* We start by 'taking' the semaphore so the ISR can 'give' it when the
00697     first interrupt occurs. */
00698     xSemaphoreTake( xSemaphore, emacNO_DELAY );
00699     portENTER_CRITICAL();
00700     {
00701       /* We want to interrupt on Rx and Tx events. */
00702       AT91C_BASE_EMAC->EMAC_IER = AT91C_EMAC_RCOMP | AT91C_EMAC_TCOMP;
00703 
00704       /* Enable the interrupts in the AIC. */
00705       AT91F_AIC_ConfigureIt( AT91C_ID_EMAC, emacINTERRUPT_LEVEL, AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL, ( void (*)( void ) ) vEMACISR );
00706             AT91C_BASE_AIC->AIC_IECR = 0x1 << AT91C_ID_EMAC;
00707     }
00708     portEXIT_CRITICAL();
00709   }
00710 }

00722 static portBASE_TYPE prvProbePHY( void )
00723 {
00724 unsigned portLONG ulPHYId1, ulPHYId2, ulStatus;
00725 portBASE_TYPE xReturn = pdPASS;
00726   
00727   /* Code supplied by Atmel (reformatted) -----------------*/
00728 
00729   /* Enable management port */
00730   AT91C_BASE_EMAC->EMAC_NCR |= AT91C_EMAC_MPE;  
00731     AT91C_BASE_EMAC->EMAC_NCFGR |= ( 2 ) << 10;
00732 
00733   /* Read the PHY ID. */
00734   vReadPHY( AT91C_PHY_ADDR, MII_PHYSID1, &ulPHYId1 );
00735   vReadPHY(AT91C_PHY_ADDR, MII_PHYSID2, &ulPHYId2 );
00736 
00737   /* AMD AM79C875:
00738       PHY_ID1 = 0x0022
00739       PHY_ID2 = 0x5541
00740       Bits 3:0 Revision Number Four bit manufacturer?s revision number.
00741         0001 stands for Rev. A, etc.
00742   */
00743   if( ( ( ulPHYId1 << 16 ) | ( ulPHYId2 & 0xfff0 ) ) != MII_DM9161_ID )
00744   {
00745     /* Did not expect this ID. */
00746     xReturn = pdFAIL;
00747   }
00748   else
00749   {
00750     ulStatus = xGetLinkSpeed();
00751 
00752     if( ulStatus != pdPASS )
00753     {
00754       xReturn = pdFAIL;
00755     }
00756   }
00757 
00758   /* Disable management port */
00759   AT91C_BASE_EMAC->EMAC_NCR &= ~AT91C_EMAC_MPE; 
00760 
00761   /* End of code supplied by Atmel ------------------------*/
00762 
00763   return xReturn;
00764 }

00767 static void vReadPHY( unsigned portCHAR ucPHYAddress, unsigned portCHAR ucAddress, unsigned portLONG *pulValue )
00768 {
00769   /* Code supplied by Atmel (reformatted) ----------------------*/
00770 
00771   AT91C_BASE_EMAC->EMAC_MAN =   (AT91C_EMAC_SOF & (0x01<<30))
00772                   | (2 << 16) | (2 << 28)
00773                   | ((ucPHYAddress & 0x1f) << 23)
00774                   | (ucAddress << 18);
00775 
00776   /* Wait until IDLE bit in Network Status register is cleared. */
00777   while( !( AT91C_BASE_EMAC->EMAC_NSR & AT91C_EMAC_IDLE ) )
00778   {
00779     __asm( "NOP" );
00780   }
00781 
00782   *pulValue = ( AT91C_BASE_EMAC->EMAC_MAN & 0x0000ffff ); 
00783 
00784   /* End of code supplied by Atmel ------------------------*/
00785 }

00810 static portBASE_TYPE xGetLinkSpeed( void )
00811 {
00812   unsigned portLONG ulBMSR, ulBMCR, ulLPA, ulMACCfg, ulSpeed, ulDuplex;
00813 
00814   /* Code supplied by Atmel (reformatted) -----------------*/
00815 
00816   /* Link status is latched, so read twice to get current value */
00817   vReadPHY(AT91C_PHY_ADDR, MII_BMSR, &ulBMSR);
00818   vReadPHY(AT91C_PHY_ADDR, MII_BMSR, &ulBMSR);
00819 
00820   if( !( ulBMSR & BMSR_LSTATUS ) )
00821   { 
00822     /* No Link. */
00823     return pdFAIL;
00824   }
00825 
00826   vReadPHY(AT91C_PHY_ADDR, MII_BMCR, &ulBMCR);
00827   if (ulBMCR & BMCR_ANENABLE)
00828   {       
00829     /* AutoNegotiation is enabled. */
00830     if (!(ulBMSR & BMSR_ANEGCOMPLETE))
00831     {
00832       /* Auto-negotitation in progress. */
00833       return pdFAIL;        
00834     }   
00835 
00836     vReadPHY(AT91C_PHY_ADDR, MII_LPA, &ulLPA);
00837     if( ( ulLPA & LPA_100FULL ) || ( ulLPA & LPA_100HALF ) )
00838     {
00839       ulSpeed = SPEED_100;
00840     }
00841     else
00842     {
00843       ulSpeed = SPEED_10;
00844     }
00845 
00846     if( ( ulLPA & LPA_100FULL ) || ( ulLPA & LPA_10FULL ) )
00847     {
00848       ulDuplex = DUPLEX_FULL;
00849     }
00850     else
00851     {
00852       ulDuplex = DUPLEX_HALF;
00853     }
00854   }
00855   else
00856   {
00857     ulSpeed = ( ulBMCR & BMCR_SPEED100 ) ? SPEED_100 : SPEED_10;
00858     ulDuplex = ( ulBMCR & BMCR_FULLDPLX ) ? DUPLEX_FULL : DUPLEX_HALF;
00859   }
00860 
00861   /* Update the MAC */
00862   ulMACCfg = AT91C_BASE_EMAC->EMAC_NCFGR & ~( AT91C_EMAC_SPD | AT91C_EMAC_FD );
00863   if( ulSpeed == SPEED_100 )
00864   {
00865     if( ulDuplex == DUPLEX_FULL )
00866     {
00867       /* 100 Full Duplex */
00868       AT91C_BASE_EMAC->EMAC_NCFGR = ulMACCfg | AT91C_EMAC_SPD | AT91C_EMAC_FD;
00869     }
00870     else
00871     {         
00872       /* 100 Half Duplex */
00873       AT91C_BASE_EMAC->EMAC_NCFGR = ulMACCfg | AT91C_EMAC_SPD;
00874     }
00875   }
00876   else
00877   {
00878     if (ulDuplex == DUPLEX_FULL)
00879     {
00880       /* 10 Full Duplex */
00881       AT91C_BASE_EMAC->EMAC_NCFGR = ulMACCfg | AT91C_EMAC_FD;
00882     }
00883     else
00884     {     /* 10 Half Duplex */
00885       AT91C_BASE_EMAC->EMAC_NCFGR = ulMACCfg;
00886     }
00887   }
00888 
00889   /* End of code supplied by Atmel ------------------------*/
00890 
00891   return pdPASS;
00892 }
