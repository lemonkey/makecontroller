00097 /** \defgroup Servo
00098   The Servo Motor subsystem controls speed and position control for up to 4 standard servo motors.
00099   Standard servos have a range of motion of approximately 180 degrees, although this varies from motor to motor.
00100   Be sure to plug in the connector with the correct orientation with regard to the GND/5V signals on the board.
00101 
00102   Because not all servo motors are created equal, and not all of them can be safely driven across all 180 degrees,
00103   the default (safe) range of motion is from values 0 to 1023.  The full range of motion is from -512 to 1536, but use this
00104   extra range cautiously if you don't know how your motor can handle it.  A little gentle experimentation should do the trick.
00105   
00106   You can also specify the speed with which the motors will respond to new position commands - a high
00107   value will result in an immediate response, while a lower value can offer some smoothing when appropriate.
00108 * \ingroup AppBoard
00109 * @{
00110 */

00112 /**
00113   Lock or unlock the I/O lines used by the servos.  
00114   Sets whether the specified Servo I/O is active.
00115   @param index An integer specifying which servo (0 - 3).
00116   @param state An integer specifying the active state - 1 (active) or 0 (inactive).
00117   @return Zero on success.
00118 */
00119 int Servo_SetActive( int index, int state )
00120 {
00121   if ( index < 0 || index >= SERVO_COUNT )
00122     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00123 
00124   if ( state )
00125   {
00126     if( Servo == NULL )
00127     {
00128       int retVal = Servo_Init( );
00129       if( retVal != CONTROLLER_OK )
00130         return retVal;
00131     }
00132     return Servo_Start( index );
00133   }
00134   else
00135   {
00136     if( Servo == NULL )
00137       return -1;
00138 
00139     int retVal = Servo_Stop( index );
00140     if( retVal != CONTROLLER_OK )
00141       return retVal;
00142 
00143     if( --Servo->users <= 0 )
00144       retVal = Servo_Deinit( );
00145 
00146     return retVal;
00147   }
00148 }

00163 /** 
00164   Set the position of the specified servo motor.
00165   Most servos like to be driven within a "safe range" which usually ends up being somewhere around 110-120
00166   degrees range of motion, or thereabouts.  Some servos don't mind being driven all the way to their 180
00167   degree range of motion limit.  With this in mind, the range of values you can send to servos connected
00168   to the Make Controller Kit is as follows:
00169   - Values from 0-1023 correspond to the normal, or "safe", range of motion.  
00170   - You can also send values from -512 all the way up to 1536 to drive the servo through its full
00171   range of motion.
00172   
00173   Note that it is sometimes possible to damage your servo by driving it too far, so proceed with a bit of
00174   caution when using the extended range until you know your servos can handle it.
00175 
00176   @param index An integer specifying which servo (0 - 3).
00177   @param position An integer specifying the servo position (0 - 1023).
00178   @return status (0 = OK).
00179 */
00180 int Servo_SetPosition( int index, int position )
00181 {
00182   if ( index < 0 || index >= SERVO_COUNT )
00183     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00184 
00185   Servo_SetActive( index, 1 );
00186 
00187   if ( position < SERVO_MIN_POSITION )
00188     position = SERVO_SAFE_MIN;
00189   if ( position > SERVO_MAX_POSITION )
00190     position = SERVO_SAFE_MAX;
00191 
00192   position += SERVO_OFFSET;
00193 
00194   DisableFIQFromThumb();
00195   Servo->control[ index ]->positionRequested = position << 6;
00196   EnableFIQFromThumb();
00197 
00198   return CONTROLLER_OK;
00199 }

00201 /** 
00202   Set the speed at which a servo will move in response to a call to Servo_SetPosition().
00203   Higher values will result in a more immediate response, while lower values will more slowly step through
00204   all the intermediate values, achieving a smoother motion.  
00205   @param index An integer specifying which servo (0 - 3).
00206   @param speed An integer specifying the servo speed (0 - 1023).
00207   @return status (0 = OK).
00208 */
00209 int Servo_SetSpeed( int index, int speed )
00210 {
00211   if ( index < 0 || index >= SERVO_COUNT )
00212     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00213   
00214   Servo_SetActive( index, 1 );
00215 
00216   if ( speed < 1 )
00217     speed = 1;
00218   if( speed > 1023 )
00219     speed = 1023;
00220   DisableFIQFromThumb();
00221   Servo->control[ index ]->speed = speed << 6;
00222   EnableFIQFromThumb();
00223 
00224   return CONTROLLER_OK;
00225 }

00260 int Servo_Start( int index )
00261 {
00262   int status;
00263 
00264   if ( index < 0 || index >= SERVO_COUNT )
00265     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00266 
00267   ServoControl* sc = Servo->control[ index ];  
00268   if ( sc->users == 0 )
00269   {
00270     int io = Servo_GetIo( index );
00271 
00272     // Try to lock the servo output
00273     status = Io_Start( io, true );
00274     if ( status != CONTROLLER_OK )
00275     {
00276       // Damn
00277       Servo->users--;
00278       sc->users--;
00279       return status;
00280     }
00281 
00282     Io_PioEnable( io );
00283     Io_SetTrue( io );
00284     Io_SetOutput( io );
00285 
00286     sc->position = (SERVO_MID_POSITION + SERVO_OFFSET) << 6;
00287     sc->speed = 1023 << 6;
00288     sc->users++;
00289   }
00290 
00291   return CONTROLLER_OK;
00292 }

00294 int Servo_Stop( int index )
00295 {
00296   if ( index < 0 || index >= SERVO_COUNT )
00297     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00298 
00299   ServoControl* sc = Servo->control[ index ]; 
00300 
00301   if ( sc->users <= 0 )
00302     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00303 
00304   if ( --sc->users == 0 )
00305   {
00306     int io = Servo_GetIo( index );
00307     Io_SetInput( io );
00308     Io_Stop( io );
00309   }
00310 
00311   return CONTROLLER_OK;
00312 }

00336 int Servo_Init()
00337 {
00338   Servo = MallocWait( sizeof( Servo_ ), 100 );
00339 
00340   Servo->users = 0;
00341   int i;
00342   for( i = 0; i < SERVO_COUNT; i++ )
00343   {
00344     Servo->control[ i ] = NULL;
00345     Servo->control[ i ] = MallocWait( sizeof( ServoControl ), 100 );
00346 
00347     ServoControl* s = Servo->control[ i ];
00348     s->users = 0;
00349     s->positionRequested = (SERVO_MID_POSITION + SERVO_OFFSET) << 6;
00350     switch( i )
00351     {
00352       case 0:
00353         s->pIoBase = AT91C_BASE_PIOB;
00354         s->pin = AT91C_PIO_PB24;
00355         break;
00356       case 1:
00357         s->pIoBase = AT91C_BASE_PIOA;
00358         s->pin = AT91C_PIO_PA23;
00359         break;
00360       case 2:
00361         s->pIoBase = AT91C_BASE_PIOA;
00362         s->pin = AT91C_PIO_PA21;
00363         break;
00364       case 3:
00365         s->pIoBase = AT91C_BASE_PIOA;
00366         s->pin = AT91C_PIO_PA22;
00367         break;
00368     }
00369   }
00370 
00371   // Global init stuff
00372   Servo->state = 0;
00373   Servo->gap = 1882; // forces 64'ish cycles a second (1894 + 2012 = 3906, 3906 * 4 = 15624, 15624 * 64 = 999936)
00374   Servo->index = 0;
00375 
00376   FastTimer_InitializeEntry( &Servo->fastTimerEntry, Servo_IRQCallback, 0, 2000, true );
00377   FastTimer_Set( &Servo->fastTimerEntry );
00378 
00379   return CONTROLLER_OK;
00380 }

00395 void Servo_IRQCallback( int id )
00396 {
00397   (void)id;
00398   int period;
00399 
00400   switch ( Servo->state )
00401   {
00402     case 0:
00403     {
00404       if ( ++Servo->index >= SERVO_COUNT || Servo->index < 0 )
00405         Servo->index = 0;
00406       ServoControl* s = Servo->control[ Servo->index ];
00407       
00408       if ( s->position != s->positionRequested )
00409       {
00410         int diff = s->positionRequested - s->position;
00411         if ( diff < 0 )
00412         {
00413           s->position -= s->speed;
00414           if ( s->position < s->positionRequested )
00415             s->position = s->positionRequested;
00416         }
00417         else
00418         {
00419           s->position += s->speed;
00420           if ( s->position > s->positionRequested )
00421             s->position = s->positionRequested;
00422         }
00423       }
00424 
00425       period = s->position >> 6;
00426       if ( period >= (SERVO_MIN_POSITION + SERVO_OFFSET) && period <= (SERVO_MAX_POSITION + SERVO_OFFSET) )
00427         s->pIoBase->PIO_CODR = s->pin;
00428       else
00429         period = SERVO_MAX_POSITION;
00430       FastTimer_SetTime( &Servo->fastTimerEntry, period );
00431       Servo->state = 1;
00432       break;
00433     }
00434     case 1:
00435     {
00436       ServoControl* s = Servo->control[ Servo->index ];
00437       period = s->position >> 6;
00438       s->pIoBase->PIO_SODR = s->pin;
00439       FastTimer_SetTime( &Servo->fastTimerEntry, Servo->gap + ( SERVO_CYCLE - period ) );
00440       Servo->state = 0;
00441       break;
00442     }
00443   }
00444 }

00448 /** \defgroup ServoOSC Servo - OSC
00449   Control Servo motors with the Application Board via OSC.
00450   \ingroup OSC
00451   
00452   \section devices Devices
00453   There are 4 Servo controllers available on the Application Board, numbered 0 - 3.\n

00455   on hooking up servos to the board.
00456   
00457   \section properties Properties
00458   Each servo controller has three properties:
00459   - position
00460   - speed
00461   - active
00462 
00463   \par Position
00464   The \b position property corresponds to the position of the servo motor within its range of motion.
00465   This value can be both read and written.  
00466   \par 
00467   Most servos like to be driven within a "safe range" which usually ends up being somewhere around 110-120
00468   degrees range of motion, or thereabouts.  Some servos don't mind being driven all the way to their 180
00469   degree range of motion limit.  With this in mind, the range of values you can send to servos connected
00470   to the Make Controller Kit is as follows:
00471   - Values from 0-1023 correspond to the normal, or "safe", range of motion.  
00472   - You can also send values from -512 to 1536 to drive the servo through its full
00473   range of motion.
00474   \par
00475   Note that it is sometimes possible to damage your servo by driving it too far, so proceed with a bit of
00476   caution when using the extended range until you know your servos can handle it.
00477   \par
00478   To set the first servo to one quarter its position, send the message
00479   \verbatim /servo/0/position 256 \endverbatim
00480   Leave the argument value off to read the position of the servo:
00481   \verbatim /servo/0/position \endverbatim
00482   
00483   \par Speed
00484   The \b speed property corresponds to the speed with which the servo responds to changes 
00485   of position.

00493   \verbatim /servo/0/speed \endverbatim
