00050 /**
00051   Set the active state of the Serial subsystem.  This is automatically set to 
00052   true by any call to Serial2_Write or Serial2_Read.
00053   @param state An integer specifying the active state - 1 (on) or 0 (off).
00054   @return CONTROLLER_OK (=0) on success.
00055 */
00056 int Serial2_SetActive( int index, int state )
00057 {
00058   if ( index < 0 || index > 1 )
00059     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00060 
00061   Serial2_* s2p = &Serial2[ index ];
00062 
00063   if ( state )
00064   {
00065     if ( s2p->users++ == 0 )
00066     {
00067       Serial2_Init( index );
00068     }
00069   }
00070   else
00071   {
00072     if ( s2p->users > 0 ) 
00073     {
00074       if ( --s2p->users == 0 )
00075       {
00076         Serial2_Deinit( index );
00077       }
00078     }
00079   }
00080   return CONTROLLER_OK;
00081 }

00097 /** 
00098   Write a block of data to the Serial port.  Will block for the time specified (in ms)
00099   if the queue fills up.
00100   @param buffer A pointer to the buffer to write from.
00101   @param count An integer specifying the number of bytes to write.
00102   @param timeout Time in milliseconds to block waiting for the queue to free up. 0 means don't wait.
00103   @return status.
00104 */
00105 int Serial2_Write( int index, uchar* buffer, int count, int timeout )
00106 {
00107   if ( index < 0 || index > 1 )
00108     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00109 
00110   Serial2_* s2p = &Serial2[ index ];
00111 
00112   if ( s2p->users == 0 )
00113   {
00114     int status = Serial2_SetActive( index, 1 );
00115     if ( status != CONTROLLER_OK )
00116       return status;
00117   }
00118 
00119   // Do the business
00120   while ( count )
00121   {
00122     if( xQueueSend( s2p->transmitQueue, buffer++, timeout ) == 0 ) 
00123       return CONTROLLER_ERROR_QUEUE_ERROR; 
00124     count--;
00125   }
00126    
00127   /* Turn on the Tx interrupt so the ISR will remove the character from the 
00128   queue and send it. This does not need to be in a critical section as 
00129   if the interrupt has already removed the character the next interrupt 
00130   will simply turn off the Tx interrupt again. */ 
00131   s2p->at91UARTRegs->US_IER = AT91C_US_TXRDY; 
00132  
00133   return CONTROLLER_OK;
00134 }

00136 /** 
00137   Read data from the Serial port.  Will block for the time specified (in ms) if 
00138   there are insufficient characters.  Blocking can be avoided if Serial2_GetReadable( )
00139   is used to determine how many characters are available to read prior to calling
00140   this function.
00141   @param buffer A pointer to the buffer to read into.
00142   @param size An integer specifying the maximum number of bytes to read.
00143   @param timeout Time in milliseconds to block waiting for the specified number of bytes. 0 means don't wait.
00144   @return number of bytes read (>=0) or error <0 .
00145 */
00146 int Serial2_Read( int index, uchar* buffer, int size, int timeout )
00147 {
00148   if ( index < 0 || index > 1 )
00149     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00150 
00151   Serial2_* s2p = &Serial2[ index ];
00152 
00153   if ( s2p->users == 0 )
00154   {
00155     int status = Serial2_SetActive( index, 1 );
00156     if ( status != CONTROLLER_OK )
00157       return status;
00158   }
00159 
00160   // Do the business
00161   int count = 0;
00162   while ( count < size )
00163   {
00164     /* Place the character in the queue of characters to be transmitted. */ 
00165     if( xQueueReceive( s2p->receiveQueue, buffer++, timeout ) == 0 )
00166       break;
00167     count++;
00168   }
00169 
00170   return count;
00171 }

00459 int Serial2_Init( int index )
00460 {
00461   if ( index < 0 || index > 1 )
00462     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00463 
00464   // Do as a one-time thing
00465   if ( Serial2_Users++ == 0 )
00466   {
00467   }
00468 
00469   Serial2_* s2p = &Serial2[ index ];
00470 
00471   int id;
00472   int rxPin;
00473   int txPin;
00474   long rxPinBit;
00475   long txPinBit;
00476   if ( index == 0 )
00477   {
00478     id = AT91C_ID_US0;
00479     rxPinBit = IO_PA00_BIT;
00480     txPinBit = IO_PA01_BIT;
00481     rxPin = IO_PA00;
00482     txPin = IO_PA01;
00483     s2p->at91UARTRegs = AT91C_BASE_US0;
00484   }
00485   else
00486   {
00487     id = AT91C_ID_US1;
00488     rxPinBit = IO_PA05_BIT;
00489     txPinBit = IO_PA06_BIT;
00490     rxPin = IO_PA05;
00491     txPin = IO_PA06;
00492     s2p->at91UARTRegs = AT91C_BASE_US1;
00493   }
00494 
00495   // Enable the peripheral clock
00496   AT91C_BASE_PMC->PMC_PCER = 1 << id;
00497 
00498   // If there are no default values, get some
00499   if ( !s2p->detailsInitialized )
00500     Serial2_SetDefault( index );
00501 
00502   int status;
00503    
00504   status = Io_StartBits( rxPinBit | txPinBit, false  );
00505   if ( status != CONTROLLER_OK )
00506     return status;
00507 
00508   Io_SetPeripheralA( rxPin );
00509   Io_SetPeripheralA( txPin );
00510   Io_SetPio( rxPin, false );
00511   Io_SetPio( txPin, false );
00512   
00513   // Create the queues
00514   s2p->receiveQueue = xQueueCreate( 100, 1 ); 
00515   s2p->transmitQueue = xQueueCreate( 100, 1 ); 
00516 
00517   // Disable interrupts
00518   s2p->at91UARTRegs->US_IDR = (unsigned int) -1;
00519 
00520   // Timeguard disabled
00521   s2p->at91UARTRegs->US_TTGR = 0;
00522 
00523   // Most of the detail setting is done in here
00524   // Also Resets TXRX and re-enables RX
00525   Serial2_SetDetails( index );
00526 
00527   unsigned int mask = 0x1 << id;    
00528                         
00529   /* Disable the interrupt on the interrupt controller */         
00530   AT91C_BASE_AIC->AIC_IDCR = mask ;                   
00531   /* Save the interrupt handler routine pointer and the interrupt priority */ 
00532   AT91C_BASE_AIC->AIC_SVR[ id ] = (unsigned int)Serial2Isr_Wrapper;     
00533   /* Store the Source Mode Register */                  
00534   AT91C_BASE_AIC->AIC_SMR[ id ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 4  ;       
00535   /* Clear the interrupt on the interrupt controller */         
00536   AT91C_BASE_AIC->AIC_ICCR = mask ;         
00537 
00538   AT91C_BASE_AIC->AIC_IECR = mask;
00539 
00540   s2p->at91UARTRegs->US_IER = AT91C_US_RXRDY; 
00541 
00542 
00543   return CONTROLLER_OK;
00544 }

00552 int Serial2_SetDetails( int index )
00553 {
00554   if ( index < 0 || index > 1 )
00555     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00556 
00557   Serial2_* s2p = &Serial2[ index ];
00558 
00559   if ( s2p->users )
00560   {
00561      // Reset receiver and transmitter
00562     s2p->at91UARTRegs->US_CR = AT91C_US_RSTRX | AT91C_US_RSTTX | AT91C_US_RXDIS | AT91C_US_TXDIS; 
00563 
00564     int baudValue; 
00565 
00566     // MCK is 47923200 for the Make Controller Kit
00567 
00568     // Calculate ( * 10 )
00569     baudValue = ( MCK * 10 ) / ( s2p->baud * 16 );
00570     // Round (and / 10)
00571     if ( ( baudValue % 10 ) >= 5 ) 
00572       baudValue = ( baudValue / 10 ) + 1; 
00573     else 
00574       baudValue /= 10;
00575 
00576     s2p->at91UARTRegs->US_BRGR = baudValue; 
00577 
00578     s2p->at91UARTRegs->US_MR = 
00579       ( ( s2p->hardwareHandshake ) ? AT91C_US_USMODE_HWHSH : AT91C_US_USMODE_NORMAL ) |
00580       ( AT91C_US_CLKS_CLOCK ) |
00581       ( ( ( s2p->bits - 5 ) << 6 ) & AT91C_US_CHRL ) |
00582       ( ( s2p->stopBits == 1 ) ? AT91C_US_NBSTOP_1_BIT : AT91C_US_NBSTOP_1_BIT ) |
00583       ( ( s2p->parity == 0 ) ? AT91C_US_PAR_NONE : ( ( s2p->parity == -1 ) ? AT91C_US_PAR_ODD : AT91C_US_PAR_EVEN ) );
00584       // 2 << 14; // this last thing puts it in loopback mode
00585 
00586 
00587     s2p->at91UARTRegs->US_CR = AT91C_US_RXEN | AT91C_US_TXEN; 
00588   }
00589   return CONTROLLER_OK;
00590 }
