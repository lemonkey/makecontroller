00047 /** \defgroup serial Serial
00048   Serial provides a way to send and receive data via the serial port.
00049 
00050   The subsystem is supplied with small input and output buffers (of 100 characters each) and at present
00051   the implementation is interrupt per character so it's not particularly fast.
00052 
00053   Permits all of the common serial characteristics to be set including:
00054   - baud - the speed of the connection (110 - > 2M) in baud or raw bits per second.  9600 baud is the default setting.
00055   - bits - the size of each character (5 - 8).  8 bits is the default setting.
00056   - stopBits - the number of stop bits transmitted (1 or 2)  1 stop bit is the default setting.
00057   - parity - the parity policy (-1 is odd, 0 is none and 1 is even).  Even is the default setting.
00058   - hardwareHandshake - whether hardware handshaking is used or not.  HardwareHandshaking is off by default.
00059 
00060   \todo Convert to DMA interface for higher performance.
00061 
00062   \ingroup Core
00063   @{
00064 */

00104 /** 
00105   Write a block of data to the Serial port.  Will block for the time specified (in ms)
00106   if the queue fills up.
00107   @param buffer A pointer to the buffer to write from.
00108   @param count An integer specifying the number of bytes to write.
00109   @param timeout Time in milliseconds to block waiting for the queue to free up. 0 means don't wait.
00110   @return status.
00111 */
00112 int Serial_Write( uchar* buffer, int count, int timeout )
00113 {
00114   if ( Serial.users == 0 )
00115   {
00116     int status = Serial_SetActive( 1 );
00117     if ( status != CONTROLLER_OK )
00118       return status;
00119   }
00120 
00121   // Do the business
00122   while ( count )
00123   {
00124     if( xQueueSend( Serial.transmitQueue, buffer++, timeout / portTICK_RATE_MS  ) == 0 ) 
00125       return CONTROLLER_ERROR_QUEUE_ERROR; 
00126     count--;
00127   }
00128    
00129   /* Turn on the Tx interrupt so the ISR will remove the character from the 
00130   queue and send it. This does not need to be in a critical section as 
00131   if the interrupt has already removed the character the next interrupt 
00132   will simply turn off the Tx interrupt again. */ 
00133   AT91C_BASE_US0->US_IER = AT91C_US_TXRDY; 
00134  
00135   return CONTROLLER_OK;
00136 }

00138 /** 
00139   Read data from the Serial port.  Will block for the time specified (in ms) if 
00140   there are insufficient characters.  Blocking can be avoided if Serial_GetReadable( )
00141   is used to determine how many characters are available to read prior to calling
00142   this function.
00143   @param buffer A pointer to the buffer to read into.
00144   @param size An integer specifying the maximum number of bytes to read.
00145   @param timeout Time in milliseconds to block waiting for the specified number of bytes. 0 means don't wait.
00146   @return number of bytes read (>=0) or error <0 .
00147 */
00148 int Serial_Read( uchar* buffer, int size, int timeout )
00149 {
00150   if ( Serial.users == 0 )
00151   {
00152     int status = Serial_SetActive( 1 );
00153     if ( status != CONTROLLER_OK )
00154       return status;
00155   }
00156 
00157   // Do the business
00158   int count = 0;
00159   while ( count < size )
00160   {
00161     /* Place the character in the queue of characters to be transmitted. */ 
00162     if( xQueueReceive( Serial.receiveQueue, buffer++, timeout ) == 0 )
00163       break;
00164     count++;
00165   }
00166 
00167   return count;
00168 }

00394 int Serial_Init()
00395 {
00396   // If there are no default values, get some
00397   if ( !Serial.detailsInitialized )
00398     Serial_SetDefault();
00399 
00400   // Enable the peripheral clock
00401   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_US0;
00402 
00403   int status;
00404    
00405   status = Io_StartBits( IO_PA00_BIT | IO_PA01_BIT, false  );
00406   if ( status != CONTROLLER_OK )
00407     return status;
00408 
00409   Io_SetPeripheralA( IO_PA00 );
00410   Io_SetPeripheralA( IO_PA01 );
00411   Io_SetPio( IO_PA00, false );
00412   Io_SetPio( IO_PA01, false );
00413   
00414   // Create the queues
00415   Serial.receiveQueue = xQueueCreate( 100, 1 ); 
00416   Serial.transmitQueue = xQueueCreate( 100, 1 ); 
00417 
00418   // Disable interrupts
00419   AT91C_BASE_US0->US_IDR = (unsigned int) -1;
00420 
00421   // Timeguard disabled
00422   AT91C_BASE_US0->US_TTGR = 0;
00423 
00424   // Most of the detail setting is done in here
00425   // Also Resets TXRX and re-enables RX
00426   Serial_SetDetails();
00427 
00428   unsigned int mask = 0x1 << AT91C_ID_US0;    
00429                         
00430   /* Disable the interrupt on the interrupt controller */         
00431   AT91C_BASE_AIC->AIC_IDCR = mask ;                   
00432   /* Save the interrupt handler routine pointer and the interrupt priority */ 
00433   AT91C_BASE_AIC->AIC_SVR[ AT91C_ID_US0 ] = (unsigned int)SerialIsr_Wrapper;      
00434   /* Store the Source Mode Register */                  
00435   AT91C_BASE_AIC->AIC_SMR[ AT91C_ID_US0 ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 4  ;       
00436   /* Clear the interrupt on the interrupt controller */         
00437   AT91C_BASE_AIC->AIC_ICCR = mask ;         
00438 
00439   AT91C_BASE_AIC->AIC_IECR = mask;
00440 
00441   AT91C_BASE_US0->US_IER = AT91C_US_RXRDY; 
00442 
00443 
00444   return CONTROLLER_OK;
00445 }

00452 int Serial_SetDetails( )
00453 {
00454   if ( Serial.users )
00455   {
00456      // Reset receiver and transmitter
00457     AT91C_BASE_US0->US_CR = AT91C_US_RSTRX | AT91C_US_RSTTX | AT91C_US_RXDIS | AT91C_US_TXDIS; 
00458 
00459     int baudValue; 
00460 
00461     // MCK is 47923200 for the Make Controller Kit
00462 
00463     // Calculate ( * 10 )
00464     baudValue = ( MCK * 10 ) / ( Serial.baud * 16 );
00465     // Round (and / 10)
00466     if ( ( baudValue % 10 ) >= 5 ) 
00467       baudValue = ( baudValue / 10 ) + 1; 
00468     else 
00469       baudValue /= 10;
00470 
00471     AT91C_BASE_US0->US_BRGR = baudValue; 
00472 
00473     AT91C_BASE_US0->US_MR = 
00474       ( ( Serial.hardwareHandshake ) ? AT91C_US_USMODE_HWHSH : AT91C_US_USMODE_NORMAL ) |
00475       ( AT91C_US_CLKS_CLOCK ) |
00476       ( ( ( Serial.bits - 5 ) << 6 ) & AT91C_US_CHRL ) |
00477       ( ( Serial.stopBits == 1 ) ? AT91C_US_NBSTOP_1_BIT : AT91C_US_NBSTOP_1_BIT ) |
00478       ( ( Serial.parity == 0 ) ? AT91C_US_PAR_NONE : ( ( Serial.parity == -1 ) ? AT91C_US_PAR_ODD : AT91C_US_PAR_EVEN ) );
00479       // 2 << 14; // this last thing puts it in loopback mode
00480 
00481 
00482     AT91C_BASE_US0->US_CR = AT91C_US_RXEN | AT91C_US_TXEN; 
00483   }
00484   return CONTROLLER_OK;
00485 }
