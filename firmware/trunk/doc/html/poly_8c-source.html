00189 void PolyTask( void* p )
00190 {
00191   (void)p;
00192   int newDipSwitch;
00193   int oldDipSwitch = -1;
00194   void* data0 = NULL;
00195   void* data1 = NULL;
00196   int readConstants = false;
00197 
00198   // Make sure everything is off
00199   Poly_LineInit();
00200 
00201   while( Poly.running )
00202   {
00203     newDipSwitch = DipSwitch_GetValue( );
00204     if( newDipSwitch != oldDipSwitch )
00205     {
00206       int newMode0;
00207       int newMode1;
00208       
00209       if ( !( newDipSwitch & 0x80 ) )
00210       {
00211         newMode0 = 0;
00212         newMode1 = 0;
00213       }
00214       else
00215       {
00216         newMode0 = (newDipSwitch & 0x0E) >> 1;
00217         newMode1 = (newDipSwitch & 0x70) >> 4;
00218       }
00219 
00220       readConstants = ( newDipSwitch & 0x01 ) ? 1 : 0;
00221 
00222       if( newMode0 != Poly.mode0 )
00223       {
00224         Poly_DeInit( Poly.mode0, data0 );
00225         data0 = Poly_Init( newMode0, 0 );
00226         Poly.mode0 = newMode0;
00227       }
00228 
00229       if( newMode1 != Poly.mode1 )
00230       {
00231         Poly_DeInit( Poly.mode1, data1 );
00232         data1 = Poly_Init( newMode1, 1 );
00233         Poly.mode1 = newMode1;
00234       }
00235 
00236       oldDipSwitch = newDipSwitch;
00237     }
00238 
00239     Poly_Run( Poly.mode0, data0, POLY_SLEEP, readConstants );
00240     Poly_Run( Poly.mode1, data1, POLY_SLEEP, readConstants );
00241 
00242     Sleep( POLY_SLEEP );
00243   }
00244 
00245   TaskDelete( 0 );
00246 }

00351 void Timer_Run( void* timerVP, int ms, int readConstants )
00352 {
00353   TimerStruct* tp = (TimerStruct*)timerVP;
00354   // Deal with current pulse
00355   if ( !tp->timerState  ) // if the timer is not expired
00356   { 
00357     tp->timerSoFar += ms;
00358 
00359     if ( tp->timerSoFar > tp->timerDuration )
00360       tp->timerSoFar = tp->timerDuration ;
00361 
00362     // Calculate servo positions
00363     int position = 2048 * tp->timerSoFar / tp->timerDuration;
00364     Servo_SetPosition( tp->baseAux, ( position >> 2 ) );
00365     int position1 = ( position < 1023) ? position : 2048 - position;
00366     Servo_SetPosition( tp->baseAux + 1, position1 );
00367     
00368     if ( !tp->timerStateHalf && tp->timerSoFar >= tp->timerDurationHalf )
00369     {
00370       DigitalOut_SetValue( tp->baseOut + 3, 1 );  // set the "timer half-expired" output
00371       AppLed_SetState( tp->baseAux + 1, 1 ); 
00372       tp->timerStateHalf = 1;    // and set the flag accordingly
00373     }
00374     
00375     if ( !tp->timerStateQuarter && tp->timerSoFar >= tp->timerDurationQuarter )
00376     {
00377       DigitalOut_SetValue( tp->baseOut + 2, 1 );  // set the "timer quarter-expired" output
00378       tp->timerStateQuarter = 1;
00379     }
00380     
00381     if ( tp->timerSoFar >= tp->timerDuration  )
00382     {
00383       DigitalOut_SetValue( tp->baseOut, 0 );  // turn off the "time remaining" output
00384       DigitalOut_SetValue( tp->baseOut + 1, 1 );  // turn on the "time expired" output
00385       AppLed_SetState( tp->baseAux, 0 ); 
00386       tp->timerState = 1;      
00387     }
00388   }
00389   
00390   // if we're authorized to read constants from inputs
00391   if ( readConstants )
00392   {
00393     // Deal with changing time 
00394     int newPotValue = AnalogIn_GetValue( tp->baseIn + 3 );    // assuming the pot jumper is connected...
00395     if ( newPotValue != tp->lastPotValue )  // if the value has changed
00396     {
00397       tp->lastPotValue = newPotValue; // save it for next time
00398       
00399       // need to adjust scale - make sure we're not dividing by zero
00400       // potvalue squared seems right, but then divided by something that will
00401       // give good control.  If the poly system works on 5ms intervals, squared
00402       // and divided by 1000, gives 2.5s at half, 10s at full on an exponential
00403       // scale.
00404       tp->timerDuration = newPotValue * newPotValue / 50;
00405       if( tp->timerDuration == 0 )
00406         tp->timerDuration = 1;
00407       
00408       tp->timerDurationHalf = tp->timerDuration / 2;
00409       if( tp->timerDurationHalf == 0 )
00410         tp->timerDurationHalf = 1;
00411       
00412       tp->timerDurationQuarter = tp->timerDuration - tp->timerDuration / 4;
00413       if( tp->timerDurationQuarter == 0 )
00414         tp->timerDurationQuarter = 1;
00415       
00416       if ( tp->timerState && tp->timerSoFar > tp->timerDuration )
00417         tp->timerSoFar = tp->timerDuration - 1;
00418     }
00419 
00420     // Read sustain mode
00421     tp->sustainMode = DigitalIn_GetValue( tp->baseIn + 2 );
00422   }
00423   else
00424   {
00425     // If the switch is off, save the constants
00426     if ( tp->readConstants )
00427       Timer_SaveConstants( tp );
00428   }
00429   tp->readConstants = readConstants;
00430   
00431   // Check for new triggers
00432   int trigger = 0;
00433   int newIn0 = DigitalIn_GetValue( tp->baseIn ); 
00434   if ( newIn0 != tp->lastIn0 || tp->sustainMode )
00435   {
00436     if ( newIn0 == 1 )    // if the main trigger input is high
00437       trigger = 1;
00438     tp->lastIn0 = newIn0;
00439   }
00440   
00441   int newIn1 = DigitalIn_GetValue( tp->baseIn + 1 );
00442   if ( newIn1 != tp->lastIn1 || tp->sustainMode )
00443   {
00444     if ( newIn1 == 0 )  // if the "inverted trigger" drops to 0
00445       trigger = 1;
00446     tp->lastIn1 = newIn1;
00447   }
00448 
00449   if ( trigger )
00450   {
00451     // set all the outputs to off except the "time expired" output
00452     DigitalOut_SetValue( tp->baseOut, 1 );
00453     AppLed_SetState( tp->baseAux, 1 );
00454     AppLed_SetState( tp->baseAux + 1, 0 );
00455     int i;
00456     for( i = (tp->baseOut + 1); i < (tp->baseOut + 4); i++ )
00457       DigitalOut_SetValue( i, 0 );
00458     tp->timerState = 0;
00459     tp->timerStateHalf = 0;
00460     tp->timerStateQuarter = 0;
00461     tp->timerSoFar = 0;   // reset the timer count
00462   }
00463 }

00553 void Follower_Run( void* followerVP, int ms, int readConstants )
00554 {
00555   FollowerStruct* fp = (FollowerStruct*)followerVP;
00556   
00557   if ( readConstants )
00558   {
00559     // For pair zero, read the rate
00560     int newRateValue = AnalogIn_GetValue( fp->baseIn + 3 );
00561     if ( newRateValue != fp->lastRateValue )
00562     {
00563       fp->lastRateValue = newRateValue;
00564       if ( newRateValue == 0 )
00565         newRateValue = 1;
00566       if ( newRateValue > 1020 )
00567         fp->rate = 0;
00568       else
00569         fp->rate = ( 10 * 1024.0 ) / (newRateValue * newRateValue );
00570     }    
00571     fp->peak = DigitalIn_GetValue( fp->baseIn + 2 );
00572   }
00573   else
00574   {
00575     if ( fp->readConstants )
00576       Follower_SaveConstants( fp );
00577   }
00578   fp->readConstants = readConstants;
00579 
00580   int set = DigitalIn_GetValue( fp->baseIn + 1 );
00581 
00582   // Now read the position
00583   int input = AnalogIn_GetValue( fp->baseIn );
00584 
00585   int diff = input - fp->output;
00586   float change = fp->rate * (float)ms;
00587         
00588   if( diff != 0 ) // if the input has changed
00589   {
00590     if( diff > 0 )  // if the input is higher than last time
00591     {
00592       fp->output += change; // increment the output by the rate
00593       if( fp->output > input || fp->peak || set )
00594         fp->output = input; // just set it to the input if it would otherwise be too high
00595       if ( diff > 1  )
00596       {
00597         AppLed_SetState( fp->baseAux, 1 );     
00598         AppLed_SetState( fp->baseAux + 1, 0 );     
00599       }
00600       PwmOut_SetAll( fp->baseOut + 1, diff, 0, 1 );
00601     }
00602     else  // the input is lower than last time
00603     {
00604       fp->output -= change; 
00605       if( fp->output < input || set )
00606         fp->output = input;
00607       if ( diff < -1  )
00608       {
00609         AppLed_SetState( fp->baseAux + 1, 1 );     
00610         AppLed_SetState( fp->baseAux, 0 );     
00611       }
00612       PwmOut_SetAll( fp->baseOut + 1, -diff, 1, 0 );
00613     }
00614 
00615     int diff2 = diff / 2;
00616     Servo_SetPosition( fp->baseAux + 1, 512 + diff2 );
00617              
00618     int out = (int)fp->output;
00619     PwmOut_SetDuty( fp->baseOut, out );
00620     Servo_SetPosition( fp->baseAux, out );
00621   }
00622   else
00623   {
00624     AppLed_SetState( fp->baseAux, 0 );
00625     AppLed_SetState( fp->baseAux + 1, 0 );
00626     PwmOut_SetDuty( fp->baseOut + 1, 0 );
00627     Servo_SetPosition( fp->baseAux + 1, 512 );
00628   }
00629 
00630 }

00700   Eeprom_Read( EEPROM_POLY_0_OSCILLATOR_PERIODOFF + ( bank * 8 ), (unsigned char*)&op->periodOff, 4 );
00701   
00702   for( i = op->baseOut; i < (op->baseOut + 4); i++)
00703     DigitalOut_SetValue( i, 0 );
00704 
00705   AppLed_SetState( op->baseAux, 1 );
00706   AppLed_SetState( op->baseAux + 1, 0 );
00707   Servo_SetPosition( op->baseAux, 512 );
00708 
00709   return (void*)op;
00710 }

00712 void Oscillator_Run( void* oscillatorVP, int ms, int readConstants )
00713 {
00714   OscillatorStruct* op = (OscillatorStruct*)oscillatorVP;
00715   
00716   if ( readConstants )
00717   {
00718     int periodOffRaw = AnalogIn_GetValue( op->baseIn + 3 );
00719     if ( periodOffRaw != op->periodOffRaw )
00720     {
00721       op->periodOffRaw = periodOffRaw;
00722       op->periodOff = (float)( periodOffRaw * periodOffRaw ) / 50.0;
00723     }
00724     int periodOnRaw = AnalogIn_GetValue( op->baseIn + 2 );
00725     if ( periodOnRaw != op->periodOnRaw )
00726     {
00727       op->periodOnRaw = periodOnRaw;
00728       if ( periodOnRaw < 2 )
00729         op->periodOn = op->periodOff;
00730       else
00731         op->periodOn = (float)( periodOnRaw * periodOffRaw ) / 50.0;
00732     }
00733     else
00734     {
00735       if ( periodOnRaw < 2 )
00736         op->periodOn = op->periodOff;
00737     }
00738   }
00739   else
00740   {
00741     if ( op->readConstants )
00742       Oscillator_SaveConstants( op );
00743   }
00744   op->readConstants = readConstants;
00745 
00746   int setOn = DigitalIn_GetValue( op->baseIn + 0 );
00747   int setOff = DigitalIn_GetValue( op->baseIn + 1 );
00748 
00749   int stateNew = op->state;
00750   if ( setOn )
00751   {
00752     stateNew = 1;
00753     op->current = 0.0;
00754   }
00755   if ( setOff )
00756   {
00757     stateNew = 0;
00758     op->current = 0.0;
00759   }
00760 
00761   op->current += ms;
00762  
00763   if ( op->current > ( ( stateNew ) ? op->periodOn : op->periodOff ) )
00764   {
00765     stateNew = !stateNew;
00766     op->current = 0.0;
00767   }
00768 
00769   if ( stateNew != op->state )
00770   {
00771     if ( stateNew )
00772     {
00773       AppLed_SetState( op->baseAux, 0 );     
00774       AppLed_SetState( op->baseAux + 1, 1 );     
00775       DigitalOut_SetValue( op->baseOut, 1 );
00776       DigitalOut_SetValue( op->baseOut + 1, 0 );
00777     }
00778     else
00779     {
00780       AppLed_SetState( op->baseAux, 1 );     
00781       AppLed_SetState( op->baseAux + 1, 0 );     
00782       DigitalOut_SetValue( op->baseOut, 0 );
00783       DigitalOut_SetValue( op->baseOut + 1, 1 );
00784       Servo_SetPosition( op->baseAux, 0 );
00785       Servo_SetPosition( op->baseAux + 1, 0 );
00786     }
00787   }
00788 
00789   if ( stateNew )
00790   {
00791     int position = 2048 * op->current / op->periodOn;
00792     Servo_SetPosition( op->baseAux, ( position >> 2 ) );
00793     int position1 = ( position < 1023) ? position : 2048 - position;
00794     Servo_SetPosition( op->baseAux + 1, position1 );
00795   }
00796 
00797   op->state = stateNew;
00798 }

00801 {
00802   OscillatorStruct* op = (OscillatorStruct*)Oscillator;
00803   if ( op != NULL )
00804   {
00805     int i;
00806 
00807     if ( op->readConstants )
00808       Oscillator_SaveConstants( op );
00809 
00810     DigitalIn_SetActive( 0, 0 );
00811     DigitalIn_SetActive( 1, 0 );
00812     AnalogIn_SetActive( 2, 0 );
00813     AnalogIn_SetActive( 3, 0 );
00814 
00815     for( i = op->baseOut; i < op->baseOut + 4; i++ )
00816     {
00817       DigitalOut_SetValue( i, 0 );
00818       DigitalOut_SetActive( i, 0 );
00819     }
00820 
00821     for( i = op->baseAux; i < op->baseAux + 2; i++ )
00822     {
00823       AppLed_SetState( i, 0 );
00824       AppLed_SetActive( i, 0 );
00825       Servo_SetPosition( i, 0 );
00826       Servo_SetActive( i, 0 );
00827     }
00828 
00829     Free( op );
00830   }
00831 }
