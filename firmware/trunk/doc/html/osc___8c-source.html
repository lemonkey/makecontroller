00307 int Osc_ReceivePacket( int channel, char* packet, int length )
00308 {
00309   // Got a packet.  Unpacket.
00310 
00311   switch ( *packet )
00312   {
00313     case '/':
00314       Osc_ReceiveMessage( channel, packet, length );
00315       break;
00316     case '#':
00317       if ( strcmp( packet, "#bundle" ) == 0 )
00318       {
00319         // skip bundle text and timetag
00320         packet += 16;
00321         length -= 16;
00322         while ( length > 0 )
00323         {
00324           // read the length (pretend packet is a pointer to integer)
00325           int messageLength = Osc_EndianSwap( *((int*)packet) );
00326           packet += 4;
00327           length -= 4;
00328           if ( messageLength <= length )
00329             Osc_ReceivePacket( channel, packet, messageLength );
00330           length -= messageLength;
00331           packet += messageLength;
00332         }
00333       }
00334       break;
00335     default:
00336       // Something else?
00337       Osc_CreateMessageNoLock( channel, "/error", ",s", "Packet Error" );
00338       break;
00339   }
00340 
00341   return Osc_SendPacket( channel );
00342 }

00344 int Osc_ReceiveMessage( int channel, char* message, int length )
00345 {
00346   // Got a packet.  Unpacket.
00347 
00348   // Confirm it's a message
00349   if ( *message == '/' )
00350   {
00351     // Plain message
00352     char* nextSlash = strchr( message + 1, '/' );
00353     if ( nextSlash != NULL )
00354       *nextSlash = 0;
00355     int i;
00356     int count = 0;
00357     for ( i = 0; i <= Osc.subsystemHighest; i++ )
00358     {
00359       OscSubsystem* sub = &Osc.subsystem[ i ];
00360       if ( Osc_PatternMatch( message + 1, sub->name ) )
00361       {
00362         count++;
00363         if ( nextSlash )
00364           (*sub->receiveMessage)( channel, nextSlash + 1, length - ( nextSlash - message ) - 1 );
00365         else
00366           (*sub->receiveMessage)( channel, 0, 0 );          
00367       }
00368     }
00369     if ( count == 0 )
00370     {
00371       char s[ OSC_SCRATCH_SIZE ];
00372       snprintf( s, OSC_SCRATCH_SIZE, "No Subsystem Match - %s", message + 1 ); 
00373       Osc_CreateMessageNoLock( channel, "/error", ",s", s );
00374     }
00375   }
00376   else
00377   {
00378     return CONTROLLER_ERROR_BAD_DATA;
00379   }
00380   return CONTROLLER_OK;
00381 }

00429 // An OSC message has arrived, check it out... although, don't change it - others need
00430 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00431 // messages in it, they will be unpacked elsewhere.  We'll only get 
00432 // individual messages here.
00433 // What is passed is the part after the first subsystem address.
00434 // So "/dipswitch/active 1" would be passed to this routine as "active 1"
00435 // Not literally like this, of course, but as OSC messages
00436 // In order to do anything, we need some extra information from 
00437 // the subsystem like what the names of its properties are, what 
00438 // the getters and setters are, etc.
00439 int Osc_IntReceiverHelper( int channel, char* message, int length, 
00440                            char* subsystemName, 
00441                            int (*propertySet)( int property, int value ),
00442                            int (*propertyGet)( int property ),
00443                            char* propertyNames[] )
00444 {
00445   if ( message == NULL )
00446   {
00447     return Osc_SubsystemError( channel, subsystemName, "No property specified" );
00448   }
00449 
00450   int propertyIndex = Osc_PropertyLookup( propertyNames, message );
00451   if ( propertyIndex == -1 )
00452   {
00453     char s[ OSC_SCRATCH_SIZE ];
00454     snprintf( s, OSC_SCRATCH_SIZE, "Unknown property - %s", message );
00455     return Osc_SubsystemError( channel, subsystemName, s );
00456   }
00457 
00458 /*
00459   int bits;
00460   int number = Osc_NumberMatch( indexCount, message, &bits );
00461   if ( number == -1 && bits == -1 )
00462     return Osc_SubsystemError( channel, subsystemName, "Bad index" );
00463 */
00464   
00465   // Sometime after the address, the data tag begins - this is the description 
00466   // of the data in the rest of the message.  It starts with a comma.  Return
00467   // where it is into 'type'.  If there is no comma, this is bad.
00468   char* type = Osc_FindDataTag( message, length );
00469   if ( type == NULL )
00470     return Osc_SubsystemError( channel, subsystemName, "No type tag" );
00471 
00472   // We can tell if there's data by seeing if the character after the comma
00473   // is a zero or not.
00474   if ( type[ 1 ] == 'i' || type[ 1 ] == 'f' )
00475   {
00476     // If there was int or float data, it was a WRITE.
00477     // So, sort of scanff-like, go get the data.  Here we pass in where the data is
00478     // thanks to the previous routine and then specify what we expect to find there
00479     // in tag terms (i.e. "i", "s", "f" and others).  Finally we pass in a set 
00480     // of pointers to the data types we want to extract.  Osc_ExtractData()
00481     // will rummage around in the message magically grabbing values for you,
00482     // reporting how many it got.  It will convert ints and floats if necessary.
00483     int value;
00484     int count = Osc_ExtractData( type, "i", &value );
00485     if ( count != 1 )
00486       return Osc_SubsystemError( channel, subsystemName, "incorrect data - need an int or float" );
00487   
00488     (*propertySet)( propertyIndex, value );
00489   }
00490   else
00491   {
00492     // No data, then.  I guess it was a read.  The XXXXOsc getters
00493     // take the channel number and use it to call
00494     // Osc_CreateMessage() which adds a new message to the outgoing
00495     // stack
00496     int value = (*propertyGet)( propertyIndex );
00497     char address[ OSC_SCRATCH_SIZE ]; 
00498     snprintf( address, OSC_SCRATCH_SIZE, "/%s/%s", subsystemName, propertyNames[ propertyIndex ] ); 
00499     Osc_CreateMessageNoLock( channel, address, ",i", value );
00500   }
00501 
00502   return CONTROLLER_OK;
00503 }

00579 // An OSC message has arrived, check it out... although, don't change it - others need
00580 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00581 // messages in it, they will be unpacked elsewhere.  We'll only get 
00582 // individual messages here.
00583 // What is passed is the part after the first subsystem address.
00584 // So "/appled/0/on 1" would be passed to this routine as "0/on 1"
00585 // Not literally like this, of course, but as OSC messages
00586 // In order to do anything, we need some extra information from 
00587 // the subsystem like what the names of its properties are, what 
00588 // the getters and setters are, etc.
00589 // It's great to have this code in one central place.  It will be easier 
00590 // to effect change from here.o
00591 int Osc_IndexIntReceiverHelper( int channel, char* message, int length, 
00592                                 int indexCount, char* subsystemName, 
00593                                 int (*propertySet)( int index, int property, int value ),
00594                                 int (*propertyGet)( int index, int property ),
00595                                 char* propertyNames[] )
00596 {
00597   // Look for the next slash - being the one that separates the index
00598   // from the property.  Note that this won't go off on a search through the buffer
00599   // since there will soon be a string terminator (i.e. a 0)
00600   char* prop = strchr( message, '/' );
00601   if ( prop == NULL )
00602    return Osc_SubsystemError( channel, subsystemName, "Format - need /subsystem/index/property" );
00603 
00604   // Now that we know where the property is, we can see if we can find it.
00605   // This is a little cheap, since we're also implying that there are no 
00606   // more address terms after the property.  That is, if testing for "speed", while
00607   // "speed" would match, "speed/other_stuff" would not.
00608   int propertyIndex = Osc_PropertyLookup( propertyNames, prop + 1 );
00609   if ( propertyIndex == -1 )
00610   {
00611     char s[ OSC_SCRATCH_SIZE ];
00612     snprintf( s, OSC_SCRATCH_SIZE, "Unknown property - %s", prop + 1 );
00613     return Osc_SubsystemError( channel, subsystemName, s );
00614   }
00615 
00616   // Here's where we try to understand what index we got.  In the world of 
00617   // OSC, this could be a pattern.  So while we could get "0/speed" we could 
00618   // also get "*/speed" or "[0-4]/speed".  This is kind of a drag, but it is 
00619   // quite nice from the user's perspective.
00620   // So to deal with this take a look at the text "0" or "{1,2}" or whatever
00621   // and produce either a nice integer in the simplest case or a set of bits 
00622   // where each bit corresponds to one of the indicies.  Clearly we don't have
00623   // to go crazy, since there are only a small finite number of them.
00624   // Osc_NumberMatch() does the work for us, producing either number = -1 and 
00625   // bits == -1 if there was no index match, or number != -1 for there was a single
00626   // number, or bits != -1 if there were several.
00627 
00628   // note that we tweak the string a bit here to make sure the next '/' is not 
00629   // mixed up with this.  Insert a string terminator.
00630   *prop = 0;
00631 
00632   int bits;
00633   int number = Osc_NumberMatch( indexCount, message, &bits );
00634   if ( number == -1 && bits == -1 )
00635     return Osc_SubsystemError( channel, subsystemName, "Bad index" );
00636   
00637   // We tweaked the '/' before - now put it back
00638   *prop = '/';
00639 
00640   // Sometime after the address, the data tag begins - this is the description 
00641   // of the data in the rest of the message.  It starts with a comma.  Return
00642   // where it is into 'type'.  If there is no comma, this is bad.
00643   char* type = Osc_FindDataTag( message, length );
00644   if ( type == NULL )
00645     return Osc_SubsystemError( channel, subsystemName, "No type tag" );
00646 
00647   // We can tell if there's data by seeing if the character after the comma
00648   // is a zero or not.
00649   if ( type[ 1 ] == 'i' || type[ 1 ] == 'f' )
00650   {
00651     // If there was int or float data, it was a WRITE.
00652     // So, sort of scanff-like, go get the data.  Here we pass in where the data is
00653     // thanks to the previous routine and then specify what we expect to find there
00654     // in tag terms (i.e. "i", "s", "f" and others).  Finally we pass in a set 
00655     // of pointers to the data types we want to extract.  Osc_ExtractData()
00656     // will rummage around in the message magically grabbing values for you,
00657     // reporting how many it got.  It will convert ints and floats if necessary.
00658     int value;
00659     int count = Osc_ExtractData( type, "i", &value );
00660     if ( count != 1 )
00661       return Osc_SubsystemError( channel, subsystemName, "incorrect data - need an int or float" );
00662   
00663     // Now with the data we need to decide what to do with it.
00664     // Is there one or many here?
00665     if ( number != -1 )
00666       (*propertySet)( number, propertyIndex, value );
00667     else
00668     {
00669       int index = 0;
00670       while ( bits > 0 && index < indexCount )
00671       { 
00672         if ( bits & 1 )
00673           (*propertySet)( index, propertyIndex, value );
00674         bits >>= 1;
00675         index++;
00676       }
00677     }
00678   }
00679   else
00680   {
00681     // No data, then.  I guess it was a read.  The XXXXOsc getters
00682     // take the channel number and use it to call
00683     // Osc_CreateMessage() which adds a new message to the outgoing
00684     // stack
00685     if ( number != -1 )
00686     {
00687       int value = (*propertyGet)( number, propertyIndex );
00688       char address[ OSC_SCRATCH_SIZE ]; 
00689       snprintf( address, OSC_SCRATCH_SIZE, "/%s/%d/%s", subsystemName, number, propertyNames[ propertyIndex ] ); 
00690       Osc_CreateMessageNoLock( channel, address, ",i", value );
00691     }
00692     else
00693     {
00694       int index = 0;
00695       while ( bits > 0 && index < indexCount )
00696       { 
00697         if ( bits & 1 )
00698         {
00699           int value = (*propertyGet)( index, propertyIndex );
00700           char address[ OSC_SCRATCH_SIZE ]; 
00701           snprintf( address, OSC_SCRATCH_SIZE, "/%s/%d/%s", subsystemName, index, propertyNames[ propertyIndex ] ); 
00702           Osc_CreateMessageNoLock( channel, address, ",i", value );
00703         }
00704         bits >>= 1;
00705         index++;
00706       }
00707     }
00708   }
00709 
00710   return CONTROLLER_OK;
00711 }

00713 int Osc_ExtractData( char* buffer, char* format, ... )
00714 {
00715   // Set up to iterate through the arguments
00716   va_list args;
00717   va_start( args, format );
00718   int count = 0;
00719 
00720   // figure out where the data starts
00721   int tagLen = strlen( buffer ) + 1;
00722   int pad = tagLen % 4;
00723   if ( pad != 0 )
00724     tagLen += ( 4 - pad );
00725   char* data = buffer + tagLen;
00726 
00727   // Going to be walking the tag string, the format string and the data
00728   char* fp;
00729   char* tp = buffer + 1; // need to skip the comma ','
00730   bool cont = true;
00731   for ( fp = format; *fp && cont; fp++ )
00732   {
00733     cont = false;
00734     switch ( *fp )
00735     {
00736       case 'i':
00737         if ( *tp == 'i' )
00738         {
00739           *(va_arg( args, int* )) = Osc_ReadInt( data );
00740           data += 4;
00741           count++;
00742           cont = true;
00743         }
00744         if ( *tp == 'f' )
00745         {
00746           *(va_arg( args, int* )) = (int)Osc_ReadFloat( data );
00747           data += 4;
00748           count++;
00749           cont = true;
00750         }
00751         
00752         break;
00753       case 'f':
00754         if ( *tp == 'f' )
00755         {
00756           *(va_arg( args, float* )) = Osc_ReadFloat( data );
00757           data += 4;
00758           count++;
00759           cont = true;
00760         }
00761         if ( *tp == 'i' )
00762         {
00763           *(va_arg( args, float* )) = (float)Osc_ReadInt( data );
00764           data += 4;
00765           count++;
00766           cont = true;
00767         }
00768         break;
00769       case 's':
00770         if ( *tp == 's' )
00771         {
00772           *(va_arg( args, char** )) = data;
00773           int len = strlen( data ) + 1;
00774           int pad = len % 4;
00775           if ( pad != 0 )
00776             len += ( 4 - pad );
00777           data += len;
00778           count++;
00779           cont = true;
00780         }
00781         break;
00782     }
00783     tp++;
00784   }
00785 
00786   //va_end( args );
00787 
00788   return count;
00789 }

00805 /**
00806   Osc_CreateMessageNoLock
00807   Must put the "," as the first format letter
00808   */
00809 int Osc_CreateMessageNoLock( int channel, char* address, char* format, ... )
00810 {
00811   if ( Osc.sendMessage == NULL )
00812     return CONTROLLER_ERROR_RESOURCE_MISSING;
00813 
00814   if ( format == NULL || *format != ',' )
00815     return CONTROLLER_ERROR_BAD_FORMAT;
00816   
00817   // try to send this message - if there's a problem somewhere, 
00818   // send the existing buffer - freeing up space, then try (once) again.
00819   int count = 0;
00820   char *bp;
00821   do
00822   {  
00823     count++;
00824 
00825     char* buffer = Osc.bufferPointer;
00826     int length = Osc.bufferRemaining;
00827   
00828     bp = buffer;
00829   
00830     // First message in the buffer?
00831     if ( bp == Osc.buffer )
00832     {
00833       bp = Osc_CreateBundle( bp, &length, 0, 0 );
00834       if ( bp == NULL )
00835         return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00836     }
00837   
00838     // Make room for the new message
00839     int* lp = (int *)bp;
00840     bp += 4;
00841     length -= 4;
00842 
00843     // remember the start of the message
00844     char* mp = bp;    
00845 
00846     if ( length > 0 )
00847     {      
00848       // Set up to iterate through the arguments
00849       va_list args;
00850       va_start( args, format );
00851     
00852       bp = Osc_CreateMessageInternal( bp, &length, address, format, args ); 
00853 
00854       //va_end( args );
00855     }
00856     else
00857       bp = 0;
00858       
00859     if ( bp != 0 )
00860     {
00861       // Set the size
00862       *lp = Osc_EndianSwap( bp - mp ); 
00863   
00864       Osc.bufferPointer = bp;
00865       Osc.bufferRemaining = length;
00866       Osc.messages++;
00867     }
00868     else
00869     {
00870       Osc_SendPacket( channel );
00871     }
00872   } while ( bp == 0 && count == 1 );
00873 
00874   return ( bp != 0 ) ? CONTROLLER_OK : CONTROLLER_ERROR_ILLEGAL_PARAMETER_VALUE;
00875 }

00877 char* Osc_CreateMessageInternal( char* bp, int* length, char* address, char* format, va_list args )
00878 {
00879   // do the address
00880   bp = Osc_WritePaddedString( bp, length, address );
00881   if ( bp == NULL )
00882     return 0;
00883 
00884   // do the type
00885   bp = Osc_WritePaddedString( bp, length, format );
00886   if ( bp == NULL )
00887     return 0;
00888 
00889   // Going to be walking the tag string, the format string and the data
00890   // skip the ',' comma
00891   char* fp;
00892   bool cont = true;
00893   for ( fp = format + 1; *fp && cont; fp++ )
00894   {
00895     switch ( *fp )
00896     {
00897       case 'i':
00898           *length -= 4;
00899           if ( *length >= 0 )
00900           {
00901             int v = va_arg( args, int );
00902             v = Osc_EndianSwap( v );
00903             *((int*)bp) = v;
00904             bp += 4;
00905           }
00906           else 
00907             cont = false;
00908         break;
00909       case 'f':
00910         *length -= 4;
00911         if ( *length >= 0 )
00912         {
00913           int v;
00914           *((float*)&v) = (float)( va_arg( args, double ) ); 
00915           v = Osc_EndianSwap( v );
00916           *((int*)bp) = v;
00917           bp += 4;
00918         }
00919         else 
00920           cont = false;
00921         break;
00922       case 's':
00923       {
00924         char* s = va_arg( args, char* );
00925         bp = Osc_WritePaddedString( bp, length, s );
00926         if ( bp == NULL )
00927           cont = false;
00928         break;
00929       }
00930       default:
00931         cont = false;
00932     }
00933   }
00934 
00935   return ( cont ) ? bp : NULL;
00936 }

00955 int Osc_NumberMatch( int count, char* message, int* bits )
00956 {
00957   int n = 0;
00958   int digits = 0;
00959   while ( isdigit( *message ) )
00960   {
00961     digits++;
00962     n = n * 10 + ( *message++ - '0' );
00963   }
00964 
00965   *bits = -1;
00966   if ( n >= count )
00967     return -1;
00968 
00969   switch ( *message )
00970   {
00971     case '*':
00972     case '?':
00973     case '[':
00974     case '{':
00975     {
00976       int i;
00977       int b = 0;
00978       char s[ 5 ];
00979       for ( i = count - 1; i >=0 ; i-- )
00980       {
00981         b <<= 1;
00982         sprintf( s, "%d", i );
00983         if ( Osc_PatternMatch( message, s ) )
00984           b |= 1;
00985       }
00986       *bits = b;
00987       return -1;
00988     }
00989     default:
00990       if ( digits == 0 )
00991         return -1;
00992       return n;
00993   }
00994 }

01022 char* Osc_WritePaddedString( char* buffer, int* length, char* string )
01023 {
01024   int tagLen = strlen( string ) + 1;
01025   int tagPadLen = tagLen;
01026   int pad = ( tagPadLen ) % 4;
01027   if ( pad != 0 )
01028     tagPadLen += ( 4 - pad );
01029  
01030   *length -= tagPadLen;
01031 
01032   if ( *length >= 0 )
01033   {
01034     strcpy( buffer, string );
01035     int i;
01036     buffer += tagLen;
01037     for ( i = tagLen; i < tagPadLen; i++ ) 
01038       *buffer++ = 0;
01039   }
01040   else
01041     return NULL;
01042 
01043   return buffer;
01044 }

01046 char* Osc_WriteTimetag( char* buffer, int* length, int a, int b )
01047 {
01048   if ( *length < 8 )
01049     return NULL;
01050 
01051   *((int*)buffer) = Osc_EndianSwap( a );
01052   buffer += 4;
01053   *((int*)buffer) = Osc_EndianSwap( b );
01054   buffer += 4;
01055   *length -= 8;
01056 
01057   return buffer;
01058 }

01060 int Osc_EndianSwap( int a )
01061 {
01062   return ( ( a & 0x000000FF ) << 24 ) |
01063          ( ( a & 0x0000FF00 ) << 8 )  | 
01064          ( ( a & 0x00FF0000 ) >> 8 )  | 
01065          ( ( a & 0xFF000000 ) >> 24 );
01066 
01067 }
