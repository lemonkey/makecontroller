00432 int Stepper_Start( int index )
00433 {
00434   int status;
00435 
00436   if ( index < 0 || index >= STEPPER_COUNT )
00437     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00438 
00439   StepperControl* sc = &Stepper.control[ index ]; 
00440   if ( sc->users++ == 0 )
00441   {
00442     if ( Stepper.users++ == 0 )
00443     {
00444       int status = Stepper_Init();
00445       if ( status != CONTROLLER_OK )
00446       {
00447         // I guess not then
00448         sc->users--;
00449         Stepper.users--;
00450         return status;
00451       }   
00452     }
00453 
00454     status = Pwm_Start( index * 2 );
00455     if ( status != CONTROLLER_OK )
00456     {
00457       sc->users--;
00458       Stepper_Deinit();
00459       Stepper.users--;
00460       return status;
00461     }
00462 
00463     status = Pwm_Start( index * 2 + 1 );
00464     if ( status != CONTROLLER_OK )
00465     {
00466       Pwm_Stop( index * 2 );
00467       sc->users--;
00468       Stepper_Deinit();
00469       Stepper.users--;
00470       return status;
00471     }
00472 
00473     // Fire the PWM's up
00474     sc->duty = 1023;
00475 
00476     int pwm = index * 2;
00477     Pwm_Set( pwm, sc->duty );
00478     Pwm_Set( pwm + 1, sc->duty );
00479 
00480     // Get IO's
00481     int i;
00482     for ( i = 0; i < 4; i++ )
00483     {
00484       int io = Stepper_GetIo( index, i );      
00485       sc->io[ i ] = io;
00486         
00487       // Try to lock the stepper output
00488       status = Io_Start( io, true );
00489       if ( status != CONTROLLER_OK )
00490       {
00491         // Damn - unlock any that we did get
00492         int j;
00493         for ( j = 0; j < i; j++ )
00494           Io_Stop( sc->io[ j ] );
00495 
00496         Pwm_Stop( pwm );
00497         Pwm_Stop( pwm + 1 );
00498 
00499         Stepper.users--;
00500         sc->users--;
00501         return status;
00502       }

00505       Io_SetOutput( io );
00506     }

00524 {

00545       int io = s->io[ i ];

00556       Stepper_Deinit();

00633     if ( s->halfStep )
00634       Stepper_SetBipolarHalfStepOutput( s, s->position );
00635     else
00636       Stepper_SetBipolarOutput( s, s->position );
00637   }
00638   else
00639   {
00640     if ( s->halfStep ) 
00641       Stepper_SetUnipolarHalfStepOutput( s, s->position );
00642     else
00643       Stepper_SetUnipolarOutput( s, s->position );
00644   }
00645 
00646   if ( s->position == s->positionRequested )
00647   {
00648     FastTimer_Cancel( &s->fastTimerEntry );
00649     s->timerRunning = false;
00650   }
00651 }

00937 /** \defgroup StepperOSC Stepper - OSC
00938   Control Stepper motors with the Application Board via OSC.  Internally it uses the PWM subsystem
00939   to control duty and the FastTimer to generate the very accurate timings for the steps.
00940   \ingroup OSC
00941   
00942   \section devices Devices
00943   There are 2 Stepper controllers available on the Application Board, numbered 0 & 1.\n
00944   See the Stepper section in the Application Board user's guide for more information
00945   on hooking steppers up to the board.
00946   
00947   \section properties Properties
00948   Each stepper controller has seven properties - 'position', 'positionrequested', 'speed', 'duty', 'bipolar', 
00949   'halfstep' and 'active'.
00950 
00951   \par Position
00952   The 'position' property corresponds to the current step position of the stepper motor
00953   This value can be both read and written.  Writing this value changes where the motor thinks it is.
00954   The initial value of this parameter is 0.
00955   \par
00956   To set the first stepper to step position 10000, send the message
00957   \verbatim /stepper/0/position 10000\endverbatim
00958   Leave the argument value off to read the position of the stepper:
00959   \verbatim /stepper/0/position \endverbatim
00960 
00961   \par PositionRequested
00962   The 'positionrequested' property describes the desired step position of the stepper motor
00963   This value can be both read and written.  Writing this value changes where the motor thinks it needs to be.
00964   The initial value of this parameter is 0.
00965   \par
00966   To set the first stepper to go to step position 10000, send the message
00967   \verbatim /stepper/0/positionrequested 10000\endverbatim
00968   Leave the argument value off to read the last requested position of the stepper:
00969   \verbatim /stepper/0/positionrequested \endverbatim
00970 
00971   \par Speed
00972   The 'speed' property corresponds to the speed with which the stepper responds to changes 
00973   of position.
00974   This value can be both read and written, and the range of values is 0 - 71000.  A speed of 1
00975   means the stepper performs a new step every millisecond.  Bigger numbers result in slower 
00976   steps.
00977   \par
00978   To set the speed of the first stepper to step at 100ms per step, send a message like
00979   \verbatim /stepper/0/speed 100 \endverbatim
00980   Adjust the argument value to one that suits your application.\n
00981   Leave the argument value off to read the speed of the stepper:
00982   \verbatim /stepper/0/speed \endverbatim
00983   
00984   \par Duty
00985   The 'duty' property corresponds to the how much of the power supply is to be sent to the
00986   stepper.  This is handy for when the stepper is static and not being required to perform too
00987   much work and reducing its power helps reduce heat dissipation.
00988   This value can be both read and written, and the range of values is 0 - 1023.  A duty of 0
00989   means the stepper gets no power, and the value of 1023 means the stepper gets full power.  
00990   \par
00991   To set the duty of the first stepper to 500, send a message like
00992   \verbatim /stepper/0/speed 500 \endverbatim
00993   Adjust the argument value to one that suits your application.\n
00994   Leave the argument value off to read the duty of the stepper:
00995   \verbatim /stepper/0/speed \endverbatim
00996 
00997   \par Bipolar
00998   The 'bipolar' property is set to the style of stepper being used.  A 0 here implies unipolar
00999   (the default) and 1 implies a bipolar stepper.
01000   This value can be both read and written.
01001 
01002   \par HalfStep
01003   The 'halfstep' property controls whether the stepper is being half stepped or not.  A 0 here implies full stepping
01004   (the default) and 1 implies a half stepping.
01005   This value can be both read and written.
01006 
01007   \par Active
01008   The 'active' property corresponds to the active state of the stepper.
01009   If the stepper is set to be active, no other tasks will be able to
01010   write to the same I/O lines.  If you're not seeing appropriate
01011   responses to your messages to a stepper, check the whether it's 
01012   locked by sending a message like
01013   \verbatim /stepper/1/active \endverbatim
01014 */
