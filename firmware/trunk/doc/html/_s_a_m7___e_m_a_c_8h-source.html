00075 /*
00076  * Initialise the EMAC driver.  If successful a semaphore is returned that
00077  * is used by the EMAC ISR to indicate that Rx packets have been received.
00078  * If the initialisation fails then NULL is returned.
00079  */
00080 xSemaphoreHandle xEMACInit( void );
00081 
00082 /*
00083  * Send ulLength bytes from pcFrom.  This copies the buffer to one of the
00084  * EMAC Tx buffers, then indicates to the EMAC that the buffer is ready.
00085  * If lEndOfFrame is true then the data being copied is the end of the frame
00086  * and the frame can be transmitted. 
00087  */
00088 portLONG lEMACSend( portCHAR *pcFrom, unsigned portLONG ulLength, portLONG lEndOfFrame );
00089 
00090 /*
00091  * Frames can be read from the EMAC in multiple sections.
00092  * Read ulSectionLength bytes from the EMAC receive buffers to pcTo.  
00093  * ulTotalFrameLength is the size of the entire frame.  Generally vEMACRead
00094  * will be repetedly called until the sum of all the ulSectionLenths totals
00095  * the value of ulTotalFrameLength.
00096  */
00097 void vEMACRead( portCHAR *pcTo, unsigned portLONG ulSectionLength, unsigned portLONG ulTotalFrameLength );
00098 
00099 /*
00100  * The EMAC driver and interrupt service routines are defined in different 
00101  * files as the driver is compiled to THUMB, and the ISR to ARM.  This function
00102  * simply passes the semaphore used to communicate between the two.
00103  */
00104 void vPassEMACSemaphore( xSemaphoreHandle xCreatedSemaphore );
00105 
00106 /* 
00107  * Called by the Tx interrupt, this function traverses the buffers used to
00108  * hold the frame that has just completed transmission and marks each as
00109  * free again.
00110  */
00111 void vClearEMACTxBuffer( void );
00112 
00113 /*
00114  * Suspend on a semaphore waiting either for the semaphore to be obtained 
00115  * or a timeout.  The semaphore is used by the EMAC ISR to indicate that
00116  * data has been received and is ready for processing.
00117  */
00118 void vEMACWaitForInput( void );
00119 
00120 /*
00121  * Return the length of the next frame in the receive buffers.
00122  */
00123 unsigned portLONG ulEMACInputLength( void );
