00093 /**
00094   Sets whether the specified Digital Out is active.
00095   This is automatically called, setting the channel active, the first time DigitalOut_SetValue() is called.\n
00096   However, the channel must be explicitly set to inactive in order for any other devices to access the I/O lines. 
00097   @param index An integer specifying which Digital Out (0-7).
00098   @param state An integer specifying the state - on (1) or off (0).
00099   @return Zero on success.
00100 */
00101 int DigitalOut_SetActive( int index, int state )
00102 {
00103   if ( index < 0 || index >= DIGITALOUT_COUNT )
00104     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00105 
00106   if ( state )
00107     return DigitalOut_Start( index );
00108   else
00109     return DigitalOut_Stop( index );
00110 }

00179 int DigitalOut_Start( int index )
00180 {
00181   int status;
00182   int enableIndex = index > 1;
00183 
00184   if ( index < 0 || index >= DIGITALOUT_COUNT )
00185     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00186 
00187   if ( DigitalOut_users[ index ]++ == 0 )
00188   {
00189     int enableIo = DigitalOut_GetEnableIo( enableIndex );
00190     if ( DigitalOut_enableUsers[ enableIndex ]++ == 0 )
00191     {
00192       status = Io_Start( enableIo, true );
00193       if ( status != CONTROLLER_OK )
00194       {
00195         DigitalOut_enableUsers[ enableIndex ]--;
00196         DigitalOut_users[ index ]--;
00197         return CONTROLLER_ERROR_CANT_LOCK;
00198       }
00199       Io_SetTrue( enableIo );
00200     }
00201 
00202     int io = DigitalOut_GetIo( index );
00203 
00204     status = Io_Start( io, true );
00205     if ( status != CONTROLLER_OK )
00206     {
00207       DigitalOut_users[ index ]--;
00208       return CONTROLLER_ERROR_CANT_LOCK;
00209     }
00210 
00211     // Got it, now set the io up right
00212     Io_PioEnable( io );
00213     Io_SetFalse( io );
00214     Io_SetOutput( io );
00215   }
00216 
00217   return CONTROLLER_OK;
00218 }

00220 int DigitalOut_Stop( int index )
00221 {
00222   if ( index < 0 || index >= DIGITALOUT_COUNT )
00223     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00224 
00225   if ( DigitalOut_users[ index ] < 1 )
00226     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00227 
00228   if ( --DigitalOut_users[ index ] == 0 )
00229   {
00230     int io = DigitalOut_GetIo( index );
00231     Io_Stop( io );
00232 
00233     int enableIndex = index >> 1;
00234     if ( --DigitalOut_enableUsers[ enableIndex ] == 0 )
00235     {
00236       int enableIo = DigitalOut_GetEnableIo( enableIndex );
00237       Io_Stop( enableIo );
00238     }
00239 
00240   }
00241   return CONTROLLER_OK;
00242 }
