00050 /** 
00051   Controls the activation of the timer
00052   @param active whether the Timer subsystem is active or not
00053   @return Zero on success.
00054   @see Timer_Set, Timer_Cancel
00055 */
00056 int Timer_SetActive( bool active )
00057 {
00058   if ( active )
00059   {
00060     if ( Timer.users++ == 0 )
00061     {
00062       int status;
00063   
00064       status = Timer_Init();  
00065       if ( status != CONTROLLER_OK )
00066       {
00067         Timer.users--;
00068         return status;
00069       }
00070     }
00071   }
00072   else
00073   {
00074     if ( --Timer.users == 0 )
00075       Timer_Deinit();
00076   }
00077   return CONTROLLER_OK;
00078 }

00091 /** 
00092   Create a new timer event.  
00093   The event is signified by a callback to the function provided, after the interval specified.  
00094   The specified ID is passed back to the function to permit one function to work for many events.  
00095   Pass repeat = true to make the event continue to create callbacks until it is canceled.
00096   @param timerCallback pointer to the callback function.  The function must
00097          be of the form <em>void callback( int id )</em>.
00098   @param id An integer specifying the ID the callback function is to be provided with.
00099   @param timeMs The time in milliseconds desired for the callback.
00100   @param repeat Set whether the timer repeats or is a one-time event.
00101   @return Status - an integer indicating success (0) or failure (non-zero).
00102   @see Timer_Start, Timer_Cancel and Timer_Stop
00103 */
00104 int Timer_Set( void (*timerCallback)( int id ), int id, int timeMs, bool repeat )
00105 {
00106   int time = timeMs * TIMER_CYCLES_PER_MS;
00107 
00108   // this could be a lot smarter - for example, modifying the current period?
00109   if ( !Timer.servicing ) 
00110     TaskEnterCritical();
00111 
00112   /// ToDo: Get the new timer record from the heap
00113 
00114   // Get one from the free queue
00115   TimerEntry* tne = Timer.freeFirst;
00116   if ( tne == NULL )
00117   {
00118     if ( !Timer.servicing ) 
00119       TaskExitCritical();
00120     return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00121   }
00122   Timer.freeFirst = tne->next;
00123 
00124   // Set the details into the free dude
00125   tne->callback = timerCallback;
00126   tne->id = id;
00127   tne->time = time;
00128   tne->timeInitial = time;
00129   tne->repeat = repeat;
00130 
00131   // Now insert it.  Two cases: 1) timer is running or 2) timer is not
00132   if ( !Timer.running )
00133   {
00134     // Nothing running right now
00135     // Assert Timer.runningFirst == NULL
00136     Timer.runningFirst = tne;
00137     tne->next = NULL;
00138 
00139     Timer_SetTimeTarget( time );
00140     Timer_Enable();
00141   }  
00142   else
00143   {
00144     // There's something running - insert the new one
00145 
00146     // Could be servicing the interrupt right now!
00147 
00148     // Calculate how long remaining
00149     int target = Timer_GetTimeTarget();
00150     int timeCurrent = Timer_GetTime();
00151     int remaining = target - timeCurrent;
00152 
00153     if ( !Timer.servicing )
00154     {
00155       // Add entry
00156       TimerEntry* first = Timer.runningFirst;
00157       Timer.runningFirst = tne;
00158       tne->next = first;
00159   
00160       if ( time < ( remaining - TIMER_MARGIN ) )
00161       {
00162         Timer_SetTimeTarget( target - ( remaining - time  ));
00163       }
00164       else
00165       {
00166         // pretend that the existing time has been with us for the whole slice
00167         tne->time += timeCurrent;
00168       }
00169     }
00170     else
00171     {
00172       // Add entry
00173       TimerEntry* first = Timer.newFirst;
00174       Timer.newFirst = tne;
00175       tne->next = first;
00176     }
00177   }
00178 
00179   if ( !Timer.servicing ) 
00180     TaskExitCritical();
00181 
00182   return CONTROLLER_OK;
00183 }

00185 /** 
00186   Cancel a timer event.
00187   The event is specified by the callback function and id provided.  All events that match 
00188   these two parameters are terminated.
00189   @param timerCallback A pointer to the callback function.
00190   @param id An integer specifying the ID of the callback function.
00191   @return 0 on success.
00192   @see Timer_Start, Timer_Set and Timer_Start
00193 */
00194 int Timer_Cancel( void (*timerCallback)( int id ), int id )
00195 {
00196   if ( !Timer.servicing ) 
00197     TaskEnterCritical();
00198 
00199   // Look through the running list
00200   TimerEntry* te = Timer.runningFirst;
00201   while ( te != NULL )
00202   {
00203     if ( te->callback == timerCallback && te->id == id )
00204     {
00205       te->callback = NULL;
00206       te->repeat = false;
00207     }
00208     te = te->next;
00209   }
00210 
00211   if ( !Timer.servicing ) 
00212     TaskExitCritical();
00213 
00214   return CONTROLLER_OK;
00215 }

00259 int Timer_Init()
00260 {
00261   // One time datastructure initialization
00262   TimerEntry* te = &Timer.entry[ 0 ];
00263   int i;
00264   Timer.freeFirst = te;
00265   for ( i = 0; i < TIMER_COUNT; i++ ) 
00266   {
00267     te->callback = NULL;
00268     if ( i < TIMER_COUNT - 1 )
00269       te->next = te + 1;
00270     else
00271       te->next = NULL;
00272     te++;
00273   }
00274 
00275   Timer.runningFirst = NULL;
00276   Timer.newFirst = NULL;
00277 
00278   Timer.count = 0;
00279   Timer.jitterTotal = 0;
00280   Timer.jitterMax = 0;  
00281   Timer.jitterMaxAllDay = 0;
00282 
00283   Timer.running = false;
00284   Timer.servicing = false;
00285 
00286   // Configure TC by enabling PWM clock
00287   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_TC0;
00288                                     
00289   unsigned int mask ;
00290   mask = 0x1 << AT91C_ID_TC0;
00291    
00292   /* Disable the interrupt on the interrupt controller */
00293   AT91C_BASE_AIC->AIC_IDCR = mask;
00294 
00295   /* Save the interrupt handler routine pointer */  
00296   AT91C_BASE_AIC->AIC_SVR[ AT91C_ID_TC0 ] = (unsigned int)Timer_Isr;
00297 
00298   /* Store the Source Mode Register */
00299   AT91C_BASE_AIC->AIC_SMR[ AT91C_ID_TC0 ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 7  ;
00300   /* Clear the interrupt on the interrupt controller */
00301   AT91C_BASE_AIC->AIC_ICCR = mask ;
00302 
00303   // Set the timer up.  We want just the basics, except when the timer compares 
00304   // with RC, retrigger
00305   //
00306   // MCK is 47923200
00307   // DIV1: A tick MCK/2 times a second
00308   // This makes every tick every 41.73344ns
00309   // DIV2: A tick MCK/8 times a second
00310   // This makes every tick every 167ns
00311   // DIV3: A tick MCK/32 times a second
00312   // This makes every tick every 668ns  
00313   // DIV4: A tick MCK/128 times a second
00314   // This makes every tick every 2.671us
00315   // DIV5: A tick MCK/1024 times a second
00316   // This makes every tick every 21.368us
00317   // CPCTRG makes the RC event reset the counter and trigger it to restart
00318   AT91C_BASE_TC0->TC_CMR = AT91C_TC_CLKS_TIMER_DIV5_CLOCK | AT91C_TC_CPCTRG;
00319                    
00320   // Only really interested in interrupts when the RC happens
00321   AT91C_BASE_TC0->TC_IDR = 0xFF; 
00322   AT91C_BASE_TC0->TC_IER = AT91C_TC_CPCS; 
00323 
00324   // load the RC value with something
00325   AT91C_BASE_TC0->TC_RC = 0xFFFF;
00326 
00327   // Enable the interrupt
00328   AT91C_BASE_AIC->AIC_IECR = mask;
00329 
00330   return CONTROLLER_OK;
00331 }
