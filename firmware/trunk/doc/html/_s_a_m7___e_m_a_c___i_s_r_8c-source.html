00054 /* The semaphore used to signal the arrival of new data to the interface
00055 task. */
00056 static xSemaphoreHandle xSemaphore = NULL;

00058 void vEMACISR( void ) __attribute__((naked));

00060 /*-----------------------------------------------------------*/
00061 /*
00062  * The EMAC ISR.  Handles both Tx and Rx complete interrupts.
00063  */
00064 void vEMACISR( void )
00065 {
00066   /* This ISR can cause a context switch, so the first statement must be a
00067   call to the portENTER_SWITCHING_ISR() macro.  This must be BEFORE any
00068   variable declarations. */
00069   portENTER_SWITCHING_ISR();
00070 
00071   /* Variable definitions can be made now. */
00072   volatile unsigned portLONG ulIntStatus, ulEventStatus;
00073   portBASE_TYPE xSwitchRequired = pdFALSE;
00074     extern void vClearEMACTxBuffer( void );
00075 
00076   /* Find the cause of the interrupt. */
00077   ulIntStatus = AT91C_BASE_EMAC->EMAC_ISR;
00078   ulEventStatus = AT91C_BASE_EMAC->EMAC_RSR;
00079 
00080   if( ( ulIntStatus & AT91C_EMAC_RCOMP ) || ( ulEventStatus & AT91C_EMAC_REC ) )
00081   {
00082     /* A frame has been received, signal the lwIP task so it can process
00083     the Rx descriptors. */
00084     xSwitchRequired = xSemaphoreGiveFromISR( xSemaphore, pdFALSE );
00085     AT91C_BASE_EMAC->EMAC_RSR = AT91C_EMAC_REC;
00086   }
00087 
00088   if( ulIntStatus & AT91C_EMAC_TCOMP )
00089   {
00090     /* A frame has been transmitted.  Mark all the buffers used by the
00091     frame just transmitted as free again. */
00092     vClearEMACTxBuffer();
00093     AT91C_BASE_EMAC->EMAC_TSR = AT91C_EMAC_COMP;
00094   }
00095 
00096   /* Clear the interrupt. */
00097   AT91C_BASE_AIC->AIC_EOICR = 0;
00098 
00099   /* If a task was woken by either a frame being received then we may need to 
00100   switch to another task. */
00101   portEXIT_SWITCHING_ISR( xSwitchRequired );
00102 }

00106 {
00107   /* Simply store the semaphore that should be used by the ISR. */
00108   xSemaphore = xCreatedSemaphore;
