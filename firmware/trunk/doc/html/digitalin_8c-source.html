00064 /**
00065   Sets whether the specified Digital In is active.
00066   @param index An integer specifying which Digital In (0-7).
00067   @param state An integer specifying the state - 1 (active) or 0 (inactive).
00068   @return Zero on success.
00069 */
00070 int DigitalIn_SetActive( int index, int state )
00071 {
00072   if ( index < 0 || index >= DIGITALIN_COUNT )
00073     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00074 
00075   if ( state )
00076     return DigitalIn_Start( index );
00077   else
00078     return DigitalIn_Stop( index );
00079 }

00081 /**
00082   Returns the active state of the Digital In.
00083   @param index An integer specifying which Digital In (0-7).
00084   @return Zero if inactive, non-zero if active.
00085 */
00086 int DigitalIn_GetActive( int index )
00087 {
00088   if ( index < 0 || index >= DIGITALIN_COUNT )
00089     return false;
00090   return DigitalIn_users[ index ] > 0;

00093 /** 
00094   Read the value of a Digital Input on the MAKE Application Board.
00095   If the voltage on the input is greater than ~0.6V, the Digital In will read high.
00096   @param index An integer specifying which Digital In (0-7).
00097   @return The value - 1 (high) or 0 (low).
00098 */
00099 int DigitalIn_GetValue( int index )
00100 {
00101   if ( index < 0 || index >= DIGITALIN_COUNT )
00102     return 0;
00103 
00104   if ( DigitalIn_users[ index ] < 1 )
00105   {
00106     int status = DigitalIn_Start( index );
00107     if ( status != CONTROLLER_OK )
00108       return status;
00109   }
00110 
00111   int io = DigitalIn_GetIo( index );
00112 
00113   if ( io >= 0 )
00114     return Io_GetValue( io );
00115   else 
00116     return AnalogIn_GetValue( -io ) > 200;
00117 }

00122 int DigitalIn_Start( int index )
00123 {
00124   int status;
00125 
00126   if ( index < 0 || index >= DIGITALIN_COUNT )
00127     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00128 
00129   if ( DigitalIn_users[ index ]++ == 0 )
00130   {
00131     int io = DigitalIn_GetIo( index );
00132 
00133     if ( io > 0 )
00134     {
00135       status = Io_Start( io, false );
00136       if ( status != CONTROLLER_OK )
00137       {
00138         DigitalIn_users[ index ]--;
00139         return CONTROLLER_ERROR_CANT_LOCK;
00140       }
00141   
00142       // Got it, now set the io up right
00143       Io_PioEnable( io );
00144       Io_SetInput( io );
00145     }
00146     else
00147       return AnalogIn_SetActive( -io, 1 );
00148   }
00149 
00150   return CONTROLLER_OK;
00151 }

00153 int DigitalIn_Stop( int index )
00154 {
00155   if ( index < 0 || index >= DIGITALIN_COUNT )
00156     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00157 
00158   if ( DigitalIn_users[ index ] < 1 )
00159     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00160 
00161   if ( --DigitalIn_users[ index ] == 0 )
00162   {
00163     int io = DigitalIn_GetIo( index );
00164     Io_Stop( io );
00165   }
00166   return CONTROLLER_OK;
00167 }

00204 /** \defgroup DigitalInOSC Digital In - OSC
00205   Read the Application Board's Digital Inputs via OSC.
00206   \ingroup OSC
00207   
00208   \section devices Devices
00209   There are 8 Digital Inputs on the Make Application Board, numbered 0 - 7.
00210   The Digital Ins are physically the same as the Analog Ins - they're on the same 
00211   signal lines and the same connectors - but reading them as Digital Ins is 
00212   slightly more efficient/quicker.\n
00213   If the voltage on the input is greater than ~0.6V, the Digital In will read high.
00214   
00215   \section properties Properties
00216   The Digital Ins have two properties - 'value' and 'active'.
00217 
00218   \par Value
00219   The 'value' property corresponds to the on/off value of a Digital In.
00220   Because you can only ever \em read the value of an input, you'll never
00221   want to include an argument at the end of your OSC message to read the value.\n
00222   To read the third Digital In, send a message like
00223   \verbatim /digitalin/2/value \endverbatim
00224   
00225   \par Active
00226   The 'active' property corresponds to the active state of a Digital In.
00227   If a Digital In is set to be active, no other tasks will be able to
00228   read from it as an Analog In.  If you're not seeing appropriate
00229   responses to your messages to the Digital In, check the whether it's 
00230   locked by sending a message like
00231   \verbatim /digitalin/0/active \endverbatim
00232   \par
00233   You can set the active flag by sending
00234   \verbatim /digitalin/0/active 1 \endverbatim
00235 */
