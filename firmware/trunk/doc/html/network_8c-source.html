00932 /**
00933   Create a checksum for the current address settings and store it in EEPROM.
00934   This should be called each time an address setting is changed so that if
00935   the board gets powered down, it will know when it comes back up whether or
00936   not the address settings is currently has are valid.
00937 
00938   @param v An integer specifying whether to validate the current settings (1)
00939   or to force them to be invalidated (0).
00940   Passing in 0 returns the address settings to their factory defaults.
00941   @return 0 on success.
00942 */
00943 int Network_SetValid( int v )
00944 {
00945   if ( v )
00946   {
00947     struct ip_addr ip, gw, mask;
00948     struct netif* mc_netif;
00949 
00950     ip.addr = Network->TempIpAddress; // these should each have been set previously
00951     mask.addr = Network->TempMask;  // by Network_SetAddress(), etc.
00952     gw.addr = Network->TempGateway;
00953     if( !Network_GetDhcpEnabled() ) // only actually change the address if we're not using DHCP
00954     {
00955       // we specify our network interface as en0 when we init
00956       mc_netif = netif_find( "en0" );
00957       if( mc_netif != NULL )
00958         netif_set_addr( mc_netif, &ip, &mask, &gw );
00959     }
00960     
00961     // but write the addresses to memory regardless, so we can use them next time we boot up without DHCP
00962     Eeprom_Write( EEPROM_SYSTEM_NET_ADDRESS, (uchar*)&ip.addr, 4 );
00963     Eeprom_Write( EEPROM_SYSTEM_NET_MASK, (uchar*)&mask.addr, 4 );
00964     Eeprom_Write( EEPROM_SYSTEM_NET_GATEWAY, (uchar*)&gw.addr, 4 );
00965 
00966     int total = Network->TempIpAddress + Network->TempMask + Network->TempGateway;
00967     Eeprom_Write( EEPROM_SYSTEM_NET_CHECK, (uchar*)&total, 4 );
00968 
00969     Network_Valid = NET_VALID;
00970   }
00971   else
00972   {
00973     int value = 0;
00974     Eeprom_Write( EEPROM_SYSTEM_NET_CHECK, (uchar*)&value, 4 );
00975     Network_Valid = NET_INVALID;
00976   }
00977 
00978   return CONTROLLER_OK;
00979 }
