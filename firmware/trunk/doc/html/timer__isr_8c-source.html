00029 extern struct Timer_ Timer;

00031 void Timer_Isr( void ) __attribute__ ((interrupt("IRQ")));
00032 
00033 void Timer_Isr( void )
00034 {
00035   int status = AT91C_BASE_TC0->TC_SR;
00036   if ( status & AT91C_TC_CPCS )
00037   {
00038     Timer.servicing = true;
00039         
00040     int jitter;
00041     Timer.count++;
00042     jitter = AT91C_BASE_TC0->TC_CV;
00043 
00044     Timer.jitterTotal += jitter;
00045     if ( jitter > Timer.jitterMax )
00046       Timer.jitterMax = jitter;
00047     if ( jitter > Timer.jitterMaxAllDay )
00048       Timer.jitterMaxAllDay = jitter;
00049 
00050     // Run through once to make the callback calls
00051     TimerEntry* te = Timer.runningFirst;
00052     TimerEntry* pe = NULL;
00053     while ( te != NULL )
00054     {
00055       TimerEntry* ne = te->next;
00056       te->time -= AT91C_BASE_TC0->TC_RC + AT91C_BASE_TC0->TC_CV;
00057       if ( te->time <= 0 )
00058       {
00059         if ( te->callback != NULL )
00060           (*te->callback)( te->id );
00061         
00062         if ( te->repeat )
00063         {
00064           te->time += te->timeInitial;
00065         }
00066         else
00067         {
00068           // remove it if necessary (do this first!)
00069           if ( pe == NULL )
00070             Timer.runningFirst = ne;
00071           else
00072             pe->next = ne;
00073                
00074           // add it to the free queue
00075           te->next = Timer.freeFirst;
00076           Timer.freeFirst = te;        
00077         }
00078       } 
00079       else
00080       {
00081         pe = te;
00082       }
00083       te = ne;
00084     }
00085 
00086     // Slightly sad thing: new entries created during the callbacks need to get added after this loop
00087     if ( Timer.newFirst != NULL )
00088     {
00089       if ( pe == NULL )
00090       {  
00091         Timer.runningFirst = Timer.newFirst; 
00092       }
00093       else
00094       {
00095         pe->next = Timer.newFirst;
00096       }
00097       Timer.newFirst = NULL;
00098     }
00099 
00100     te = Timer.runningFirst;
00101     if ( te != NULL )
00102     {
00103       // Run through again to determine the next time
00104       int nextTime = -1;
00105       while ( te != NULL )
00106       {
00107         if ( nextTime == -1 || te->time < nextTime )
00108           nextTime = te->time;
00109         te = te->next;
00110       }
00111       // Add in whatever we're at now
00112       nextTime += AT91C_BASE_TC0->TC_CV;
00113       // Make sure it's not too big
00114       if ( nextTime > 0xFFFF )
00115         nextTime = 0xFFFF;
00116       AT91C_BASE_TC0->TC_RC = nextTime;
00117     }
00118     else
00119     {
00120       AT91C_BASE_TC0->TC_CCR = AT91C_TC_CLKDIS;
00121       Timer.running = false;
00122     }
00123 
00124     Timer.servicing = false;
00125   }
00126 
00127   /* Clear AIC to complete ISR processing */
00128   AT91C_BASE_AIC->AIC_EOICR = 0;
00129 }
