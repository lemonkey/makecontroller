00018 /** @defgroup OSC
00019   Communicate with the Make Controller Kit via OSC.
00020   
00021   \section osc OSC
00022   "Open Sound Control (OSC) is a protocol for communication among computers, sound synthesizers, 
00023   and other multimedia devices that is optimized for modern networking technology."  
00024   With OSC implemented on the Make Controller Kit, it can already talk to 
00025   a wide variety of environments and devices like Java, Max/MSP, Pd, Flash, Processing, SuperCollider,
00026   and many others.
00027   
00028   OSC is based on the notion of \b messages, which are composed of an address, and the data to 
00029   be sent to that address.  The address looks a lot like a URL that you might type into
00030   your internet browser.  Each element in the address is like a directory, with other 
00031   elements inside it, and each element is separated from the next by a slash (/).  Each OSC
00032   message must also start with a slash.  
00033   
00034   \par Example:
00035   For instance, the OSC address
00036   \code /make/controller/kit \endcode
00037   says, "start in the 'make' directory, go down to the 'controller' directory, and 
00038   finally to the 'kit' directory.  
00039   
00040   Any number of \b argument \b values can be sent to that address by including them in the message
00041   after the address.  These values can be integers (ints), floats, or strings.  
00042   
00043   \par Example:
00044   If we wanted to send the value 35.4 to the address above, we would create the message
00045   \code /make/controller/kit 35.4 \endcode
00046   with a space between the address and the data.\n\n
00047   Additional data can be added, each separated by a space
00048   \code /make/controller/kit 35.4 lawn 12 \endcode
00049   
00050   \section osc_mck OSC & the Make Controller Kit
00051   Many devices on the Make Controller Kit can be addressed via OSC.  In sending messages to them,
00052   we need to know the OSC address, and the appropriate argument values to send.
00053   
00054   The Make Controller Kit is orgranized, for OSC, into \b subsystems.  Each subsystem has one or more
00055   \b devices, and each device has one or more \b properties.  To address a particular device, you'll
00056   need to create an OSC message specifying the address in that format: 
00057   \code /subsystem/device/property \endcode
00058   Each of the modules above provide the details for each of the subsystems on the board, along with their
00059   devices and properties.  A simple example is given below.
00060   
00061   \par Example:
00062   To create an OSC message to turn an LED on, first identify the appropriate \b subsystem.
00063   In this case, the subsystem is called \b appled.\n\n
00064   There are 4 LEDs, so we need to specify which one to control.
00065   The LEDs are numbered 0 -3, so choosing the first LED means the \b device value is 0.\n\n
00066   The \b property of the LED that turns it on and off is its 'state'.\n\n
00067   Lastly, we must specify what the state should actually be, by including an \b argument value after the address.
00068   To turn it on, this value should be 1.  0 would turn it off.\n
00069   The complete OSC message looks like
00070   \code /appled/0/state 1 \endcode
00071 */

00357 int Osc_ReceivePacket( int channel, char* packet, int length )
00358 {
00359   // Got a packet.  Unpacket.
00360 
00361   switch ( *packet )
00362   {
00363     case '/':
00364       Osc_ReceiveMessage( channel, packet, length );
00365       break;
00366     case '#':
00367       if ( strcmp( packet, "#bundle" ) == 0 )
00368       {
00369         // skip bundle text and timetag
00370         packet += 16;
00371         length -= 16;
00372         while ( length > 0 )
00373         {
00374           // read the length (pretend packet is a pointer to integer)
00375           int messageLength = Osc_EndianSwap( *((int*)packet) );
00376           packet += 4;
00377           length -= 4;
00378           if ( messageLength <= length )
00379             Osc_ReceivePacket( channel, packet, messageLength );
00380           length -= messageLength;
00381           packet += messageLength;
00382         }
00383       }
00384       break;
00385     default:
00386       // Something else?
00387       Osc_CreateMessage( channel, "/error", ",s", "Packet Error" );
00388       break;
00389   }
00390 
00391   return Osc_SendPacket( channel );
00392 }

00394 int Osc_ReceiveMessage( int channel, char* message, int length )
00395 {
00396   // Got a packet.  Unpacket.
00397 
00398   // Confirm it's a message
00399   if ( *message == '/' )
00400   {
00401     // Plain message
00402     char* nextSlash = strchr( message + 1, '/' );
00403     if ( nextSlash != NULL )
00404       *nextSlash = 0;
00405     int i;
00406     int count = 0;
00407     for ( i = 0; i <= Osc.subsystemHighest; i++ )
00408     {
00409       OscSubsystem* sub = &Osc.subsystem[ i ];
00410       if ( Osc_PatternMatch( message + 1, sub->name ) )
00411       {
00412         count++;
00413         if ( nextSlash )
00414           (*sub->receiveMessage)( channel, nextSlash + 1, length - ( nextSlash - message ) - 1 );
00415         else
00416           (*sub->receiveMessage)( channel, 0, 0 );          
00417       }
00418     }
00419     if ( count == 0 )
00420     {
00421       char s[ OSC_SCRATCH_SIZE ];
00422       snprintf( s, OSC_SCRATCH_SIZE, "No Subsystem Match - %s", message + 1 ); 
00423       Osc_CreateMessage( channel, "/error", ",s", s );
00424     }
00425   }
00426   else
00427   {
00428     return CONTROLLER_ERROR_BAD_DATA;
00429   }
00430   return CONTROLLER_OK;
00431 }

00499 // An OSC message has arrived, check it out... although, don't change it - others need
00500 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00501 // messages in it, they will be unpacked elsewhere.  We'll only get 
00502 // individual messages here.
00503 // What is passed is the part after the first subsystem address.
00504 // So "/dipswitch/active 1" would be passed to this routine as "active 1"
00505 // Not literally like this, of course, but as OSC messages
00506 // In order to do anything, we need some extra information from 
00507 // the subsystem like what the names of its properties are, what 
00508 // the getters and setters are, etc.
00509 int Osc_BlobReceiverHelper( int channel, char* message, int length, 
00510                            char* subsystemName, 
00511                            int (*blobPropertySet)( int property, uchar* buffer, int length ),
00512                            int (*blobPropertyGet)( int property, uchar* buffer, int size ),
00513                            char* blobPropertyNames[] )
00514 {
00515   if ( message == NULL )
00516     return CONTROLLER_ERROR_NO_PROPERTY;
00517 
00518   int propertyIndex = Osc_PropertyLookup( blobPropertyNames, message );
00519   if ( propertyIndex == -1 )
00520     return CONTROLLER_ERROR_UNKNOWN_PROPERTY;
00521 
00522   // Sometime after the address, the data tag begins - this is the description 
00523   // of the data in the rest of the message.  It starts with a comma.  Return
00524   // where it is into 'type'.  If there is no comma, this is bad.
00525   char* type = Osc_FindDataTag( message, length );
00526   if ( type == NULL )
00527     return CONTROLLER_ERROR_NO_TYPE_TAG;
00528 
00529   // We can tell if there's data by seeing if the character after the comma
00530   // is a zero or not.
00531   if ( type[ 1 ] != 0 )
00532   {
00533     if ( type[ 1 ] == 'b' )
00534     {
00535       unsigned char *buffer;
00536       int size;
00537       int count = Osc_ExtractData( type, "b", &buffer, &size );
00538       if ( count != 1 )
00539         return CONTROLLER_ERROR_BAD_DATA;
00540     
00541       (*blobPropertySet)( propertyIndex, buffer, size );
00542     }
00543     else
00544     {
00545       if ( type[ 1 ] == 's' )
00546       {
00547         unsigned char *buffer;
00548         int count = Osc_ExtractData( type, "s", &buffer );
00549         if ( count != 1 )
00550           return CONTROLLER_ERROR_BAD_DATA;
00551     
00552         (*blobPropertySet)( propertyIndex, buffer, strlen( (char*)buffer ) );
00553       }
00554       else
00555         return CONTROLLER_ERROR_BAD_DATA;
00556     }
00557       
00558   }
00559   else
00560   {
00561     // No data, then.  I guess it was a read.  The XXXXOsc getters
00562     // take the channel number and use it to call
00563     // Osc_CreateMessage() which adds a new message to the outgoing
00564     // stack
00565     uchar buffer[ OSC_SCRATCH_SIZE ];
00566     int size = (*blobPropertyGet)( propertyIndex, buffer, OSC_SCRATCH_SIZE );
00567     char address[ OSC_SCRATCH_SIZE ]; 
00568     snprintf( address, OSC_SCRATCH_SIZE, "/%s/%s", subsystemName, blobPropertyNames[ propertyIndex ] ); 
00569     Osc_CreateMessage( channel, address, ",b", buffer, size );
00570   }
00571 
00572   return CONTROLLER_OK;
00573 }

00575 // An OSC message has arrived, check it out... although, don't change it - others need
00576 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00577 // messages in it, they will be unpacked elsewhere.  We'll only get 
00578 // individual messages here.
00579 // What is passed is the part after the first subsystem address.
00580 // So "/dipswitch/active 1" would be passed to this routine as "active 1"
00581 // Not literally like this, of course, but as OSC messages

00583 // the subsystem like what the names of its properties are, what 
00584 // the getters and setters are, etc.
00585 int Osc_IntReceiverHelper( int channel, char* message, int length, 
00586                            char* subsystemName, 
00587                            int (*propertySet)( int property, int value ),
00588                            int (*propertyGet)( int property ),
00589                            char* propertyNames[] )
00590 {
00591   if ( message == NULL )
00592     return CONTROLLER_ERROR_NO_PROPERTY;
00593 
00594   int propertyIndex = Osc_PropertyLookup( propertyNames, message );
00595   if ( propertyIndex == -1 )
00596     return CONTROLLER_ERROR_UNKNOWN_PROPERTY;
00597 
00598 /*
00599   int bits;
00600   int number = Osc_NumberMatch( indexCount, message, &bits );
00601   if ( number == -1 && bits == -1 )
00602     return Osc_SubsystemError( channel, subsystemName, "Bad index" );
00603 */
00604   
00605   // Sometime after the address, the data tag begins - this is the description 
00606   // of the data in the rest of the message.  It starts with a comma.  Return
00607   // where it is into 'type'.  If there is no comma, this is bad.
00608   char* type = Osc_FindDataTag( message, length );
00609   if ( type == NULL )
00610     return CONTROLLER_ERROR_NO_TYPE_TAG;
00611 
00612   // We can tell if there's data by seeing if the character after the comma
00613   // is a zero or not.
00614   if ( type[ 1 ] == 'i' || type[ 1 ] == 'f' )
00615   {
00616     // If there was int or float data, it was a WRITE.
00617     // So, sort of scanff-like, go get the data.  Here we pass in where the data is
00618     // thanks to the previous routine and then specify what we expect to find there
00619     // in tag terms (i.e. "i", "s", "f" and others).  Finally we pass in a set 
00620     // of pointers to the data types we want to extract.  Osc_ExtractData()
00621     // will rummage around in the message magically grabbing values for you,
00622     // reporting how many it got.  It will convert ints and floats if necessary.
00623     int value;
00624     int count = Osc_ExtractData( type, "i", &value );
00625     if ( count != 1 )
00626       return CONTROLLER_ERROR_BAD_DATA;
00627   
00628     (*propertySet)( propertyIndex, value );
00629   }

00715 // An OSC message has arrived, check it out... although, don't change it - others need
00716 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00717 // messages in it, they will be unpacked elsewhere.  We'll only get 
00718 // individual messages here.
00719 // What is passed is the part after the first subsystem address.
00720 // So "/appled/0/on 1" would be passed to this routine as "0/on 1"
00721 // Not literally like this, of course, but as OSC messages
00722 // In order to do anything, we need some extra information from 
00723 // the subsystem like what the names of its properties are, what 
00724 // the getters and setters are, etc.
00725 // It's great to have this code in one central place.  It will be easier 
00726 // to effect change from here.o
00727 int Osc_IndexIntReceiverHelper( int channel, char* message, int length, 
00728                                 int indexCount, char* subsystemName, 
00729                                 int (*propertySet)( int index, int property, int value ),
00730                                 int (*propertyGet)( int index, int property ),
00731                                 char* propertyNames[] )
00732 {
00733   // Look for the next slash - being the one that separates the index
00734   // from the property.  Note that this won't go off on a search through the buffer
00735   // since there will soon be a string terminator (i.e. a 0)
00736   char* prop = strchr( message, '/' );
00737   if ( prop == NULL )
00738     return CONTROLLER_ERROR_BAD_FORMAT;
00739 
00740   // Now that we know where the property is, we can see if we can find it.
00741   // This is a little cheap, since we're also implying that there are no 
00742   // more address terms after the property.  That is, if testing for "speed", while
00743   // "speed" would match, "speed/other_stuff" would not.
00744   int propertyIndex = Osc_PropertyLookup( propertyNames, prop + 1 );
00745   if ( propertyIndex == -1 )
00746     return CONTROLLER_ERROR_UNKNOWN_PROPERTY;
00747 
00748   // Here's where we try to understand what index we got.  In the world of 
00749   // OSC, this could be a pattern.  So while we could get "0/speed" we could 
00750   // also get "*/speed" or "[0-4]/speed".  This is kind of a drag, but it is 
00751   // quite nice from the user's perspective.
00752   // So to deal with this take a look at the text "0" or "{1,2}" or whatever
00753   // and produce either a nice integer in the simplest case or a set of bits 
00754   // where each bit corresponds to one of the indicies.  Clearly we don't have
00755   // to go crazy, since there are only a small finite number of them.
00756   // Osc_NumberMatch() does the work for us, producing either number = -1 and 
00757   // bits == -1 if there was no index match, or number != -1 for there was a single
00758   // number, or bits != -1 if there were several.
00759 
00760   // note that we tweak the string a bit here to make sure the next '/' is not 
00761   // mixed up with this.  Insert a string terminator.
00762   *prop = 0;
00763 
00764   int bits;
00765   int number = Osc_NumberMatch( indexCount, message, &bits );
00766   if ( number == -1 && bits == -1 )
00767     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00768   
00769   // We tweaked the '/' before - now put it back
00770   *prop = '/';
00771 
00772   // Sometime after the address, the data tag begins - this is the description 
00773   // of the data in the rest of the message.  It starts with a comma.  Return
00774   // where it is into 'type'.  If there is no comma, this is bad.
00775   char* type = Osc_FindDataTag( message, length );
00776   if ( type == NULL )
00777     return CONTROLLER_ERROR_NO_TYPE_TAG;
00778 
00779   // We can tell if there's data by seeing if the character after the comma
00780   // is a zero or not.
00781   if ( type[ 1 ] == 'i' || type[ 1 ] == 'f' )
00782   {
00783     // If there was int or float data, it was a WRITE.
00784     // So, sort of scanff-like, go get the data.  Here we pass in where the data is
00785     // thanks to the previous routine and then specify what we expect to find there
00786     // in tag terms (i.e. "i", "s", "f" and others).  Finally we pass in a set 
00787     // of pointers to the data types we want to extract.  Osc_ExtractData()
00788     // will rummage around in the message magically grabbing values for you,
00789     // reporting how many it got.  It will convert ints and floats if necessary.
00790     int value;
00791     int count = Osc_ExtractData( type, "i", &value );
00792     if ( count != 1 )
00793       return CONTROLLER_ERROR_INCORRECT_DATA_TYPE;
00794   
00795     // Now with the data we need to decide what to do with it.
00796     // Is there one or many here?
00797     if ( number != -1 )
00798       (*propertySet)( number, propertyIndex, value );
00799     else
00800     {
00801       int index = 0;
00802       while ( bits > 0 && index < indexCount )
00803       { 
00804         if ( bits & 1 )
00805           (*propertySet)( index, propertyIndex, value );
00806         bits >>= 1;
00807         index++;
00808       }
00809     }
00810   }
00811   else
00812   {
00813     // No data, then.  I guess it was a read.  The XXXXOsc getters
00814     // take the channel number and use it to call
00815     // Osc_CreateMessage() which adds a new message to the outgoing
00816     // stack
00817     if ( number != -1 )
00818     {
00819       int value = (*propertyGet)( number, propertyIndex );
00820       char address[ OSC_SCRATCH_SIZE ]; 
00821       snprintf( address, OSC_SCRATCH_SIZE, "/%s/%d/%s", subsystemName, number, propertyNames[ propertyIndex ] ); 
00822       Osc_CreateMessage( channel, address, ",i", value );
00823     }
00824     else
00825     {
00826       int index = 0;
00827       while ( bits > 0 && index < indexCount )
00828       { 
00829         if ( bits & 1 )
00830         {
00831           int value = (*propertyGet)( index, propertyIndex );
00832           char address[ OSC_SCRATCH_SIZE ]; 
00833           snprintf( address, OSC_SCRATCH_SIZE, "/%s/%d/%s", subsystemName, index, propertyNames[ propertyIndex ] ); 
00834           Osc_CreateMessage( channel, address, ",i", value );
00835         }
00836         bits >>= 1;
00837         index++;
00838       }
00839     }
00840   }
00841 
00842   return CONTROLLER_OK;
00843 }

00878 // OSC_ExtractData takes a buffer (i.e. a point in the incoming OSC message)
00879 // And a format e.g. "i" "bb", etc. and unpacks them to the var args
00880 // The var args need to be pointers to memory ready to receive the values.
00881 // In the case of blobs, there need to be three parameters: char** buffer, 
00882 // and int* size on the param list.  The buffer gets a pointer into the 
00883 // right place in the incoming buffer and the size value gets assigned
00884 int Osc_ExtractData( char* buffer, char* format, ... )
00885 {
00886   // Set up to iterate through the arguments
00887   va_list args;
00888   va_start( args, format );
00889   int count = 0;
00890 
00891   // figure out where the data starts
00892   int tagLen = strlen( buffer ) + 1;
00893   int pad = tagLen % 4;
00894   if ( pad != 0 )
00895     tagLen += ( 4 - pad );
00896   char* data = buffer + tagLen;
00897 
00898   // Going to be walking the tag string, the format string and the data
00899   char* fp;
00900   char* tp = buffer + 1; // need to skip the comma ','
00901   bool cont = true;
00902   for ( fp = format; *fp && cont; fp++ )
00903   {
00904     cont = false;
00905     switch ( *fp )
00906     {
00907       case 'i':
00908         if ( *tp == 'i' )
00909         {
00910           *(va_arg( args, int* )) = Osc_ReadInt( data );
00911           data += 4;
00912           count++;
00913           cont = true;
00914         }
00915         if ( *tp == 'f' )
00916         {
00917           *(va_arg( args, int* )) = (int)Osc_ReadFloat( data );
00918           data += 4;
00919           count++;
00920           cont = true;
00921         }
00922         
00923         break;
00924       case 'f':
00925         if ( *tp == 'f' )
00926         {
00927           *(va_arg( args, float* )) = Osc_ReadFloat( data );
00928           data += 4;
00929           count++;
00930           cont = true;
00931         }
00932         if ( *tp == 'i' )
00933         {
00934           *(va_arg( args, float* )) = (float)Osc_ReadInt( data );
00935           data += 4;
00936           count++;
00937           cont = true;
00938         }
00939         break;
00940       case 's':
00941         if ( *tp == 's' )
00942         {
00943           *(va_arg( args, char** )) = data;
00944           int len = strlen( data ) + 1;
00945           int pad = len % 4;
00946           if ( pad != 0 )
00947             len += ( 4 - pad );
00948           data += len;
00949           count++;
00950           cont = true;
00951         }
00952         break;
00953       case 'b':
00954         if ( *tp == 'b' )
00955         {
00956           int length = Osc_ReadInt( data );
00957           data += 4;
00958           *(va_arg( args, char** )) = data;
00959           *(va_arg( args, int* )) = length;
00960           int pad = length % 4;
00961           if ( pad != 0 )
00962             length += ( 4 - pad );
00963           data += length;
00964           count++;
00965           cont = true;
00966         }
00967         break;
00968     }
00969     tp++;
00970   }
00971 
00972   //va_end( args );
00973 
00974   return count;
00975 }

00991 /**
00992   Osc_CreateMessageNoLock
00993   Must put the "," as the first format letter
00994   */
00995 int Osc_CreateMessage( int channel, char* address, char* format, ... )
00996 {
00997   if ( address == NULL || format == NULL || *format != ',' )
00998     return CONTROLLER_ERROR_BAD_DATA;
00999 
01000   if ( channel < 0 || channel >= OSC_CHANNEL_COUNT )
01001     return CONTROLLER_ERROR_ILLEGAL_INDEX;
01002 
01003   OscChannel* ch = &Osc.channel[ channel ];
01004 
01005   if ( !ch->running )
01006     return CONTROLLER_ERROR_SUBSYSTEM_INACTIVE;
01007 
01008   if ( channel == OSC_CHANNEL_UDP && ch->replyAddress == 0 )
01009     return CONTROLLER_ERROR_NO_ADDRESS;
01010 
01011   // Check for sender
01012   if ( ch->sendMessage == NULL )
01013     return CONTROLLER_ERROR_RESOURCE_MISSING;
01014 
01015   if ( Osc_Lock( ch ) != CONTROLLER_OK )
01016     return CONTROLLER_ERROR_LOCK_ERROR;
01017 
01018   if ( ch->bufferPointer == NULL )
01019     Osc_ResetChannel( ch );
01020 
01021   // try to send this message - if there's a problem somewhere, 
01022   // send the existing buffer - freeing up space, then try (once) again.
01023   int count = 0;
01024   char *bp;
01025   do
01026   {  
01027     count++;
01028 
01029     char* buffer = ch->bufferPointer;
01030     int length = ch->bufferRemaining;
01031   
01032     bp = buffer;
01033   
01034     // First message in the buffer?
01035     if ( bp == ch->buffer )
01036     {
01037       bp = Osc_CreateBundle( bp, &length, 0, 0 );
01038       if ( bp == NULL )
01039         return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
01040     }
01041   
01042     // Make room for the new message
01043     int* lp = (int *)bp;
01044     bp += 4;
01045     length -= 4;
01046 
01047     // remember the start of the message
01048     char* mp = bp;    
01049 
01050     if ( length > 0 )
01051     {      
01052       // Set up to iterate through the arguments
01053       va_list args;
01054       va_start( args, format );
01055     
01056       bp = Osc_CreateMessageInternal( bp, &length, address, format, args ); 
01057 
01058       //va_end( args );
01059     }
01060     else
01061       bp = 0;
01062       
01063     if ( bp != 0 )
01064     {
01065       // Set the size
01066       *lp = Osc_EndianSwap( bp - mp ); 
01067   
01068       ch->bufferPointer = bp;
01069       ch->bufferRemaining = length;
01070       ch->messages++;
01071     }
01072     else
01073     {
01074       Osc_SendPacketInternal( ch );
01075     }
01076   } while ( bp == 0 && count == 1 );
01077 
01078   Osc_Unlock( ch );
01079 
01080   return ( bp != 0 ) ? CONTROLLER_OK : CONTROLLER_ERROR_ILLEGAL_PARAMETER_VALUE;
01081 }

01083 char* Osc_CreateMessageInternal( char* bp, int* length, char* address, char* format, va_list args )
01084 {
01085   // do the address
01086   bp = Osc_WritePaddedString( bp, length, address );

01089 
01090   // do the type
01091   bp = Osc_WritePaddedString( bp, length, format );

01093     return 0;
01094 
01095   // Going to be walking the tag string, the format string and the data
01096   // skip the ',' comma
01097   char* fp;
01098   bool cont = true;
01099   for ( fp = format + 1; *fp && cont; fp++ )
01100   {
01101     switch ( *fp )

01104           *length -= 4;

01193   int digits = 0;
01194   while ( isdigit( *message ) )
