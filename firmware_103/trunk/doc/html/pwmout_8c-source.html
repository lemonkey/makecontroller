00071 /** \defgroup PwmOut
00072   The PWM Out subsystem underlies the DC Motor subsystem and controls the 4 PWM signals on the SAM7X.
00073   Each PWM device controls a pair of Digital Outs - an A and a B channel:
00074   - PwmOut 0 - Digital Outs 0 (A) and 1 (B).
00075   - PwmOut 1 - Digital Outs 2 (A) and 3 (B).
00076   - PwmOut 2 - Digital Outs 4 (A) and 5 (B).
00077   - PwmOut 3 - Digital Outs 6 (A) and 7 (B).
00078   
00079   The A and B channels of a PWM device can be set independently to be inverted, or not, from one another
00080   in order to control motors, lights, etc.
00081   \ingroup AppBoard

00330 int PwmOut_Start( int index )
00331 {
00332   int status;
00333 
00334   if ( index < 0 || index > 3 )
00335     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00336 
00337   if ( PwmOut_users[ index ]++ == 0 )
00338   {
00339     // Look up the IO's that will be used
00340     int ioA;
00341     int ioB;
00342     PwmOut_GetIos( index, &ioA, &ioB );
00343   
00344     // Try to get the IO's
00345     status = Io_Start( ioA, true );
00346     if ( status != CONTROLLER_OK )
00347     {
00348       PwmOut_users[ index ]--;
00349       return status;
00350     }
00351   
00352     status = Io_Start( ioB, true );
00353     if ( status != CONTROLLER_OK )
00354     {
00355       PwmOut_users[ index ]--;
00356       // better give back ioA since we did get that
00357       Io_Stop( ioA );
00358       return status;
00359     }
00360     
00361     // Make sure we can get the PWM channel
00362     status = Pwm_Start( index );
00363     if ( status != CONTROLLER_OK )
00364     {
00365       PwmOut_users[ index ]--;
00366       // better give back the io's since we did get them
00367       Io_Stop( ioA );
00368       Io_Stop( ioB );
00369       return status;
00370     }
00371 
00372     // Set all the io's right
00373   
00374     // Enable as PIOs
00375     Io_PioEnable( ioA );
00376     Io_PioEnable( ioB );
00377   
00378     // Switch them off
00379     Io_SetFalse( ioA );
00380     Io_SetFalse( ioB );
00381   
00382     // Turn them into outputs
00383     Io_SetOutput( ioA );
00384     Io_SetOutput( ioB );
00385 
00386     PwmOut_SetDuty( index, 0 );
00387     Io_SetTrue( ioA );
00388   }
00389 
00390   return CONTROLLER_OK;
00391 }

00450 /** \defgroup PWMOutOSC PWM Out - OSC
00451   Generate PWM signals with the Application Board via OSC.
00452   \ingroup OSC
00453   
00454   \section devices Devices
00455   There are 4 PWM controllers available on the Application Board, numbered 0 - 3.
00456   
00457   Each PWM Out controls a pair of Digital Outs - an A and a B channel:
00458   - PwmOut 0 - Digital Outs 0 (A) and 1 (B).
00459   - PwmOut 1 - Digital Outs 2 (A) and 3 (B).
00460   - PwmOut 2 - Digital Outs 4 (A) and 5 (B).
00461   - PwmOut 3 - Digital Outs 6 (A) and 7 (B).
00462   
00463   Each channel can also be set to invert the given PWM signal.
00464   
00465   \section properties Properties
00466   Each PWM Out has four properties - 'duty', 'invA', 'invB', and 'active'.
00467 
00468   \par Duty
00469   The 'duty' property corresponds to the duty at which a load connected to the output is being driven.
00470   This value can be both read and written.  The range of values expected by the board
00471   is from 0 - 1023.
00472   \par
00473   To generate a 75% on PWM signal, send a message like
00474   \verbatim /pwmout/1/duty 768 \endverbatim
00475   Leave the argument value off to read the duty:
00476   \verbatim /pwmout/1/duty \endverbatim
00477   
00478   \par invA
00479   The 'invA' property corresponds to the inversion of the A channel of a PWM Out.
00480   This value can be both read and written, and the range of values expected is simply 
00481   0 or 1.  1 means inverted and 0 means normal.  0 is the default.
00482   \par
00483   To set the A channel of the second PWM Out as inverted, send the message
00484   \verbatim /pwmout/1/invA 1 \endverbatim
00485   Note that the A channel of PWM Out 1 is Digital Out 2.
00486   
00487   \par invB
00488   The 'invB' property corresponds to the inversion of the B channel of a PWM Out.
00489   This value can be both read and written, and the range of values expected is simply 
00490   0 or 1.  1 means inverted and 0 means normal.  0 is the default.
00491   \par
00492   To set the B channel of the fourth PWM Out back to normal, send the message
00493   \verbatim /pwmout/1/invB 0 \endverbatim
00494   Note that the A channel of PWM Out 1 is Digital Out 7.
00495   
00496   \par Active
00497   The 'active' property corresponds to the active state of the PWM Out.
00498   If the device is set to be active, no other tasks will be able to
00499   use its 2 digital out lines.  If you're not seeing appropriate
00500   responses to your messages to the PWM Out, check the whether it's 
00501   locked by sending a message like
00502   \verbatim /pwmout/0/active \endverbatim
00503   \par
00504   If you're no longer using the PWM Out, it's a good idea to free the 2 Digital Outs by 
00505   sending the message
00506   \verbatim /pwmout/0/active 0 \endverbatim
00507 */
