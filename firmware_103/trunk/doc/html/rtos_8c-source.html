00028 /** \defgroup Rtos
00029   The FreeRTOS subsystem provides a real-time operating system (RTOS) for the MAKE Controller.
00030   FreeRTOS is an open source Real-Time Operating System.  It implements scheduling algorithms that 
00031   give time to "concurrent" tasks according to their given priority.  It allows programmers to 
00032   focus on programming each process on the board more or less separately, letting the RTOS 
00033   dictate when each of those tasks should be given processor time.
00034 
00035   Each process on the board is known as a task, and each task is generally implemented as a 
00036   continuous loop.  It is important to ensure that each task will block at some point so that
00037   other tasks will have a chance to run - otherwise the Controller will lock up. The easiest way to
00038   do this is with a call to Sleep( ) when the task is not urgent. 
00039   
00040   More info at http://www.freertos.org
00041 * \ingroup Controller
00042 * @{
00043 */
00044 
00045 /** 
00046   Put a task to sleep.
00047   This lets the processor give time to other tasks.  
00048   @param timems An integer specifying how long to sleep in milliseconds.
00049 */
00050 void Sleep( int timems )
00051 {
00052   vTaskDelay( timems / portTICK_RATE_MS );
00053 }
00054 
00055 /** 
00056   Give up the remaining time allotted to this task by the processor.
00057   The presently running task immeidately gives up the remaining time it has in the current timeslice 
00058   so that other tasks can run.  While Sleep() will wait the specified number of milliseconds 
00059   before continuing, TaskYield() will cause the task to take processor time again 
00060   as soon as it's available.
00061 */
00062 void TaskYield( )
00063 {
00064   taskYIELD(); 
00065 }
00066 
00067 /** 
00068   Create a new task.
00069   Always be sure to block a task when not urgent to give priority to other tasks.
00070   @param taskCode A pointer to the function that will run when the task starts.
00071   @param name A character string containing the name of the task.
00072   @param stackDepth An integer specifying in bytes the size of the task's stack.
00073   @param parameters A void* parameter which will be passed into the task at create time.
00074   @param priority An integer specifying the task's priority (0 lowest - 7 highest priority).
00075   @return A pointer to the handle for the newly created task.
00076 */
00077 void* TaskCreate( void (taskCode)( void*), char* name, int stackDepth, void* parameters, int priority )
00078 {
00079   void* taskHandle;
00080   xTaskCreate( taskCode, name, stackDepth, parameters, priority, &taskHandle );
00081   return taskHandle;
00082 }
00083 
00084 /** 
00085   Delete an existing task.
00086   @param task A pointer to the handle of the task to be deleted.
00087 */
00088 void TaskDelete( void* task )
00089 {
00090   vTaskDelete( task );

00093 /** 
00094   Specify that a task is entering a period during which it should not be interrupted.  
00095   Be sure to minimize use of this, and to call TaskExitCritical() as soon as the 
00096   task has completed its critical stage.  Other tasks can't get any processor time while in
00097   this state, and performance might be compromised.
00098 */
00099 void  TaskEnterCritical( )
00100 {
00101   taskENTER_CRITICAL();
00102 }
00103 
00104 /** 
00105   Specify that a task is exiting a critical period, in response to TaskEnterCritical().
00106   Once the task exits a critical period, the RTOS will resume switching this task as usual.  
00107 */
00108 void  TaskExitCritical( )
00109 {
00110   taskEXIT_CRITICAL();
00111 }
