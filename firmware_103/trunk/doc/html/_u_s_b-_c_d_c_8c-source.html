00034   USB Communications Device Class driver.
00035   Implements task vUSBCDCTask and provides an Abstract Control Model serial 
00036   interface.  Control is through endpoint 0, device-to-host notification is 
00037   provided by interrupt-in endpoint 3, and raw data is transferred through 
00038   bulk endpoints 1 and 2.
00039 
00040   - developed from original FreeRTOS HID example by Scott Miller
00041   - modified to support 3.2 GCC by najay
00042 */

00136 void vUSBCDCTask( void *pvParameters )
00137 {
00138 xISRStatus *pxMessage;
00139 unsigned portLONG ulStatus;
00140 unsigned portLONG ulRxBytes;
00141 unsigned portCHAR ucByte;
00142 portBASE_TYPE xByte;
00143 
00144   ( void ) pvParameters;
00145 
00146   /* Disconnect USB device from hub.  For debugging - causes host to register reset */
00147   portENTER_CRITICAL();
00148      vDetachUSBInterface();
00149   portEXIT_CRITICAL();
00150   
00151   vTaskDelay( portTICK_RATE_MS * 60 );
00152 
00153   /* Init USB interface */
00154   portENTER_CRITICAL();
00155     vInitUSBInterface();
00156   portEXIT_CRITICAL();
00157   
00158   /* Main task loop.  Process incoming endpoint 0 interrupts, handle data transfers. */
00159    
00160   /* MAKINGTHINGS: USB RUNNING */
00161   Usb_Running = 1;
00162 
00163   for( ;; )
00164   {
00165     /* Look for data coming from the ISR. */
00166     if( xQueueReceive( xUSBInterruptQueue, &pxMessage, usbSHORTEST_DELAY ) )
00167     {
00168       if( pxMessage->ulISR & AT91C_UDP_EPINT0 )
00169       {
00170         /* All endpoint 0 interrupts are handled here. */
00171         prvProcessEndPoint0Interrupt( pxMessage );
00172       }
00173 
00174       if( pxMessage->ulISR & AT91C_UDP_ENDBUSRES )
00175       {
00176         /* End of bus reset - reset the endpoints and de-configure. */
00177         prvResetEndPoints();    
00178       }
00179     }
00180     
00181     /* See if we're ready to send and receive data. */
00182     if( eDriverState == eREADY_TO_SEND && ucControlState ) 
00183     {
00184       if( ( !(AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] & AT91C_UDP_TXPKTRDY) ) && uxQueueMessagesWaiting( xTxCDC ) )
00185       {
00186         for( xByte = 0; xByte < 64; xByte++ )
00187         {          
00188           if( !xQueueReceive( xTxCDC, &ucByte, 0 ) ) 
00189           {
00190             /* No data buffered to transmit. */
00191             break;
00192           }
00193 
00194           /* Got a byte to transmit. */
00195           AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_2 ] = ucByte;
00196         } 
00197         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] |= AT91C_UDP_TXPKTRDY;
00198       }
00199 
00200       /* Check for incoming data (host-to-device) on endpoint 1. */
00201       while( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] & (AT91C_UDP_RX_DATA_BK0 | AT91C_UDP_RX_DATA_BK1) )
00202       {
00203         ulRxBytes = (AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] >> 16) & usbRX_COUNT_MASK;
00204 
00205         /* Only process FIFO if there's room to store it in the queue */
00206         if( ulRxBytes < ( USB_CDC_QUEUE_SIZE - uxQueueMessagesWaiting( xRxCDC ) ) )
00207         {
00208           while( ulRxBytes-- )
00209           {
00210             ucByte = AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_1 ];
00211             xQueueSend( xRxCDC, &ucByte, 0 );
00212           }
00213 
00214           /* Release the FIFO */
00215           portENTER_CRITICAL();
00216           {
00217             ulStatus = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ];
00218             usbCSR_CLEAR_BIT( &ulStatus, uiCurrentBank );
00219             AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] = ulStatus;
00220           }
00221           portEXIT_CRITICAL();
00222 
00223           /* Re-enable endpoint 1's interrupts */
00224           AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT1;
00225         
00226           /* Update the current bank in use */
00227           if( uiCurrentBank == AT91C_UDP_RX_DATA_BK0 ) 
00228           {
00229             uiCurrentBank = AT91C_UDP_RX_DATA_BK1;
00230           }
00231           else 
00232           {
00233             uiCurrentBank = AT91C_UDP_RX_DATA_BK0;
00234           }
00235 
00236         }
00237         else 
00238         {
00239           break;
00240         }
00241       }
00242     }
00243   }
00244 }

00265 static void prvSendZLP( void )
00266 {
00267 unsigned portLONG ulStatus;
00268 
00269   /* Wait until the FIFO is free - even though we are not going to use it.
00270   THERE IS NO TIMEOUT HERE! */
00271   while( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_TXPKTRDY )
00272   {
00273     vTaskDelay( usbSHORTEST_DELAY );
00274   }
00275 
00276   portENTER_CRITICAL();
00277   {
00278     /* Cancel any further pending data */
00279     pxControlTx.ulTotalDataLength = pxControlTx.ulNextCharIndex;
00280 
00281     /* Set the TXPKTRDY bit to cause a transmission with no data. */
00282     ulStatus = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ];
00283     usbCSR_SET_BIT( &ulStatus, AT91C_UDP_TXPKTRDY );
00284     AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] = ulStatus;
00285   }
00286   portEXIT_CRITICAL();
00287 }

00305 static void prvResetEndPoints( void )
00306 {
00307 unsigned portLONG ulTemp;
00308 
00309   eDriverState = eJUST_RESET;
00310   ucControlState = 0;
00311 
00312   /* Reset all the end points. */
00313   AT91C_BASE_UDP->UDP_RSTEP  = usbEND_POINT_RESET_MASK;
00314   AT91C_BASE_UDP->UDP_RSTEP  = ( unsigned portLONG ) 0x00;
00315 
00316   /* Enable data to be sent and received. */
00317   AT91C_BASE_UDP->UDP_FADDR = AT91C_UDP_FEN;
00318 
00319   /* Repair the configuration end point. */
00320   portENTER_CRITICAL();
00321   {
00322     ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ];
00323     usbCSR_SET_BIT( &ulTemp, ( ( unsigned portLONG ) ( AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_CTRL ) ) );
00324     AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] = ulTemp;
00325     AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT0;
00326   }
00327   portEXIT_CRITICAL();
00328   uiCurrentBank = AT91C_UDP_RX_DATA_BK0;
00329 }

00332 static void prvProcessEndPoint0Interrupt( xISRStatus *pxMessage )
00333 {
00334 static xUSB_REQUEST xRequest;
00335 unsigned portLONG ulRxBytes;
00336 
00337   /* Get number of bytes received, if any */
00338   ulRxBytes = pxMessage->ulCSR0 >> 16;
00339   ulRxBytes &= usbRX_COUNT_MASK;
00340 
00341   if( pxMessage->ulCSR0 & AT91C_UDP_TXCOMP )
00342   {
00343     /* We received a TX complete interrupt.  What we do depends on
00344     what we sent to get this interrupt. */
00345 
00346     if( eDriverState == eJUST_GOT_CONFIG )
00347     {
00348       /* We sent an acknowledgement of a SET_CONFIG request.  We
00349       are now at the end of the enumeration.
00350       
00351       TODO: Config 0 sets unconfigured state, should enter Address state.
00352       Request for unsupported config should stall. */
00353       AT91C_BASE_UDP->UDP_GLBSTATE = AT91C_UDP_CONFG;
00354       
00355       /* Set up endpoints */
00356       portENTER_CRITICAL();
00357       {
00358         unsigned portLONG ulTemp;
00359 
00360         /* Set endpoint 1 to bulk-out */
00361         ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ];         
00362         usbCSR_SET_BIT( &ulTemp, AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_BULK_OUT );
00363         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_1 ] = ulTemp;   
00364         AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT1;
00365         /* Set endpoint 2 to bulk-in */
00366         ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ];         
00367         usbCSR_SET_BIT( &ulTemp, AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_BULK_IN );
00368         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_2 ] = ulTemp;   
00369         AT91C_BASE_UDP->UDP_IER = AT91C_UDP_EPINT2;
00370           /* Set endpoint 3 to interrupt-in, enable it, and enable interrupts */
00371         ulTemp = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_3 ];         
00372         usbCSR_SET_BIT( &ulTemp, AT91C_UDP_EPEDS | AT91C_UDP_EPTYPE_INT_IN );
00373         AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_3 ] = ulTemp;   
00374         /*AT91F_UDP_EnableIt( AT91C_BASE_UDP, AT91C_UDP_EPINT3 );        */
00375       }
00376       portEXIT_CRITICAL();
00377 
00378       eDriverState = eREADY_TO_SEND;
00379     }   
00380     else if( eDriverState == eJUST_GOT_ADDRESS )
00381     {
00382       /* We sent an acknowledgement of a SET_ADDRESS request.  Move
00383       to the addressed state. */
00384       if( ulReceivedAddress != ( unsigned portLONG ) 0 )
00385       {     
00386         AT91C_BASE_UDP->UDP_GLBSTATE = AT91C_UDP_FADDEN;
00387       }
00388       else
00389       {
00390         AT91C_BASE_UDP->UDP_GLBSTATE = 0;
00391       }     
00392 
00393       AT91C_BASE_UDP->UDP_FADDR = ( AT91C_UDP_FEN | ulReceivedAddress );    
00394       eDriverState = eNOTHING;
00395     }
00396     else
00397     {   
00398       /* The TXCOMP was not for any special type of transmission.  See
00399       if there is any more data to send. */
00400       prvSendNextSegment();
00401     }
00402   }
00403 
00404   if( pxMessage->ulCSR0 & AT91C_UDP_RX_DATA_BK0 )
00405   {
00406     /* Received a control data packet.  May be a 0-length ACK or a data stage. */
00407     unsigned portCHAR ucBytesToGet;
00408    
00409     /* Got data.  Cancel any outgoing data. */
00410     pxControlTx.ulNextCharIndex = pxControlTx.ulTotalDataLength;
00411     
00412      /* Determine how many bytes we need to receive. */
00413     ucBytesToGet = pxControlRx.ulTotalDataLength - pxControlRx.ulNextCharIndex;
00414     if( ucBytesToGet > ulRxBytes ) 
00415     { 
00416       ucBytesToGet = ulRxBytes;
00417     }
00418 
00419     /* If we're not expecting any data, it's an ack - just quit now. */
00420     if( !ucBytesToGet )
00421     {
00422        return;
00423     }
00424 
00425     /* Get the required data and update the index. */
00426     memcpy( pxControlRx.ucBuffer, pxMessage->ucFifoData, ucBytesToGet );
00427     pxControlRx.ulNextCharIndex += ucBytesToGet;  
00428   }
00429 
00430   if( pxMessage->ulCSR0 & AT91C_UDP_RXSETUP )
00431   {
00432     /* Received a SETUP packet.  May be followed by data packets. */
00433 
00434     if( ulRxBytes >= usbEXPECTED_NUMBER_OF_BYTES )
00435     {       
00436       /* Create an xUSB_REQUEST variable from the raw bytes array. */
00437 
00438       xRequest.ucReqType = pxMessage->ucFifoData[ usbREQUEST_TYPE_INDEX ];
00439       xRequest.ucRequest = pxMessage->ucFifoData[ usbREQUEST_INDEX ];
00440 
00441       xRequest.usValue = pxMessage->ucFifoData[ usbVALUE_HIGH_BYTE ];
00442       xRequest.usValue <<= 8;
00443       xRequest.usValue |= pxMessage->ucFifoData[ usbVALUE_LOW_BYTE ];
00444             
00445       xRequest.usIndex = pxMessage->ucFifoData[ usbINDEX_HIGH_BYTE ];
00446       xRequest.usIndex <<= 8;
00447       xRequest.usIndex |= pxMessage->ucFifoData[ usbINDEX_LOW_BYTE ];
00448       
00449       xRequest.usLength = pxMessage->ucFifoData[ usbLENGTH_HIGH_BYTE ];
00450       xRequest.usLength <<= 8;
00451       xRequest.usLength |= pxMessage->ucFifoData[ usbLENGTH_LOW_BYTE ];
00452 
00453       pxControlRx.ulNextCharIndex = 0;
00454       if( ! (xRequest.ucReqType & 0x80) ) /* Host-to-Device transfer, may need to get data first */
00455       {
00456         if( xRequest.usLength > usbMAX_CONTROL_MESSAGE_SIZE )
00457         { 
00458           /* Too big!  No space for control data, stall and abort. */
00459           prvSendStall();
00460           return;
00461         }
00462 
00463         pxControlRx.ulTotalDataLength = xRequest.usLength;
00464       }
00465       else
00466       {
00467         /* We're sending the data, don't wait for any. */
00468         pxControlRx.ulTotalDataLength = 0; 
00469       }
00470     }
00471   }
00472 
00473   /* See if we've got a pending request and all its associated data ready */
00474   if( ( pxMessage->ulCSR0 & ( AT91C_UDP_RX_DATA_BK0 | AT91C_UDP_RXSETUP ) ) 
00475     && ( pxControlRx.ulNextCharIndex >= pxControlRx.ulTotalDataLength ) )
00476   {
00477     unsigned portCHAR ucRequest;
00478 
00479     /* Manipulate the ucRequestType and the ucRequest parameters to 
00480     generate a zero based request selection.  This is just done to 
00481     break up the requests into subsections for clarity.  The 
00482     alternative would be to have more huge switch statement that would
00483     be difficult to optimise. */
00484     ucRequest = ( ( xRequest.ucReqType & 0x60 ) >> 3 );
00485     ucRequest |= ( xRequest.ucReqType & 0x03 );
00486       
00487     switch( ucRequest )
00488     {
00489       case usbSTANDARD_DEVICE_REQUEST:  
00490         /* Standard Device request */
00491         prvHandleStandardDeviceRequest( &xRequest );
00492         break;
00493 
00494       case usbSTANDARD_INTERFACE_REQUEST: 
00495         /* Standard Interface request */
00496         prvHandleStandardInterfaceRequest( &xRequest );
00497         break;
00498 
00499       case usbSTANDARD_END_POINT_REQUEST: 
00500         /* Standard Endpoint request */
00501         prvHandleStandardEndPointRequest( &xRequest );
00502         break;
00503 
00504       case usbCLASS_INTERFACE_REQUEST:  
00505         /* Class Interface request */
00506         prvHandleClassInterfaceRequest( &xRequest );
00507         break;
00508 
00509       default:  /* This is not something we want to respond to. */
00510         prvSendStall(); 
00511     }
00512   }
00513 }

00516 static void prvGetStandardDeviceDescriptor( xUSB_REQUEST *pxRequest )
00517 {
00518   /* The type is in the high byte.  Return whatever has been requested. */
00519   switch( ( pxRequest->usValue & 0xff00 ) >> 8 )
00520   {
00521     case usbDESCRIPTOR_TYPE_DEVICE:
00522       prvSendControlData( ( unsigned portCHAR * ) &pxDeviceDescriptor, pxRequest->usLength, sizeof( pxDeviceDescriptor ), pdTRUE );
00523       break;
00524 
00525     case usbDESCRIPTOR_TYPE_CONFIGURATION:
00526       prvSendControlData( ( unsigned portCHAR * ) &( pxConfigDescriptor ), pxRequest->usLength, sizeof( pxConfigDescriptor ), pdTRUE );
00527       break;
00528 
00529     case usbDESCRIPTOR_TYPE_STRING:
00530 
00531       /* The index to the string descriptor is the lower byte. */
00532       switch( pxRequest->usValue & 0xff )
00533       {     
00534         case usbLANGUAGE_STRING:
00535           prvSendControlData( ( unsigned portCHAR * ) &pxLanguageStringDescriptor, pxRequest->usLength, sizeof(pxLanguageStringDescriptor), pdTRUE );
00536           break;
00537 
00538         case usbMANUFACTURER_STRING:
00539           prvSendControlData( ( unsigned portCHAR * ) &pxManufacturerStringDescriptor, pxRequest->usLength, sizeof( pxManufacturerStringDescriptor ), pdTRUE );
00540           break;
00541 
00542         case usbPRODUCT_STRING:
00543           prvSendControlData( ( unsigned portCHAR * ) &pxProductStringDescriptor, pxRequest->usLength, sizeof( pxProductStringDescriptor ), pdTRUE );
00544           break;
00545 
00546         case usbCONFIGURATION_STRING:
00547           prvSendControlData( ( unsigned portCHAR * ) &pxConfigurationStringDescriptor, pxRequest->usLength, sizeof( pxConfigurationStringDescriptor ), pdTRUE );
00548           break;
00549 
00550         case usbINTERFACE_STRING:
00551           prvSendControlData( ( unsigned portCHAR * ) &pxInterfaceStringDescriptor, pxRequest->usLength, sizeof( pxInterfaceStringDescriptor ), pdTRUE );
00552           break;
00553 
00554         default:
00555           prvSendStall();
00556           break;
00557       }
00558       break;
00559 
00560     default:
00561       prvSendStall();
00562       break;
00563   }
00564 }

00567 static void prvHandleStandardDeviceRequest( xUSB_REQUEST *pxRequest )
00568 {
00569 unsigned portSHORT usStatus = 0;
00570 
00571   switch( pxRequest->ucRequest )
00572   {
00573     case usbGET_STATUS_REQUEST:
00574       /* Just send two byte dummy status. */
00575       prvSendControlData( ( unsigned portCHAR * ) &usStatus, sizeof( usStatus ), sizeof( usStatus ), pdFALSE );
00576       break;
00577 
00578     case usbGET_DESCRIPTOR_REQUEST:
00579       /* Send device descriptor */
00580       prvGetStandardDeviceDescriptor( pxRequest );
00581       break;
00582 
00583     case usbGET_CONFIGURATION_REQUEST:
00584       /* Send selected device configuration */
00585       prvSendControlData( ( unsigned portCHAR * ) &ucUSBConfig, sizeof( ucUSBConfig ), sizeof( ucUSBConfig ), pdFALSE );
00586       break;
00587 
00588     case usbSET_FEATURE_REQUEST:
00589       prvSendZLP();
00590       break;
00591 
00592     case usbSET_ADDRESS_REQUEST:      
00593       /* Get assigned address and send ack, but don't implement new address until we get a TXCOMP */
00594       prvSendZLP();     
00595       eDriverState = eJUST_GOT_ADDRESS;     
00596       ulReceivedAddress = ( unsigned portLONG ) pxRequest->usValue;
00597       break;
00598 
00599     case usbSET_CONFIGURATION_REQUEST:
00600       /* Ack SET_CONFIGURATION request, but don't implement until TXCOMP */
00601       ucUSBConfig = ( unsigned portCHAR ) ( pxRequest->usValue & 0xff );
00602       eDriverState = eJUST_GOT_CONFIG;
00603       prvSendZLP();
00604       break;
00605 
00606     default:

00645       break;

00737   { 
00738     /* Not enough RAM to create queues!. */

00819 static void prvSendControlData( unsigned portCHAR *pucData, unsigned portSHORT usRequestedLength, unsigned portLONG ulLengthToSend, portLONG lSendingDescriptor )
00820 {
00821   if( ( ( unsigned portLONG ) usRequestedLength < ulLengthToSend ) )
00822   {
00823     /* Cap the data length to that requested. */
00824     ulLengthToSend = ( unsigned portSHORT ) usRequestedLength;
00825   }
00826   else if( ( ulLengthToSend < ( unsigned portLONG ) usRequestedLength ) && lSendingDescriptor )
00827   {
00828     /* We are sending a descriptor.  If the descriptor is an exact 
00829     multiple of the FIFO length then it will have to be terminated
00830     with a NULL packet.  Set the state to indicate this if
00831     necessary. */
00832     if( ( ulLengthToSend % usbFIFO_LENGTH ) == 0 )
00833     {
00834       eDriverState = eSENDING_EVEN_DESCRIPTOR;
00835     }
00836   }
00837 
00838   /* Here we assume that the previous message has been sent.  THERE IS NO
00839   BUFFER OVERFLOW PROTECTION HERE.
00840 
00841   Copy the data to send into the buffer as we cannot send it all at once
00842   (if it is greater than 8 bytes in length). */
00843   memcpy( pxControlTx.ucBuffer, pucData, ulLengthToSend );
00844 
00845   /* Reinitialise the buffer index so we start sending from the start of 
00846   the data. */
00847   pxControlTx.ulTotalDataLength = ulLengthToSend;
00848   pxControlTx.ulNextCharIndex = ( unsigned portLONG ) 0;
00849 
00850   /* Send the first 8 bytes now.  The rest will get sent in response to 
00851   TXCOMP interrupts. */
00852   prvSendNextSegment();
00853 }
00854 /*-----------------------------------------------------------*/
00855 
00856 static void prvSendNextSegment( void )
00857 {
00858 volatile unsigned portLONG ulNextLength, ulStatus, ulLengthLeftToSend;
00859 
00860   /* Is there any data to send? */
00861   if( pxControlTx.ulTotalDataLength > pxControlTx.ulNextCharIndex )
00862   {
00863     ulLengthLeftToSend = pxControlTx.ulTotalDataLength - pxControlTx.ulNextCharIndex;
00864   
00865     /* We can only send 8 bytes to the fifo at a time. */
00866     if( ulLengthLeftToSend > usbFIFO_LENGTH )
00867     {
00868       ulNextLength = usbFIFO_LENGTH;
00869     }
00870     else
00871     {
00872       ulNextLength = ulLengthLeftToSend;
00873     }
00874 
00875     /* Wait until we can place data in the fifo.  THERE IS NO TIMEOUT
00876     HERE! */
00877     while( AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] & AT91C_UDP_TXPKTRDY )
00878     {
00879       vTaskDelay( usbSHORTEST_DELAY );
00880     }
00881 
00882     /* Write the data to the FIFO. */
00883     while( ulNextLength > ( unsigned portLONG ) 0 )
00884     {
00885       AT91C_BASE_UDP->UDP_FDR[ usbEND_POINT_0 ] = pxControlTx.ucBuffer[ pxControlTx.ulNextCharIndex ];
00886   
00887       ulNextLength--;
00888       pxControlTx.ulNextCharIndex++;
00889     }
00890   
00891     /* Start the transmission. */
00892     portENTER_CRITICAL();
00893     {
00894       ulStatus = AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ];
00895       usbCSR_SET_BIT( &ulStatus, ( ( unsigned portLONG ) 0x10 ) );
00896       AT91C_BASE_UDP->UDP_CSR[ usbEND_POINT_0 ] = ulStatus;
00897     }
00898     portEXIT_CRITICAL();
00899   }
00900   else
00901   {
00902     /* There is no data to send.  If we were sending a descriptor and the 
00903     descriptor was an exact multiple of the max packet size then we need
00904     to send a null to terminate the transmission. */
00905     if( eDriverState == eSENDING_EVEN_DESCRIPTOR )
00906     {
00907       prvSendZLP();
00908       eDriverState = eNOTHING;
00909     }
00910   }
00911 }
