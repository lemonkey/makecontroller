00059 /** \defgroup Servo
00060 * The Servo Motor subsystem controls speed and position control for up to 4 standard servo motors.
00061 * Up to 4 standard servo motors can be controlled with the MAKE Application Board.\n\n
00062 * Standard servos have a range of motion of approximately 180 degrees, although this varies from motor to motor.
00063 * Be sure to plug in the connector with the correct orientation with regard to the GND/5V signals on the board.\n\n
00064 * Set the position of each motor by specifying a value from 0-1023, which will be mapped to the full range of motion
00065 * of the motor.  You can also specify the speed with which the motors will respond to new position commands - a high
00066 * value will result in an immediate response, while a lower value can offer some smoothing when appropriate.
00067 * \ingroup AppBoard
00068 * @{
00069 */

00104   Most servos move within a 180 degree range.  The 0 - 1023 range of the position parameter represents
00105   the full range of motion of the motor.
00106   @param index An integer specifying which servo (0 - 3).
00107   @param position An integer specifying the servo position (0 - 1023).
00108   @return status (0 = OK).
00109 */
00110 int Servo_SetPosition( int index, int position )
00111 {
00112   if ( index < 0 || index >= SERVO_COUNT )
00113     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00114 
00115   if ( Servo.control[ index ].users < 1 )
00116   {
00117     int status = Servo_Start( index );
00118     if ( status != CONTROLLER_OK )
00119       return status;
00120   }
00121 
00122   if ( position < 0 )
00123     position = 0;
00124   if ( position > 1023 )
00125     position = 1023;
00126 
00127   DisableFIQFromThumb();
00128   Servo.control[ index ].positionRequested = position << 6;
00129   EnableFIQFromThumb();
00130 
00131   return CONTROLLER_OK;

00207 int Servo_Start( int index )
00208 {
00209   int status;
00210 
00211   if ( index < 0 || index >= SERVO_COUNT )
00212     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00213 
00214   ServoControl* sc = &Servo.control[ index ]; 
00215   if ( sc->users++ == 0 )
00216   {
00217     if ( Servo.users++ == 0 )
00218     {
00219       int status = Servo_Init();
00220       if ( status != CONTROLLER_OK )
00221       {
00222         // I guess not then
00223         sc->users--;
00224         Servo.users--;
00225         return status;
00226       }   
00227     }
00228 
00229     int io = Servo_GetIo( index );
00230 
00231     // Try to lock the servo output
00232     status = Io_Start( io, true );
00233     if ( status != CONTROLLER_OK )
00234     {
00235       // Damn
00236       Servo.users--;
00237       sc->users--;
00238       return status;
00239     }
00240 
00241     Io_PioEnable( io );
00242     Io_SetTrue( io );
00243     Io_SetOutput( io );
00244 
00245     ServoControl* s = &Servo.control[ index ]; 
00246     s->position = 512 << 6;
00247     s->speed = -1;

00254 {
00255   if ( index < 0 || index >= SERVO_COUNT )
00256     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00257 
00258   ServoControl* sc = &Servo.control[ index ]; 
00259 
00260   if ( sc->users <= 0 )

00264   {
00265     int io = Servo_GetIo( index );

00300 int Servo_Init()
00301 {
00302   ServoControl* s = &Servo.control[ 0 ]; 
00303   s->pIoBase = AT91C_BASE_PIOB;
00304   s->pin = AT91C_PIO_PB24;
00305   s++;
00306   s->pIoBase = AT91C_BASE_PIOA;
00307   s->pin = AT91C_PIO_PA23;
00308   s++;
00309   s->pIoBase = AT91C_BASE_PIOA;
00310   s->pin = AT91C_PIO_PA21;
00311   s++;
00312   s->pIoBase = AT91C_BASE_PIOA;
00313   s->pin = AT91C_PIO_PA22;
00314   s++;
00315 
00316   // Global init stuff
00317   Servo.state = 0;
00318   Servo.gap = 1882; // forces 64'ish cycles a second (1894 + 2012 = 3906, 3906 * 4 = 15624, 15624 * 64 = 999936)
00319   Servo.index = 0;
00320 
00321   // Configure TC by enabling PWM clock
00322   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_TC1;
00323                                     
00324   unsigned int mask ;
00325   mask = 0x1 << AT91C_ID_TC1 | 0x01;
00326 
00327   /* Disable the interrupt on the interrupt controller */
00328   AT91C_BASE_AIC->AIC_IDCR = mask;
00329 
00330   AT91C_BASE_AIC->AIC_SVR[ AT91C_ID_FIQ ] = (unsigned int)FastIsr;
00331 
00332   /* Store the Source Mode Register */
00333   AT91C_BASE_AIC->AIC_SMR[ AT91C_ID_TC1 ] = AT91C_AIC_SRCTYPE_INT_HIGH_LEVEL | 7  ;
00334   /* Clear the interrupt on the interrupt controller */
00335   AT91C_BASE_AIC->AIC_ICCR = mask ;
00336 
00337   // Set the timer up.  We want just the basics, except when the timer compares 
00338   // with RC, retrigger
00339   //
00340   // MCK is 47923200
00341   // DIV1: A tick MCK/2 times a second
00342   // This makes every tick every 41.73344ns
00343   // DIV2: A tick MCK/8 times a second
00344   // This makes every tick every 167ns
00345   // DIV3: A tick MCK/32 times a second
00346   // This makes every tick every 668ns
00347   // DIV4: A tick MCK/128 times a second
00348   // This makes every tick every 2.671us
00349   // DIV5: A tick MCK/1024 times a second
00350   // This makes every tick every 21.368us
00351   AT91C_BASE_TC1->TC_CMR = AT91C_TC_CLKS_TIMER_DIV2_CLOCK |  AT91C_TC_CPCTRG;
00352                    
00353   // Only really interested in interrupts when the RC happens
00354   AT91C_BASE_TC1->TC_IDR = 0xFF; 
00355   AT91C_BASE_TC1->TC_IER = AT91C_TC_CPCS; 
00356 
00357   // load the RC value with something
00358   AT91C_BASE_TC1->TC_RC = 0xFFFF;
00359 
00360   // Make it fast forcing
00361   AT91C_BASE_AIC->AIC_FFER = 0x1 << AT91C_ID_TC1;
00362 
00363   // Enable the interrupt
00364   AT91C_BASE_AIC->AIC_IECR = mask;
00365 
00366   // Enable the device
00367   AT91C_BASE_TC1->TC_CCR = AT91C_TC_CLKEN | AT91C_TC_SWTRG;
00368 
00369   return CONTROLLER_OK;
00370 }

00381 /** \defgroup ServoOSC Servo - OSC
00382   Control Servo motors with the Application Board via OSC.
00383   \ingroup OSC
00384   
00385   \section devices Devices
00386   There are 4 Servo controllers available on the Application Board, numbered 0 - 3.\n
00387   See the servo section in the Application Board user's guide for more information
00388   on hooking up servos to the board.
00389   
00390   \section properties Properties
00391   Each servo controller has three properties - 'position', 'speed', and 'active'.
00392 
00393   \par Position
00394   The 'position' property corresponds to the position of the servo motor within its range of motion.
00395   This value can be both read and written.  The relevant range of values is from 0 - 1023.  
00396   A position of 0 sets the servo to one extreme of its range of motion and 1023 to its opposite extreme.
00397   \par
00398   To set the first servo to one quarter its position, send the message
00399   \verbatim /servo/0/position 256 \endverbatim
00400   Leave the argument value off to read the position of the servo:
00401   \verbatim /servo/0/position \endverbatim
00402   
00403   \par Speed
00404   The 'speed' property corresponds to the speed with which the servo responds to changes 
00405   of position.
00406   This value can be both read and written, and the range of values is 0 - 1023.  A speed of 1023
00407   means the servo responds immediately, and lower values will result in slower, and often smoother, 
00408   responses.  
00409   \par
00410   To set the speed of the first servo to just under full speed, send a message like
00411   \verbatim /servo/0/speed 975 \endverbatim
00412   Adjust the argument value to one that suits your application.\n
00413   Leave the argument value off to read the position of the servo:
00414   \verbatim /servo/0/speed \endverbatim
00415   
00416   \par Active
00417   The 'active' property corresponds to the active state of the servo.
00418   If the servo is set to be active, no other tasks will be able to
00419   write to the same I/O lines.  If you're not seeing appropriate
00420   responses to your messages to a servo, check the whether it's 
00421   locked by sending a message like
00422   \verbatim /servo/3/state \endverbatim
00423 */
