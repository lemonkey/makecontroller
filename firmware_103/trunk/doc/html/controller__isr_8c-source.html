00034 void DisableFIQFromThumb( void )
00035 {
00036   asm volatile ( "STMDB SP!, {R0}" );   /* Push R0.                 */
00037   asm volatile ( "MRS   R0, CPSR" );    /* Get CPSR.                */
00038   asm volatile ( "ORR   R0, R0, #0x40" ); /* Disable FIQ.           */
00039   asm volatile ( "MSR   CPSR, R0" );    /* Write back modified value.       */
00040   asm volatile ( "LDMIA SP!, {R0}" );   /* Pop R0.                  */
00041   asm volatile ( "BX    R14" );       /* Return back to thumb.          */
00042 }
00043     
00044 void EnableFIQFromThumb( void )
00045 {
00046   asm volatile ( "STMDB SP!, {R0}" );   /* Push R0.                 */  
00047   asm volatile ( "MRS   R0, CPSR" );    /* Get CPSR.                */  
00048   asm volatile ( "BIC   R0, R0, #0x40" ); /* Enable FIQ.              */  
00049   asm volatile ( "MSR   CPSR, R0" );    /* Write back modified value.       */  
00050   asm volatile ( "LDMIA SP!, {R0}" );   /* Pop R0.                  */
00051   asm volatile ( "BX    R14" );       /* Return back to thumb.          */
00052 }

00059 void FastIsr( void )
00060 {
00061   int status;
00062   
00063   status = AT91C_BASE_TC0->TC_SR;
00064   if ( status & AT91C_TC_CPCS )
00065   {
00066     AT91C_BASE_AIC->AIC_ICCR = 0x1 << AT91C_ID_TC0;
00067   }
00068 
00069   status = AT91C_BASE_TC1->TC_SR;
00070   if ( status & AT91C_TC_CPCS )
00071   {
00072     AT91C_BASE_AIC->AIC_ICCR = 0x1 << AT91C_ID_TC1;
00073 
00074     // int jitter = AT91C_BASE_TC1->TC_CV;
00075     int period;
00076 
00077     switch ( Servo.state )
00078     {
00079       case 0:
00080       {
00081         if ( ++Servo.index >= SERVO_COUNT || Servo.index < 0 )
00082           Servo.index = 0;
00083         ServoControl* s = &Servo.control[ Servo.index ];
00084         
00085         if ( s->position != s->positionRequested )
00086         {
00087           if ( s->speed == -1 )
00088             s->position = s->positionRequested;
00089           else
00090           {
00091             int diff = s->positionRequested - s->position;
00092             if ( diff < 0 )
00093             {
00094               s->position -= s->speed;
00095               if ( s->position < s->positionRequested )
00096                 s->position = s->positionRequested;
00097             }
00098             else
00099             {
00100               s->position += s->speed;
00101               if ( s->position > s->positionRequested )
00102                 s->position = s->positionRequested;
00103             }
00104           }
00105         }
00106 
00107         period = s->position >> 6;
00108         if ( period >= 0 && period <= 1023 )
00109         {
00110           s->pIoBase->PIO_CODR = s->pin;
00111         }
00112         else
00113           period = 1023;
00114         AT91C_BASE_TC1->TC_RC = ( period + 988 ) * 6;
00115         Servo.state = 1;
00116         break;
00117       }
00118       case 1:
00119       {
00120         ServoControl* s = &Servo.control[ Servo.index ];
00121         period = s->position >> 6;
00122         s->pIoBase->PIO_SODR = s->pin;
00123         AT91C_BASE_TC1->TC_RC = ( Servo.gap + ( 1023 - period ) ) * 6;
00124         Servo.state = 0;
00125         break;
00126       }
00127     }
00128   }
00129  
00130   status = AT91C_BASE_TC2->TC_SR;
00131   if ( status & AT91C_TC_CPCS )
00132   {
00133     AT91C_BASE_AIC->AIC_ICCR = 0x1 << AT91C_ID_TC2;
00134   }
00135 }
