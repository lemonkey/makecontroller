00061 /** \defgroup Network
00062   The Network subsystem manages the Ethernet controller.  
00063   This subsystem is based on LwIP, an open source TCP/IP stack.  It provides TCP
00064   and UPD sockets and handles storage and retrieval of the IP address, address 
00065   mask, gateway address.  Also handled is the device's MAC address.
00066 
00067   IP Address, mask and gatway default to 192.168.0.200, 255.255.255.0 and 192.168.0.1.
00068   The MAC address defaults to AC.DE.48.55.x.y where x & y are calculated from the 
00069   unit's serial number which is handled by the \ref System subsystem.
00070 
00071   From OSC this subsystem can be addressed as "network".  It has the following 
00072   properties:
00073     \li active - can be used to activate / deactivate the Network subsystem and to read its status
00074     \li address - reading or writing the ip address in string a1.a2.a3.a4 form

00076     \li gateway - reading or writing the ip address mask in string g1.g2.g3.g4 form
00077     \li valid - reading or asserting the validity of the currently stored address set
00078     \li mac - reading the MAC address in string form.  To change it see the \ref System subsystem

00202 /**
00203   Create a checksum for the current address settings and store it in EEPROM.
00204   This should be called each time an address setting is changed so that if
00205   the board gets powered down, it will know when it comes back up whether or
00206   not the address settings is currently has are valid.
00207 
00208   @param v An integer specifying whether to validate the current settings (1)
00209   or to force them to be invalidated (0).
00210   Passing in 0 returns the address settings to their factory defaults.
00211   @return 0 on success.
00212 */
00213 int Network_SetValid( int v )
00214 {
00215   if ( v )
00216   {
00217     int address;
00218     Eeprom_Read( EEPROM_SYSTEM_NET_ADDRESS, (uchar*)&address, 4 );
00219   
00220     int mask;
00221     Eeprom_Read( EEPROM_SYSTEM_NET_MASK, (uchar*)&mask, 4 );
00222   
00223     int gateway;
00224     Eeprom_Read( EEPROM_SYSTEM_NET_GATEWAY, (uchar*)&gateway, 4 );
00225   
00226     int total = address + mask + gateway;
00227     Eeprom_Write( EEPROM_SYSTEM_NET_CHECK, (uchar*)&total, 4 );
00228 
00229     Network_Valid = NET_VALID;
00230   }
00231   else
00232   {
00233     int value = 0;
00234     Eeprom_Write( EEPROM_SYSTEM_NET_CHECK, (uchar*)&value, 4 );
00235   }
00236 
00237   return CONTROLLER_OK;
00238 }

00404 /** 
00405   Read from a TCP socket.
00406   @param socket A pointer to the existing socket.
00407   @param data A pointer to the buffer to read to.
00408   @param length An integer specifying the length in bytes of how much data should be read.
00409   @return An integer: length of data read if successful, zero on failure.
00410   @see lwIP
00411 */
00412 int SocketRead( void* socket, void* data, int length )
00413 {
00414   struct netconn *conn = socket;
00415   struct netbuf *buf;
00416   int buflen;
00417 
00418   if ( conn->readingbuf == NULL )
00419   {
00420     buf = netconn_recv( conn );
00421     if ( buf == NULL )
00422       return 0;
00423 
00424     buflen = netbuf_len( buf );
00425     /* copy the contents of the received buffer into
00426     the supplied memory pointer mem */
00427     netbuf_copy( buf, data, length );
00428 
00429     // Test to see if the buffer is done, or needs to be saved
00430     if ( buflen <= length )
00431     {
00432       netbuf_delete( buf );
00433     }
00434     else
00435     {
00436       conn->readingbuf = buf;
00437       conn->readingoffset = length;
00438       buflen = length;
00439     }
00440   }
00441   else
00442   {
00443     buf = conn->readingbuf;
00444     buflen = netbuf_len( buf ) - conn->readingoffset;
00445     netbuf_copy_partial( buf, data, length, conn->readingoffset );
00446 
00447     if ( buflen <= length )
00448     {
00449       netbuf_delete( buf );
00450       conn->readingbuf = NULL;
00451     }  
00452     else
00453     {
00454       conn->readingoffset += length;
00455       buflen = length;
00456     }
00457   }
00458 
00459   return buflen;
00460 }

00693   emacETHADDR4 = ( serialNumber >> 8 ) & 0xFF;
00694   // Low nibble of the third byte - gives us around 1M serial numbers
