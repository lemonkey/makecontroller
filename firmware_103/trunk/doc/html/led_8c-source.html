00047 /** \defgroup Led
00048   The Led subsystem controls the single green LED on the MAKE Controller Board.
00049   There are two LEDs on the MAKE Controller Board - one green and one red.  The red LED is simply
00050   a power indicator and cannot be controlled by the Controller.  The green LED can be used for
00051   program feedback.  In many MakingThings applications, for example, it is set to blink once a
00052   second, showing the board's "heartbeat" and letting the user know that the board is running.
00053 * \ingroup Controller
00054 * @{
00055 */
00056 
00057 /**
00058   Sets whether the Led subsystem is active.
00059   @param state An integer specifying the active state of the LED system - 1 (on) or 0 (off).
00060   @return Zero on success.
00061 */
00062 int Led_SetActive( int state )
00063 {
00064   if ( state )
00065     return Led_Start( );
00066   else
00067     return Led_Stop( );
00068 }
00069 
00070 /**
00071   Read the active state of the LED system.
00072   @return State - 1/non-zero (on) or 0 (off).
00073 */
00074 int Led_GetActive( )
00075 {
00076   return Led_users > 0;
00077 }
00078 
00079 /** 
00080   Control the LED on the MAKE Controller Board.
00081   @param state An integer specifying the state - on (1) or off (0).
00082   @return 0 on success.
00083 */
00084 int Led_SetState( int state )
00085 {
00086   if ( Led_users == 0 )
00087   {
00088     int status = Led_Start();
00089     if ( status != CONTROLLER_OK )
00090       return status;
00091   }
00092 
00093   return Io_SetValue( LED_IO, !state );
00094 }
00095 
00096 /** 
00097   Read the state of the LED on the MAKE Controller Board.
00098   @return State - 1/non-zero (on) or 0 (off).
00099 */
00100 int Led_GetState( )
00101 {
00102   if ( Led_users == 0 )
00103   {
00104     int status = Led_Start();
00105     if ( status != CONTROLLER_OK )
00106       return status;
00107   }
00108 
00109   return Io_GetValue( LED_IO ) ? 0 : 1;
00110 }
00111 
00112 /** @}
00113 */
00114 
00115 int Led_Start()
00116 {
00117   if ( Led_users == 0 )
00118   {
00119     int status;
00120     status = Io_Start( LED_IO, true );
00121     if ( status != CONTROLLER_OK )
00122       return status;
00123     Io_SetFalse( LED_IO );
00124     Io_PioEnable( LED_IO );
00125     Io_SetOutput( LED_IO );
00126   }
00127   Led_users++;
00128  
00129   return CONTROLLER_OK;
00130 }
00131 
00132 int Led_Stop()
00133 {
00134   if ( Led_users == 1 )
00135   {
00136     Io_Stop( LED_IO );
00137     Io_SetFalse( LED_IO );
00138     Io_PioEnable( LED_IO );
00139     Io_SetOutput( LED_IO );
00140   }
00141   Led_users--;
00142 
00143   return CONTROLLER_OK;
00144 }
