00071 /**

00074   @param duty The duty - (0 - 1023).
00075   @return 0 on success.

00079   if ( index < 0 || index > PWM_COUNT )
00080     return CONTROLLER_ERROR_ILLEGAL_INDEX;

00105 int Pwm_Start( int channel )
00106 {
00107   int status;
00108 
00109   if ( channel < 0 || channel > PWM_COUNT )
00110     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00111 
00112   // Make sure the channel isn't already being used
00113   int c = 1 << channel;
00114   if ( c & Pwm.channels )
00115     return CONTROLLER_ERROR_CANT_LOCK;
00116 
00117   // lock the correct select line 
00118   int io = Pwm_GetChannelIo( channel );
00119   // Try to lock the pin
00120   status = Io_Start( io, true );
00121   if ( status != CONTROLLER_OK )
00122     return status;
00123 
00124   // mark the channel as being used
00125   Pwm.channels |= c;
00126 
00127   // Disable the PIO for the IO Line
00128   Io_PioDisable( io );
00129 
00130   Io_SetPeripheralA( io );
00131 
00132   if ( Pwm.users++ == 0 )
00133   {   
00134     status = Pwm_Init();
00135   }
00136 
00137   // Enable the current channel
00138   AT91C_BASE_PWMC->PWMC_ENA = 1 << channel;
00139 
00140   return CONTROLLER_OK;
00141 }

00143 int Pwm_Stop( int channel )
00144 {
00145   if ( Pwm.users <= 0 )
00146     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00147 
00148   if ( channel < 0 || channel > PWM_COUNT )
00149     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00150 
00151   int c = 1 << channel;
00152   if ( !( c & Pwm.channels ) )
00153     return CONTROLLER_ERROR_NOT_LOCKED;
00154 
00155   // Get the IO
00156   int io = Pwm_GetChannelIo( channel );
00157 
00158   // Set the pin to OFF
00159   Io_SetValue( io, false );
00160   Io_PioEnable( io );
00161   
00162   // release the pin
00163   Io_Stop( io );
00164 
00165   Pwm.channels &= ~c;
00166 
00167   // Disable the PWM's
00168   AT91C_BASE_PWMC->PWMC_DIS = 1 << channel;
00169 
00170   if ( --Pwm.users == 0 )
00171     Pwm_Deinit();
00172 
00173   return CONTROLLER_OK;
00174 }

00176 int Pwm_Init()
00177 {
00178   // Configure PMC by enabling PWM clock
00179   AT91C_BASE_PMC->PMC_PCER = 1 << AT91C_ID_PWMC;
00180 
00181   // Initially they're stopped
00182   AT91C_BASE_PWMC->PWMC_DIS = AT91C_PWMC_CHID0 | AT91C_PWMC_CHID1 | AT91C_PWMC_CHID2 | AT91C_PWMC_CHID3;
00183   
00184   // Set the Clock A divider
00185   AT91C_BASE_PWMC->PWMC_MR = (( 4 << 8 ) | 0x08 );  // MCK selection or'ed with Divider
00186 
00187   // Set the Clock
00188   int i;
00189   for ( i = 0; i < 4; i++ )
00190   {
00191     AT91S_PWMC_CH *pwm = &AT91C_BASE_PWMC->PWMC_CH[ i ];
00192 
00193     pwm->PWMC_CMR =
00194       // AT91C_PWMC_CPRE_MCK |  // Divider Clock ? 
00195          AT91C_PWMC_CPRE_MCKA |    //Divider Clock A
00196       // AT91C_PWMC_CPRE_MCKB;  //Divider Clock B
00197       // AT91C_PWMC_CPD;  // Channel Update Period 
00198          AT91C_PWMC_CPOL; // Channel Polarity Invert
00199       // AT91C_PWMC_CALG ; // Channel Alignment Center
00200 
00201     // Set the Period register (sample size bit fied )
00202     pwm->PWMC_CPRDR = PWM_DUTY_MAX;
00203   
00204     // Set the duty cycle register (output value)
00205     pwm->PWMC_CDTYR = 0;
00206 
00207     // Initialise the Update register write only
00208     pwm->PWMC_CUPDR = 0 ;
00209   }
00210 
00211   return CONTROLLER_OK;
00212 }
