00306 int Osc_ReceivePacket( int channel, char* packet, int length )
00307 {
00308   // Got a packet.  Unpacket.
00309 
00310   switch ( *packet )
00311   {
00312     case '/':
00313       Osc_ReceiveMessage( channel, packet, length );
00314       break;
00315     case '#':
00316       if ( strcmp( packet, "#bundle" ) == 0 )
00317       {
00318         // skip bundle text and timetag
00319         packet += 16;
00320         length -= 16;
00321         while ( length > 0 )
00322         {
00323           // read the length (pretend packet is a pointer to integer)
00324           int messageLength = Osc_EndianSwap( *((int*)packet) );
00325           packet += 4;
00326           length -= 4;
00327           if ( messageLength <= length )
00328             Osc_ReceivePacket( channel, packet, messageLength );
00329           length -= messageLength;
00330           packet += messageLength;
00331         }
00332       }
00333       break;
00334     default:
00335       // Something else?
00336       Osc_CreateMessage( channel, "/error", ",s", "Packet Error" );
00337       break;
00338   }
00339 
00340   return Osc_SendPacket( channel );
00341 }

00343 int Osc_ReceiveMessage( int channel, char* message, int length )
00344 {
00345   // Got a packet.  Unpacket.
00346 
00347   // Confirm it's a message
00348   if ( *message == '/' )
00349   {
00350     // Plain message
00351     char* nextSlash = strchr( message + 1, '/' );
00352     if ( nextSlash != NULL )
00353       *nextSlash = 0;
00354     int i;
00355     int count = 0;
00356     for ( i = 0; i <= Osc.subsystemHighest; i++ )
00357     {
00358       OscSubsystem* sub = &Osc.subsystem[ i ];
00359       if ( Osc_PatternMatch( message + 1, sub->name ) )
00360       {
00361         count++;
00362         if ( nextSlash )
00363           (*sub->receiveMessage)( channel, nextSlash + 1, length - ( nextSlash - message ) - 1 );
00364         else
00365           (*sub->receiveMessage)( channel, 0, 0 );          
00366       }
00367     }
00368     if ( count == 0 )
00369     {
00370       char s[ OSC_SCRATCH_SIZE ];
00371       snprintf( s, OSC_SCRATCH_SIZE, "No Subsystem Match - %s", message + 1 ); 
00372       Osc_CreateMessage( channel, "/error", ",s", s );
00373     }
00374   }
00375   else
00376   {
00377     return CONTROLLER_ERROR_BAD_DATA;
00378   }
00379   return CONTROLLER_OK;
00380 }

00448 // An OSC message has arrived, check it out... although, don't change it - others need
00449 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00450 // messages in it, they will be unpacked elsewhere.  We'll only get 
00451 // individual messages here.
00452 // What is passed is the part after the first subsystem address.
00453 // So "/dipswitch/active 1" would be passed to this routine as "active 1"
00454 // Not literally like this, of course, but as OSC messages
00455 // In order to do anything, we need some extra information from 
00456 // the subsystem like what the names of its properties are, what 
00457 // the getters and setters are, etc.
00458 int Osc_IntReceiverHelper( int channel, char* message, int length, 
00459                            char* subsystemName, 
00460                            int (*propertySet)( int property, int value ),
00461                            int (*propertyGet)( int property ),
00462                            char* propertyNames[] )
00463 {
00464   if ( message == NULL )
00465   {
00466     return Osc_SubsystemError( channel, subsystemName, "No property specified" );
00467   }
00468 
00469   int propertyIndex = Osc_PropertyLookup( propertyNames, message );
00470   if ( propertyIndex == -1 )
00471   {
00472     char s[ OSC_SCRATCH_SIZE ];
00473     snprintf( s, OSC_SCRATCH_SIZE, "Unknown property - %s", message );
00474     return Osc_SubsystemError( channel, subsystemName, s );
00475   }
00476 
00477 /*
00478   int bits;
00479   int number = Osc_NumberMatch( indexCount, message, &bits );
00480   if ( number == -1 && bits == -1 )
00481     return Osc_SubsystemError( channel, subsystemName, "Bad index" );
00482 */
00483   
00484   // Sometime after the address, the data tag begins - this is the description 
00485   // of the data in the rest of the message.  It starts with a comma.  Return
00486   // where it is into 'type'.  If there is no comma, this is bad.
00487   char* type = Osc_FindDataTag( message, length );
00488   if ( type == NULL )
00489     return Osc_SubsystemError( channel, subsystemName, "No type tag" );
00490 
00491   // We can tell if there's data by seeing if the character after the comma
00492   // is a zero or not.
00493   if ( type[ 1 ] == 'i' || type[ 1 ] == 'f' )
00494   {
00495     // If there was int or float data, it was a WRITE.
00496     // So, sort of scanff-like, go get the data.  Here we pass in where the data is
00497     // thanks to the previous routine and then specify what we expect to find there
00498     // in tag terms (i.e. "i", "s", "f" and others).  Finally we pass in a set 
00499     // of pointers to the data types we want to extract.  Osc_ExtractData()
00500     // will rummage around in the message magically grabbing values for you,
00501     // reporting how many it got.  It will convert ints and floats if necessary.
00502     int value;
00503     int count = Osc_ExtractData( type, "i", &value );
00504     if ( count != 1 )
00505       return Osc_SubsystemError( channel, subsystemName, "incorrect data - need an int or float" );
00506   
00507     (*propertySet)( propertyIndex, value );
00508   }
00509   else
00510   {
00511     // No data, then.  I guess it was a read.  The XXXXOsc getters
00512     // take the channel number and use it to call
00513     // Osc_CreateMessage() which adds a new message to the outgoing
00514     // stack
00515     int value = (*propertyGet)( propertyIndex );
00516     char address[ OSC_SCRATCH_SIZE ]; 
00517     snprintf( address, OSC_SCRATCH_SIZE, "/%s/%s", subsystemName, propertyNames[ propertyIndex ] ); 
00518     Osc_CreateMessage( channel, address, ",i", value );
00519   }
00520 
00521   return CONTROLLER_OK;
00522 }

00598 // An OSC message has arrived, check it out... although, don't change it - others need
00599 // to use it.  If the original OSC packet was a BUNDLE - i.e. had many 
00600 // messages in it, they will be unpacked elsewhere.  We'll only get 
00601 // individual messages here.
00602 // What is passed is the part after the first subsystem address.
00603 // So "/appled/0/on 1" would be passed to this routine as "0/on 1"
00604 // Not literally like this, of course, but as OSC messages
00605 // In order to do anything, we need some extra information from 
00606 // the subsystem like what the names of its properties are, what 
00607 // the getters and setters are, etc.
00608 // It's great to have this code in one central place.  It will be easier 
00609 // to effect change from here.o
00610 int Osc_IndexIntReceiverHelper( int channel, char* message, int length, 
00611                                 int indexCount, char* subsystemName, 
00612                                 int (*propertySet)( int index, int property, int value ),
00613                                 int (*propertyGet)( int index, int property ),
00614                                 char* propertyNames[] )
00615 {
00616   // Look for the next slash - being the one that separates the index
00617   // from the property.  Note that this won't go off on a search through the buffer
00618   // since there will soon be a string terminator (i.e. a 0)
00619   char* prop = strchr( message, '/' );
00620   if ( prop == NULL )
00621    return Osc_SubsystemError( channel, subsystemName, "Format - need /subsystem/index/property" );
00622 
00623   // Now that we know where the property is, we can see if we can find it.
00624   // This is a little cheap, since we're also implying that there are no 
00625   // more address terms after the property.  That is, if testing for "speed", while
00626   // "speed" would match, "speed/other_stuff" would not.
00627   int propertyIndex = Osc_PropertyLookup( propertyNames, prop + 1 );
00628   if ( propertyIndex == -1 )
00629   {
00630     char s[ OSC_SCRATCH_SIZE ];
00631     snprintf( s, OSC_SCRATCH_SIZE, "Unknown property - %s", prop + 1 );
00632     return Osc_SubsystemError( channel, subsystemName, s );
00633   }
00634 
00635   // Here's where we try to understand what index we got.  In the world of 
00636   // OSC, this could be a pattern.  So while we could get "0/speed" we could 
00637   // also get "*/speed" or "[0-4]/speed".  This is kind of a drag, but it is 
00638   // quite nice from the user's perspective.
00639   // So to deal with this take a look at the text "0" or "{1,2}" or whatever
00640   // and produce either a nice integer in the simplest case or a set of bits 
00641   // where each bit corresponds to one of the indicies.  Clearly we don't have
00642   // to go crazy, since there are only a small finite number of them.
00643   // Osc_NumberMatch() does the work for us, producing either number = -1 and 
00644   // bits == -1 if there was no index match, or number != -1 for there was a single
00645   // number, or bits != -1 if there were several.
00646 
00647   // note that we tweak the string a bit here to make sure the next '/' is not 
00648   // mixed up with this.  Insert a string terminator.
00649   *prop = 0;
00650 
00651   int bits;
00652   int number = Osc_NumberMatch( indexCount, message, &bits );
00653   if ( number == -1 && bits == -1 )
00654     return Osc_SubsystemError( channel, subsystemName, "Bad index" );
00655   
00656   // We tweaked the '/' before - now put it back
00657   *prop = '/';
00658 
00659   // Sometime after the address, the data tag begins - this is the description 
00660   // of the data in the rest of the message.  It starts with a comma.  Return
00661   // where it is into 'type'.  If there is no comma, this is bad.
00662   char* type = Osc_FindDataTag( message, length );
00663   if ( type == NULL )
00664     return Osc_SubsystemError( channel, subsystemName, "No type tag" );
00665 
00666   // We can tell if there's data by seeing if the character after the comma
00667   // is a zero or not.
00668   if ( type[ 1 ] == 'i' || type[ 1 ] == 'f' )
00669   {
00670     // If there was int or float data, it was a WRITE.
00671     // So, sort of scanff-like, go get the data.  Here we pass in where the data is
00672     // thanks to the previous routine and then specify what we expect to find there
00673     // in tag terms (i.e. "i", "s", "f" and others).  Finally we pass in a set 
00674     // of pointers to the data types we want to extract.  Osc_ExtractData()
00675     // will rummage around in the message magically grabbing values for you,
00676     // reporting how many it got.  It will convert ints and floats if necessary.
00677     int value;
00678     int count = Osc_ExtractData( type, "i", &value );
00679     if ( count != 1 )
00680       return Osc_SubsystemError( channel, subsystemName, "incorrect data - need an int or float" );
00681   
00682     // Now with the data we need to decide what to do with it.
00683     // Is there one or many here?
00684     if ( number != -1 )
00685       (*propertySet)( number, propertyIndex, value );
00686     else
00687     {
00688       int index = 0;
00689       while ( bits > 0 && index < indexCount )
00690       { 
00691         if ( bits & 1 )
00692           (*propertySet)( index, propertyIndex, value );
00693         bits >>= 1;
00694         index++;
00695       }
00696     }
00697   }
00698   else
00699   {
00700     // No data, then.  I guess it was a read.  The XXXXOsc getters
00701     // take the channel number and use it to call
00702     // Osc_CreateMessage() which adds a new message to the outgoing
00703     // stack
00704     if ( number != -1 )
00705     {
00706       int value = (*propertyGet)( number, propertyIndex );
00707       char address[ OSC_SCRATCH_SIZE ]; 
00708       snprintf( address, OSC_SCRATCH_SIZE, "/%s/%d/%s", subsystemName, number, propertyNames[ propertyIndex ] ); 
00709       Osc_CreateMessage( channel, address, ",i", value );
00710     }
00711     else
00712     {
00713       int index = 0;
00714       while ( bits > 0 && index < indexCount )
00715       { 
00716         if ( bits & 1 )
00717         {
00718           int value = (*propertyGet)( index, propertyIndex );
00719           char address[ OSC_SCRATCH_SIZE ]; 
00720           snprintf( address, OSC_SCRATCH_SIZE, "/%s/%d/%s", subsystemName, index, propertyNames[ propertyIndex ] ); 
00721           Osc_CreateMessage( channel, address, ",i", value );
00722         }
00723         bits >>= 1;
00724         index++;
00725       }
00726     }
00727   }
00728 
00729   return CONTROLLER_OK;
00730 }

00732 int Osc_ExtractData( char* buffer, char* format, ... )
00733 {
00734   // Set up to iterate through the arguments
00735   va_list args;
00736   va_start( args, format );
00737   int count = 0;
00738 
00739   // figure out where the data starts
00740   int tagLen = strlen( buffer ) + 1;
00741   int pad = tagLen % 4;
00742   if ( pad != 0 )
00743     tagLen += ( 4 - pad );
00744   char* data = buffer + tagLen;
00745 
00746   // Going to be walking the tag string, the format string and the data
00747   char* fp;
00748   char* tp = buffer + 1; // need to skip the comma ','
00749   bool cont = true;
00750   for ( fp = format; *fp && cont; fp++ )
00751   {
00752     cont = false;
00753     switch ( *fp )
00754     {
00755       case 'i':
00756         if ( *tp == 'i' )
00757         {
00758           *(va_arg( args, int* )) = Osc_ReadInt( data );
00759           data += 4;
00760           count++;
00761           cont = true;
00762         }
00763         if ( *tp == 'f' )
00764         {
00765           *(va_arg( args, int* )) = (int)Osc_ReadFloat( data );
00766           data += 4;
00767           count++;
00768           cont = true;
00769         }
00770         
00771         break;
00772       case 'f':
00773         if ( *tp == 'f' )
00774         {
00775           *(va_arg( args, float* )) = Osc_ReadFloat( data );
00776           data += 4;
00777           count++;
00778           cont = true;
00779         }
00780         if ( *tp == 'i' )
00781         {
00782           *(va_arg( args, float* )) = (float)Osc_ReadInt( data );
00783           data += 4;
00784           count++;
00785           cont = true;
00786         }
00787         break;
00788       case 's':
00789         if ( *tp == 's' )
00790         {
00791           *(va_arg( args, char** )) = data;
00792           int len = strlen( data ) + 1;
00793           int pad = len % 4;
00794           if ( pad != 0 )
00795             len += ( 4 - pad );
00796           data += len;
00797           count++;
00798           cont = true;
00799         }
00800         break;
00801     }
00802     tp++;
00803   }
00804 
00805   //va_end( args );
00806 
00807   return count;
00808 }

00824 /**
00825   Osc_CreateMessageNoLock
00826   Must put the "," as the first format letter
00827   */
00828 int Osc_CreateMessage( int channel, char* address, char* format, ... )
00829 {
00830   if ( address == NULL || format == NULL || *format != ',' )
00831     return CONTROLLER_ERROR_BAD_DATA;
00832 
00833   if ( channel < 0 || channel >= OSC_CHANNEL_COUNT )
00834     return CONTROLLER_ERROR_ILLEGAL_INDEX;
00835 
00836   OscChannel* ch = &Osc.channel[ channel ];
00837 
00838   if ( !ch->running )
00839     return CONTROLLER_ERROR_SUBSYSTEM_INACTIVE;
00840 
00841   if ( channel == OSC_CHANNEL_UDP && ch->replyAddress == 0 )
00842     return CONTROLLER_ERROR_NO_ADDRESS;
00843 
00844   // Check for sender
00845   if ( ch->sendMessage == NULL )
00846     return CONTROLLER_ERROR_RESOURCE_MISSING;
00847 
00848   if ( Osc_Lock( ch ) != CONTROLLER_OK )
00849     return CONTROLLER_ERROR_LOCK_ERROR;
00850 
00851   if ( ch->bufferPointer == NULL )
00852     Osc_ResetChannel( ch );
00853 
00854   // try to send this message - if there's a problem somewhere, 
00855   // send the existing buffer - freeing up space, then try (once) again.
00856   int count = 0;
00857   char *bp;
00858   do
00859   {  
00860     count++;
00861 
00862     char* buffer = ch->bufferPointer;
00863     int length = ch->bufferRemaining;
00864   
00865     bp = buffer;
00866   
00867     // First message in the buffer?
00868     if ( bp == ch->buffer )
00869     {
00870       bp = Osc_CreateBundle( bp, &length, 0, 0 );
00871       if ( bp == NULL )
00872         return CONTROLLER_ERROR_INSUFFICIENT_RESOURCES;
00873     }
00874   
00875     // Make room for the new message
00876     int* lp = (int *)bp;
00877     bp += 4;
00878     length -= 4;
00879 
00880     // remember the start of the message
00881     char* mp = bp;    
00882 
00883     if ( length > 0 )
00884     {      
00885       // Set up to iterate through the arguments
00886       va_list args;
00887       va_start( args, format );
00888     
00889       bp = Osc_CreateMessageInternal( bp, &length, address, format, args ); 
00890 
00891       //va_end( args );
00892     }
00893     else
00894       bp = 0;
00895       
00896     if ( bp != 0 )
00897     {
00898       // Set the size
00899       *lp = Osc_EndianSwap( bp - mp ); 
00900   
00901       ch->bufferPointer = bp;
00902       ch->bufferRemaining = length;
00903       ch->messages++;
00904     }
00905     else
00906     {
00907       Osc_SendPacketInternal( ch );
00908     }
00909   } while ( bp == 0 && count == 1 );
00910 
00911   Osc_Unlock( ch );
00912 
00913   return ( bp != 0 ) ? CONTROLLER_OK : CONTROLLER_ERROR_ILLEGAL_PARAMETER_VALUE;
00914 }

00916 char* Osc_CreateMessageInternal( char* bp, int* length, char* address, char* format, va_list args )
00917 {
00918   // do the address
00919   bp = Osc_WritePaddedString( bp, length, address );
00920   if ( bp == NULL )
00921     return 0;
00922 
00923   // do the type
00924   bp = Osc_WritePaddedString( bp, length, format );
00925   if ( bp == NULL )
00926     return 0;
00927 
00928   // Going to be walking the tag string, the format string and the data
00929   // skip the ',' comma
00930   char* fp;
00931   bool cont = true;
00932   for ( fp = format + 1; *fp && cont; fp++ )
00933   {
00934     switch ( *fp )
00935     {
00936       case 'i':
00937           *length -= 4;
00938           if ( *length >= 0 )
00939           {
00940             int v = va_arg( args, int );
00941             v = Osc_EndianSwap( v );
00942             *((int*)bp) = v;
00943             bp += 4;
00944           }
00945           else 
00946             cont = false;
00947         break;
00948       case 'f':
00949         *length -= 4;
00950         if ( *length >= 0 )
00951         {
00952           int v;
00953           *((float*)&v) = (float)( va_arg( args, double ) ); 
00954           v = Osc_EndianSwap( v );
00955           *((int*)bp) = v;
00956           bp += 4;
00957         }
00958         else 
00959           cont = false;
00960         break;
00961       case 's':
00962       {
00963         char* s = va_arg( args, char* );
00964         bp = Osc_WritePaddedString( bp, length, s );
00965         if ( bp == NULL )
00966           cont = false;
00967         break;
00968       }
00969       default:
00970         cont = false;
00971     }
00972   }
00973 
00974   return ( cont ) ? bp : NULL;
00975 }

01014 int Osc_NumberMatch( int count, char* message, int* bits )
01015 {
01016   int n = 0;
01017   int digits = 0;
01018   while ( isdigit( *message ) )
01019   {
01020     digits++;
01021     n = n * 10 + ( *message++ - '0' );
01022   }
01023 
01024   *bits = -1;
01025   if ( n >= count )
01026     return -1;
01027 
01028   switch ( *message )
01029   {
01030     case '*':
01031     case '?':
01032     case '[':
01033     case '{':
01034     {
01035       int i;
01036       int b = 0;
01037       char s[ 5 ];
01038       for ( i = count - 1; i >=0 ; i-- )
01039       {
01040         b <<= 1;
01041         sprintf( s, "%d", i );
01042         if ( Osc_PatternMatch( message, s ) )
01043           b |= 1;
01044       }
01045       *bits = b;
01046       return -1;
01047     }
01048     default:
01049       if ( digits == 0 )
01050         return -1;
01051       return n;
01052   }
01053 }

01081 char* Osc_WritePaddedString( char* buffer, int* length, char* string )
01082 {
01083   int tagLen = strlen( string ) + 1;
01084   int tagPadLen = tagLen;
01085   int pad = ( tagPadLen ) % 4;
01086   if ( pad != 0 )
01087     tagPadLen += ( 4 - pad );
01088  
01089   *length -= tagPadLen;
01090 
01091   if ( *length >= 0 )
01092   {
01093     strcpy( buffer, string );
01094     int i;
01095     buffer += tagLen;
01096     for ( i = tagLen; i < tagPadLen; i++ ) 
01097       *buffer++ = 0;
01098   }
01099   else
01100     return NULL;
01101 
01102   return buffer;
01103 }

01105 char* Osc_WriteTimetag( char* buffer, int* length, int a, int b )
01106 {
01107   if ( *length < 8 )
01108     return NULL;
01109 
01110   *((int*)buffer) = Osc_EndianSwap( a );
01111   buffer += 4;
01112   *((int*)buffer) = Osc_EndianSwap( b );
01113   buffer += 4;
01114   *length -= 8;
01115 
01116   return buffer;
01117 }

01119 int Osc_EndianSwap( int a )
01120 {
01121   return ( ( a & 0x000000FF ) << 24 ) |
01122          ( ( a & 0x0000FF00 ) << 8 )  | 
01123          ( ( a & 0x00FF0000 ) >> 8 )  | 
01124          ( ( a & 0xFF000000 ) >> 24 );
01125 
01126 }
