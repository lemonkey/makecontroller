00043 /**
00044   Sets whether the DIP Switch is active.
00045   @param state An integer specifying the state of the DIP Switch - 1 (on) or 0 (off).
00046   @return Zero on success.

00048 int DipSwitch_SetActive( int state )

00050   if ( state )
00051     return DipSwitch_Start(  );

00065 /** 
00066   Read the current configuration of the on-board DIP switch.
00067   @return An integer address corresponding to one of the 256 possible configurations.
00068   @see SPI
00069 */
00070 int DipSwitch_GetValue( )
00071 {
00072   if ( DipSwitch_users == 0 )
00073   {
00074     int status = DipSwitch_Start();
00075     if ( status != CONTROLLER_OK )
00076       return status;
00077   }
00078 
00079   Spi_Lock();
00080 
00081   unsigned char c[ 2 ];
00082 
00083   c[ 0 ] = 0xFE;
00084   c[ 1 ] = 0xFF;
00085   
00086   Spi_ReadWriteBlock( DIPSWITCH_DEVICE, c, 2 );
00087 
00088   Spi_Unlock();
00089 
00090   int r = ( c[ 1 ] & 0x01 ) << 8 | 
00091           ( c[ 0 ] & 0x01 ) << 7 | 
00092             c[ 1 ] >> 1; 
00093 
00094   return r;
00095 }

00100 int DipSwitch_Start()
00101 {
00102   int status;
00103   if ( DipSwitch_users++ == 0 )
00104   {
00105     // Start the SPI
00106     status = Spi_Start( DIPSWITCH_DEVICE );
00107     if ( status != CONTROLLER_OK )
00108     {
00109       DipSwitch_users--;
00110       return status;
00111     }
00112     
00113     // Configure the channel
00114     status = Spi_Configure( DIPSWITCH_DEVICE, 8, 4, 0, 1 );
00115     if ( status != CONTROLLER_OK )
00116     {
00117       // Undo all the setup.  Sigh.
00118       DipSwitch_Stop();
00119       return status;
00120     }
00121   }
00122   return CONTROLLER_OK;
00123 }
00124 
00125 int DipSwitch_Stop()
00126 {
00127   if ( DipSwitch_users <= 0 )
00128     return CONTROLLER_ERROR_TOO_MANY_STOPS;
00129   
00130   if ( --DipSwitch_users == 0 )
00131     Spi_Stop( DIPSWITCH_DEVICE );
00132 
00133   return CONTROLLER_OK;
00134 }

00138 /** \defgroup DIPSwitchOSC DIP Switch - OSC
00139   Read the Application Board's DIP Switch via OSC.
00140   \ingroup OSC
00141   
00142   \section devices Devices
00143   There's a single DIP Switch the Make Application Board. Because there's
00144   only one device, a device index is not included in any OSC messages to the
00145   DIP Switch.
00146   
00147   \section properties Properties
00148   The DIP Switch has two properties - 'value' and 'active'.
00149 
00150   \par Value
00151   The 'value' property corresponds to current configuration of the DIP Switch.
00152   The DIP switch is 8-position, which means that it will send a value from 0-255
00153   depending on the orientation of each of the switches.  Convert the binary number
00154   represented by the switches to decimal to arrive at this value.
00155   \par
00156   Because you can only ever \em read the value of an input, you'll never
00157   want to include an argument at the end of your OSC message to read the value.\n
00158   To read from the DIP Switch, send the message
00159   \verbatim /dipswitch/value \endverbatim
00160   Note the lack of a device index, and the lack of an argument value.
00161   
00162   \par Active
00163   The 'active' property corresponds to the active state of the DIP Switch.
00164   If the DIP Switch is set to be active, no other tasks will be able to
00165   use its I/O lines.  This might be useful if the Controller is being used with
00166   another piece of hardware.  If you're not seeing appropriate
00167   responses to your messages to the DIP Switch, check the whether it's 
00168   locked by sending a message like
00169   \verbatim /dipswitch/active \endverbatim
00170   \par
00171   You can set the active flag by sending
00172   \verbatim /dipswitch/active 1 \endverbatim
00173 */
