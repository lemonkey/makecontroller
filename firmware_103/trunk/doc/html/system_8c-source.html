00035 /** \defgroup System
00036     The System subsystem monitors and controls several aspects of the system. 
00037     From OSC this subsystem can be addressed as "system".  It has the following 
00038     properties:
00039     \li active (R/W) - not currently meaningful
00040     \li freememory (R) - returns the remaining free memory size
00041     \li samba (W) - requests the board erase itself and return to SAMBA state
00042     \li reset (W) - requests the board reboot
00043     \li serialnumber (R/W) - permits the reading and writing of the serial number 
00044     \li buildnumber (R) - permits the reading of the board's build number
00045 
00046 * \ingroup Controller

00124 /**
00125   Returns the board to SAMBA mode, erasing all of FLASH.
00126   Leaves the board in a non-deterministic state awaiting a power cycle or reset.
00127   @return CONTROLLER_OK if OK.
00128 */
00129 int System_SetSamba( int sure )
00130 {
00131   if ( sure )
00132   {
00133     // Wait for End Of Programming
00134     while( !(AT91C_BASE_MC->MC_FSR & AT91C_MC_FRDY) );
00135     
00136     // Send Boot From Flash Command
00137     AT91C_BASE_MC->MC_FCR = (AT91C_MC_FCMD_CLR_GP_NVM |  (( (2) << 8) & AT91C_MC_PAGEN) | (0x5A << 24));
00138 
00139     AT91C_BASE_RSTC->RSTC_RCR = ( AT91C_RSTC_EXTRST | AT91C_RSTC_PROCRST | AT91C_RSTC_PERRST | (0xA5 << 24 ) );
00140 
00141     // Wait for End Of Programming
00142     while( !(AT91C_BASE_MC->MC_FSR & AT91C_MC_FRDY) );
00143   
00144     if ( AT91C_BASE_MC->MC_FSR & AT91C_MC_PROGE )
00145       return 0;
00146   
00147     if ( AT91C_BASE_MC->MC_FSR & AT91C_MC_GPNVM2 )
00148       return 0;
00149   }
00150 
00151   // Never will do this
00152   return 1;
00153 }
