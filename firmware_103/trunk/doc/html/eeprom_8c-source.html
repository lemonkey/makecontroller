00061 /** \defgroup Eeprom
00062   Eeprom allows for the persistent storage of 32k bytes data. 
00063   The last 1k (1024) bytes of this space are reserved for system
00064   use storing things like the board's IP address, serial number, build version, etc.
00065 
00066   Eeprom is activated by the first use of the Eeprom_Write() or Eeprom_Read()
00067   functions.  Whether its running or not can be determined by calling the Eeprom_GetActive() 
00068   function.  It can be turned off by calling Eeprom_SetActive( ) passing in false or 0.
00069 
00070   Internally, Eeprom relies on \ref Spi, so activating Eeprom also activates \ref Spi.
00071 
00072   \ingroup Controller
00073   @{
00074 */
00075 
00076 /**
00077   Set the active state of the EEPROM subsystem.  This is automatically set to 
00078   true by any call to Eeprom_Write or Eeprom_Read.
00079   @param state An integer specifying the active state - 1 (on) or 0 (off).
00080   @return CONTROLLER_OK (=0) on success.
00081 */
00082 int Eeprom_SetActive( int state )
00083 {
00084   if ( state )
00085     return Eeprom_Start();
00086   else
00087     return Eeprom_Stop();
00088 }

00099 /** 
00100   Write data to the EEPROM.
00101   @param address An integer specifying the address.
00102   @param buffer A pointer to the buffer to write from.
00103   @param count An integer specifying the number of bytes to write.
00104   @return status.
00105 */
00106 int Eeprom_Write( int address, uchar* buffer, int count )
00107 {
00108   if ( address < 0 || address > EEPROM_SIZE )
00109     return CONTROLLER_ERROR_BAD_ADDRESS;
00110 
00111   if ( Eeprom_users == 0 )
00112   {
00113     int status = Eeprom_Start();
00114     if ( status != CONTROLLER_OK )
00115       return status;
00116   }
00117 
00118   Spi_Lock();
00119 
00120   Eeprom_Ready( );
00121 
00122   Eeprom_WriteEnable();    
00123 
00124   uchar c[ count + 4 ];
00125 
00126   c[ 0 ] = EEPROM_INSTRUCTION_WRITE;
00127   c[ 1 ] = (unsigned char) address >> 8;
00128   c[ 2 ] = (unsigned char)( address & 0xFF );
00129   c[ 3 ] = 0;
00130 
00131   int i;
00132   for ( i = 0; i < count; i++ )
00133   {
00134     c[ i + 3 ] = buffer[ i ];
00135   }
00136   
00137   Spi_ReadWriteBlock( EEPROM_DEVICE, c, 3 + count );
00138 
00139   Spi_Unlock();
00140 
00141   return CONTROLLER_OK;
00142 }

00200 int Eeprom_Start()
00201 {
00202   int status;
00203 
00204   if ( Eeprom_users++ == 0 )
00205   {    
00206     // Attempt to open the SPI device
00207     status = Spi_Start( EEPROM_DEVICE );
00208     if ( status != CONTROLLER_OK )
00209     {
00210       Eeprom_users--;
00211       return status;
00212     }
00213   
00214     // We're here means that we got the Spi open, now set it up
00215     status = Spi_Configure( EEPROM_DEVICE, 8, 4, 0, 1 );
00216     if ( status != CONTROLLER_OK )
00217     {
00218       // Undo all the setup.  Sigh.
00219       Eeprom_Stop();
00220       return status;
00221     }
00222   }
00223   
00224   return CONTROLLER_OK;
00225 }

00227 int Eeprom_Stop()
00228 {
00229   if ( Eeprom_users <= 0 )
00230     return CONTROLLER_ERROR_NOT_LOCKED;
00231 
00232   if ( --Eeprom_users == 0 )
00233     Spi_Stop( EEPROM_DEVICE );
00234 
00235   return CONTROLLER_OK;
00236 }

00265 void Eeprom_Ready( void )
00266 {
00267   int status;
00268   do
00269   {
00270     unsigned char c[ 2 ];
00271 
00272     c[ 0 ] = EEPROM_INSTRUCTION_RDSR;
00273     c[ 1 ] = 0;
00274   
00275     Spi_ReadWriteBlock( EEPROM_DEVICE, c, 2 );
00276 
00277     status = c[ 1 ] != 0xFF;
00278 
00279   } while ( !status );
00280 }
